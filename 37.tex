ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ وَٱلصَّـٰٓفَّٰتِ صَفًّۭا ﴿١﴾}\\
2.\  & \mytextarabic{فَٱلزَّٰجِرَٰتِ زَجْرًۭا ﴿٢﴾}\\
3.\  & \mytextarabic{فَٱلتَّٰلِيَـٰتِ ذِكْرًا ﴿٣﴾}\\
4.\  & \mytextarabic{إِنَّ إِلَـٰهَكُمْ لَوَٟحِدٌۭ ﴿٤﴾}\\
5.\  & \mytextarabic{رَّبُّ ٱلسَّمَـٰوَٟتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا وَرَبُّ ٱلْمَشَـٰرِقِ ﴿٥﴾}\\
6.\  & \mytextarabic{إِنَّا زَيَّنَّا ٱلسَّمَآءَ ٱلدُّنْيَا بِزِينَةٍ ٱلْكَوَاكِبِ ﴿٦﴾}\\
7.\  & \mytextarabic{وَحِفْظًۭا مِّن كُلِّ شَيْطَٰنٍۢ مَّارِدٍۢ ﴿٧﴾}\\
8.\  & \mytextarabic{لَّا يَسَّمَّعُونَ إِلَى ٱلْمَلَإِ ٱلْأَعْلَىٰ وَيُقْذَفُونَ مِن كُلِّ جَانِبٍۢ ﴿٨﴾}\\
9.\  & \mytextarabic{دُحُورًۭا ۖ وَلَهُمْ عَذَابٌۭ وَاصِبٌ ﴿٩﴾}\\
10.\  & \mytextarabic{إِلَّا مَنْ خَطِفَ ٱلْخَطْفَةَ فَأَتْبَعَهُۥ شِهَابٌۭ ثَاقِبٌۭ ﴿١٠﴾}\\
11.\  & \mytextarabic{فَٱسْتَفْتِهِمْ أَهُمْ أَشَدُّ خَلْقًا أَم مَّنْ خَلَقْنَآ ۚ إِنَّا خَلَقْنَـٰهُم مِّن طِينٍۢ لَّازِبٍۭ ﴿١١﴾}\\
12.\  & \mytextarabic{بَلْ عَجِبْتَ وَيَسْخَرُونَ ﴿١٢﴾}\\
13.\  & \mytextarabic{وَإِذَا ذُكِّرُوا۟ لَا يَذْكُرُونَ ﴿١٣﴾}\\
14.\  & \mytextarabic{وَإِذَا رَأَوْا۟ ءَايَةًۭ يَسْتَسْخِرُونَ ﴿١٤﴾}\\
15.\  & \mytextarabic{وَقَالُوٓا۟ إِنْ هَـٰذَآ إِلَّا سِحْرٌۭ مُّبِينٌ ﴿١٥﴾}\\
16.\  & \mytextarabic{أَءِذَا مِتْنَا وَكُنَّا تُرَابًۭا وَعِظَـٰمًا أَءِنَّا لَمَبْعُوثُونَ ﴿١٦﴾}\\
17.\  & \mytextarabic{أَوَءَابَآؤُنَا ٱلْأَوَّلُونَ ﴿١٧﴾}\\
18.\  & \mytextarabic{قُلْ نَعَمْ وَأَنتُمْ دَٟخِرُونَ ﴿١٨﴾}\\
19.\  & \mytextarabic{فَإِنَّمَا هِىَ زَجْرَةٌۭ وَٟحِدَةٌۭ فَإِذَا هُمْ يَنظُرُونَ ﴿١٩﴾}\\
20.\  & \mytextarabic{وَقَالُوا۟ يَـٰوَيْلَنَا هَـٰذَا يَوْمُ ٱلدِّينِ ﴿٢٠﴾}\\
21.\  & \mytextarabic{هَـٰذَا يَوْمُ ٱلْفَصْلِ ٱلَّذِى كُنتُم بِهِۦ تُكَذِّبُونَ ﴿٢١﴾}\\
22.\  & \mytextarabic{۞ ٱحْشُرُوا۟ ٱلَّذِينَ ظَلَمُوا۟ وَأَزْوَٟجَهُمْ وَمَا كَانُوا۟ يَعْبُدُونَ ﴿٢٢﴾}\\
23.\  & \mytextarabic{مِن دُونِ ٱللَّهِ فَٱهْدُوهُمْ إِلَىٰ صِرَٰطِ ٱلْجَحِيمِ ﴿٢٣﴾}\\
24.\  & \mytextarabic{وَقِفُوهُمْ ۖ إِنَّهُم مَّسْـُٔولُونَ ﴿٢٤﴾}\\
25.\  & \mytextarabic{مَا لَكُمْ لَا تَنَاصَرُونَ ﴿٢٥﴾}\\
26.\  & \mytextarabic{بَلْ هُمُ ٱلْيَوْمَ مُسْتَسْلِمُونَ ﴿٢٦﴾}\\
27.\  & \mytextarabic{وَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٍۢ يَتَسَآءَلُونَ ﴿٢٧﴾}\\
28.\  & \mytextarabic{قَالُوٓا۟ إِنَّكُمْ كُنتُمْ تَأْتُونَنَا عَنِ ٱلْيَمِينِ ﴿٢٨﴾}\\
29.\  & \mytextarabic{قَالُوا۟ بَل لَّمْ تَكُونُوا۟ مُؤْمِنِينَ ﴿٢٩﴾}\\
30.\  & \mytextarabic{وَمَا كَانَ لَنَا عَلَيْكُم مِّن سُلْطَٰنٍۭ ۖ بَلْ كُنتُمْ قَوْمًۭا طَٰغِينَ ﴿٣٠﴾}\\
31.\  & \mytextarabic{فَحَقَّ عَلَيْنَا قَوْلُ رَبِّنَآ ۖ إِنَّا لَذَآئِقُونَ ﴿٣١﴾}\\
32.\  & \mytextarabic{فَأَغْوَيْنَـٰكُمْ إِنَّا كُنَّا غَٰوِينَ ﴿٣٢﴾}\\
33.\  & \mytextarabic{فَإِنَّهُمْ يَوْمَئِذٍۢ فِى ٱلْعَذَابِ مُشْتَرِكُونَ ﴿٣٣﴾}\\
34.\  & \mytextarabic{إِنَّا كَذَٟلِكَ نَفْعَلُ بِٱلْمُجْرِمِينَ ﴿٣٤﴾}\\
35.\  & \mytextarabic{إِنَّهُمْ كَانُوٓا۟ إِذَا قِيلَ لَهُمْ لَآ إِلَـٰهَ إِلَّا ٱللَّهُ يَسْتَكْبِرُونَ ﴿٣٥﴾}\\
36.\  & \mytextarabic{وَيَقُولُونَ أَئِنَّا لَتَارِكُوٓا۟ ءَالِهَتِنَا لِشَاعِرٍۢ مَّجْنُونٍۭ ﴿٣٦﴾}\\
37.\  & \mytextarabic{بَلْ جَآءَ بِٱلْحَقِّ وَصَدَّقَ ٱلْمُرْسَلِينَ ﴿٣٧﴾}\\
38.\  & \mytextarabic{إِنَّكُمْ لَذَآئِقُوا۟ ٱلْعَذَابِ ٱلْأَلِيمِ ﴿٣٨﴾}\\
39.\  & \mytextarabic{وَمَا تُجْزَوْنَ إِلَّا مَا كُنتُمْ تَعْمَلُونَ ﴿٣٩﴾}\\
40.\  & \mytextarabic{إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿٤٠﴾}\\
41.\  & \mytextarabic{أُو۟لَـٰٓئِكَ لَهُمْ رِزْقٌۭ مَّعْلُومٌۭ ﴿٤١﴾}\\
42.\  & \mytextarabic{فَوَٟكِهُ ۖ وَهُم مُّكْرَمُونَ ﴿٤٢﴾}\\
43.\  & \mytextarabic{فِى جَنَّـٰتِ ٱلنَّعِيمِ ﴿٤٣﴾}\\
44.\  & \mytextarabic{عَلَىٰ سُرُرٍۢ مُّتَقَـٰبِلِينَ ﴿٤٤﴾}\\
45.\  & \mytextarabic{يُطَافُ عَلَيْهِم بِكَأْسٍۢ مِّن مَّعِينٍۭ ﴿٤٥﴾}\\
46.\  & \mytextarabic{بَيْضَآءَ لَذَّةٍۢ لِّلشَّـٰرِبِينَ ﴿٤٦﴾}\\
47.\  & \mytextarabic{لَا فِيهَا غَوْلٌۭ وَلَا هُمْ عَنْهَا يُنزَفُونَ ﴿٤٧﴾}\\
48.\  & \mytextarabic{وَعِندَهُمْ قَـٰصِرَٰتُ ٱلطَّرْفِ عِينٌۭ ﴿٤٨﴾}\\
49.\  & \mytextarabic{كَأَنَّهُنَّ بَيْضٌۭ مَّكْنُونٌۭ ﴿٤٩﴾}\\
50.\  & \mytextarabic{فَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٍۢ يَتَسَآءَلُونَ ﴿٥٠﴾}\\
51.\  & \mytextarabic{قَالَ قَآئِلٌۭ مِّنْهُمْ إِنِّى كَانَ لِى قَرِينٌۭ ﴿٥١﴾}\\
52.\  & \mytextarabic{يَقُولُ أَءِنَّكَ لَمِنَ ٱلْمُصَدِّقِينَ ﴿٥٢﴾}\\
53.\  & \mytextarabic{أَءِذَا مِتْنَا وَكُنَّا تُرَابًۭا وَعِظَـٰمًا أَءِنَّا لَمَدِينُونَ ﴿٥٣﴾}\\
54.\  & \mytextarabic{قَالَ هَلْ أَنتُم مُّطَّلِعُونَ ﴿٥٤﴾}\\
55.\  & \mytextarabic{فَٱطَّلَعَ فَرَءَاهُ فِى سَوَآءِ ٱلْجَحِيمِ ﴿٥٥﴾}\\
56.\  & \mytextarabic{قَالَ تَٱللَّهِ إِن كِدتَّ لَتُرْدِينِ ﴿٥٦﴾}\\
57.\  & \mytextarabic{وَلَوْلَا نِعْمَةُ رَبِّى لَكُنتُ مِنَ ٱلْمُحْضَرِينَ ﴿٥٧﴾}\\
58.\  & \mytextarabic{أَفَمَا نَحْنُ بِمَيِّتِينَ ﴿٥٨﴾}\\
59.\  & \mytextarabic{إِلَّا مَوْتَتَنَا ٱلْأُولَىٰ وَمَا نَحْنُ بِمُعَذَّبِينَ ﴿٥٩﴾}\\
60.\  & \mytextarabic{إِنَّ هَـٰذَا لَهُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿٦٠﴾}\\
61.\  & \mytextarabic{لِمِثْلِ هَـٰذَا فَلْيَعْمَلِ ٱلْعَـٰمِلُونَ ﴿٦١﴾}\\
62.\  & \mytextarabic{أَذَٟلِكَ خَيْرٌۭ نُّزُلًا أَمْ شَجَرَةُ ٱلزَّقُّومِ ﴿٦٢﴾}\\
63.\  & \mytextarabic{إِنَّا جَعَلْنَـٰهَا فِتْنَةًۭ لِّلظَّـٰلِمِينَ ﴿٦٣﴾}\\
64.\  & \mytextarabic{إِنَّهَا شَجَرَةٌۭ تَخْرُجُ فِىٓ أَصْلِ ٱلْجَحِيمِ ﴿٦٤﴾}\\
65.\  & \mytextarabic{طَلْعُهَا كَأَنَّهُۥ رُءُوسُ ٱلشَّيَـٰطِينِ ﴿٦٥﴾}\\
66.\  & \mytextarabic{فَإِنَّهُمْ لَءَاكِلُونَ مِنْهَا فَمَالِـُٔونَ مِنْهَا ٱلْبُطُونَ ﴿٦٦﴾}\\
67.\  & \mytextarabic{ثُمَّ إِنَّ لَهُمْ عَلَيْهَا لَشَوْبًۭا مِّنْ حَمِيمٍۢ ﴿٦٧﴾}\\
68.\  & \mytextarabic{ثُمَّ إِنَّ مَرْجِعَهُمْ لَإِلَى ٱلْجَحِيمِ ﴿٦٨﴾}\\
69.\  & \mytextarabic{إِنَّهُمْ أَلْفَوْا۟ ءَابَآءَهُمْ ضَآلِّينَ ﴿٦٩﴾}\\
70.\  & \mytextarabic{فَهُمْ عَلَىٰٓ ءَاثَـٰرِهِمْ يُهْرَعُونَ ﴿٧٠﴾}\\
71.\  & \mytextarabic{وَلَقَدْ ضَلَّ قَبْلَهُمْ أَكْثَرُ ٱلْأَوَّلِينَ ﴿٧١﴾}\\
72.\  & \mytextarabic{وَلَقَدْ أَرْسَلْنَا فِيهِم مُّنذِرِينَ ﴿٧٢﴾}\\
73.\  & \mytextarabic{فَٱنظُرْ كَيْفَ كَانَ عَـٰقِبَةُ ٱلْمُنذَرِينَ ﴿٧٣﴾}\\
74.\  & \mytextarabic{إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿٧٤﴾}\\
75.\  & \mytextarabic{وَلَقَدْ نَادَىٰنَا نُوحٌۭ فَلَنِعْمَ ٱلْمُجِيبُونَ ﴿٧٥﴾}\\
76.\  & \mytextarabic{وَنَجَّيْنَـٰهُ وَأَهْلَهُۥ مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿٧٦﴾}\\
77.\  & \mytextarabic{وَجَعَلْنَا ذُرِّيَّتَهُۥ هُمُ ٱلْبَاقِينَ ﴿٧٧﴾}\\
78.\  & \mytextarabic{وَتَرَكْنَا عَلَيْهِ فِى ٱلْءَاخِرِينَ ﴿٧٨﴾}\\
79.\  & \mytextarabic{سَلَـٰمٌ عَلَىٰ نُوحٍۢ فِى ٱلْعَـٰلَمِينَ ﴿٧٩﴾}\\
80.\  & \mytextarabic{إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿٨٠﴾}\\
81.\  & \mytextarabic{إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿٨١﴾}\\
82.\  & \mytextarabic{ثُمَّ أَغْرَقْنَا ٱلْءَاخَرِينَ ﴿٨٢﴾}\\
83.\  & \mytextarabic{۞ وَإِنَّ مِن شِيعَتِهِۦ لَإِبْرَٰهِيمَ ﴿٨٣﴾}\\
84.\  & \mytextarabic{إِذْ جَآءَ رَبَّهُۥ بِقَلْبٍۢ سَلِيمٍ ﴿٨٤﴾}\\
85.\  & \mytextarabic{إِذْ قَالَ لِأَبِيهِ وَقَوْمِهِۦ مَاذَا تَعْبُدُونَ ﴿٨٥﴾}\\
86.\  & \mytextarabic{أَئِفْكًا ءَالِهَةًۭ دُونَ ٱللَّهِ تُرِيدُونَ ﴿٨٦﴾}\\
87.\  & \mytextarabic{فَمَا ظَنُّكُم بِرَبِّ ٱلْعَـٰلَمِينَ ﴿٨٧﴾}\\
88.\  & \mytextarabic{فَنَظَرَ نَظْرَةًۭ فِى ٱلنُّجُومِ ﴿٨٨﴾}\\
89.\  & \mytextarabic{فَقَالَ إِنِّى سَقِيمٌۭ ﴿٨٩﴾}\\
90.\  & \mytextarabic{فَتَوَلَّوْا۟ عَنْهُ مُدْبِرِينَ ﴿٩٠﴾}\\
91.\  & \mytextarabic{فَرَاغَ إِلَىٰٓ ءَالِهَتِهِمْ فَقَالَ أَلَا تَأْكُلُونَ ﴿٩١﴾}\\
92.\  & \mytextarabic{مَا لَكُمْ لَا تَنطِقُونَ ﴿٩٢﴾}\\
93.\  & \mytextarabic{فَرَاغَ عَلَيْهِمْ ضَرْبًۢا بِٱلْيَمِينِ ﴿٩٣﴾}\\
94.\  & \mytextarabic{فَأَقْبَلُوٓا۟ إِلَيْهِ يَزِفُّونَ ﴿٩٤﴾}\\
95.\  & \mytextarabic{قَالَ أَتَعْبُدُونَ مَا تَنْحِتُونَ ﴿٩٥﴾}\\
96.\  & \mytextarabic{وَٱللَّهُ خَلَقَكُمْ وَمَا تَعْمَلُونَ ﴿٩٦﴾}\\
97.\  & \mytextarabic{قَالُوا۟ ٱبْنُوا۟ لَهُۥ بُنْيَـٰنًۭا فَأَلْقُوهُ فِى ٱلْجَحِيمِ ﴿٩٧﴾}\\
98.\  & \mytextarabic{فَأَرَادُوا۟ بِهِۦ كَيْدًۭا فَجَعَلْنَـٰهُمُ ٱلْأَسْفَلِينَ ﴿٩٨﴾}\\
99.\  & \mytextarabic{وَقَالَ إِنِّى ذَاهِبٌ إِلَىٰ رَبِّى سَيَهْدِينِ ﴿٩٩﴾}\\
100.\  & \mytextarabic{رَبِّ هَبْ لِى مِنَ ٱلصَّـٰلِحِينَ ﴿١٠٠﴾}\\
101.\  & \mytextarabic{فَبَشَّرْنَـٰهُ بِغُلَـٰمٍ حَلِيمٍۢ ﴿١٠١﴾}\\
102.\  & \mytextarabic{فَلَمَّا بَلَغَ مَعَهُ ٱلسَّعْىَ قَالَ يَـٰبُنَىَّ إِنِّىٓ أَرَىٰ فِى ٱلْمَنَامِ أَنِّىٓ أَذْبَحُكَ فَٱنظُرْ مَاذَا تَرَىٰ ۚ قَالَ يَـٰٓأَبَتِ ٱفْعَلْ مَا تُؤْمَرُ ۖ سَتَجِدُنِىٓ إِن شَآءَ ٱللَّهُ مِنَ ٱلصَّـٰبِرِينَ ﴿١٠٢﴾}\\
103.\  & \mytextarabic{فَلَمَّآ أَسْلَمَا وَتَلَّهُۥ لِلْجَبِينِ ﴿١٠٣﴾}\\
104.\  & \mytextarabic{وَنَـٰدَيْنَـٰهُ أَن يَـٰٓإِبْرَٰهِيمُ ﴿١٠٤﴾}\\
105.\  & \mytextarabic{قَدْ صَدَّقْتَ ٱلرُّءْيَآ ۚ إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿١٠٥﴾}\\
106.\  & \mytextarabic{إِنَّ هَـٰذَا لَهُوَ ٱلْبَلَـٰٓؤُا۟ ٱلْمُبِينُ ﴿١٠٦﴾}\\
107.\  & \mytextarabic{وَفَدَيْنَـٰهُ بِذِبْحٍ عَظِيمٍۢ ﴿١٠٧﴾}\\
108.\  & \mytextarabic{وَتَرَكْنَا عَلَيْهِ فِى ٱلْءَاخِرِينَ ﴿١٠٨﴾}\\
109.\  & \mytextarabic{سَلَـٰمٌ عَلَىٰٓ إِبْرَٰهِيمَ ﴿١٠٩﴾}\\
110.\  & \mytextarabic{كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿١١٠﴾}\\
111.\  & \mytextarabic{إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿١١١﴾}\\
112.\  & \mytextarabic{وَبَشَّرْنَـٰهُ بِإِسْحَـٰقَ نَبِيًّۭا مِّنَ ٱلصَّـٰلِحِينَ ﴿١١٢﴾}\\
113.\  & \mytextarabic{وَبَٰرَكْنَا عَلَيْهِ وَعَلَىٰٓ إِسْحَـٰقَ ۚ وَمِن ذُرِّيَّتِهِمَا مُحْسِنٌۭ وَظَالِمٌۭ لِّنَفْسِهِۦ مُبِينٌۭ ﴿١١٣﴾}\\
114.\  & \mytextarabic{وَلَقَدْ مَنَنَّا عَلَىٰ مُوسَىٰ وَهَـٰرُونَ ﴿١١٤﴾}\\
115.\  & \mytextarabic{وَنَجَّيْنَـٰهُمَا وَقَوْمَهُمَا مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿١١٥﴾}\\
116.\  & \mytextarabic{وَنَصَرْنَـٰهُمْ فَكَانُوا۟ هُمُ ٱلْغَٰلِبِينَ ﴿١١٦﴾}\\
117.\  & \mytextarabic{وَءَاتَيْنَـٰهُمَا ٱلْكِتَـٰبَ ٱلْمُسْتَبِينَ ﴿١١٧﴾}\\
118.\  & \mytextarabic{وَهَدَيْنَـٰهُمَا ٱلصِّرَٰطَ ٱلْمُسْتَقِيمَ ﴿١١٨﴾}\\
119.\  & \mytextarabic{وَتَرَكْنَا عَلَيْهِمَا فِى ٱلْءَاخِرِينَ ﴿١١٩﴾}\\
120.\  & \mytextarabic{سَلَـٰمٌ عَلَىٰ مُوسَىٰ وَهَـٰرُونَ ﴿١٢٠﴾}\\
121.\  & \mytextarabic{إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿١٢١﴾}\\
122.\  & \mytextarabic{إِنَّهُمَا مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿١٢٢﴾}\\
123.\  & \mytextarabic{وَإِنَّ إِلْيَاسَ لَمِنَ ٱلْمُرْسَلِينَ ﴿١٢٣﴾}\\
124.\  & \mytextarabic{إِذْ قَالَ لِقَوْمِهِۦٓ أَلَا تَتَّقُونَ ﴿١٢٤﴾}\\
125.\  & \mytextarabic{أَتَدْعُونَ بَعْلًۭا وَتَذَرُونَ أَحْسَنَ ٱلْخَـٰلِقِينَ ﴿١٢٥﴾}\\
126.\  & \mytextarabic{ٱللَّهَ رَبَّكُمْ وَرَبَّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿١٢٦﴾}\\
127.\  & \mytextarabic{فَكَذَّبُوهُ فَإِنَّهُمْ لَمُحْضَرُونَ ﴿١٢٧﴾}\\
128.\  & \mytextarabic{إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿١٢٨﴾}\\
129.\  & \mytextarabic{وَتَرَكْنَا عَلَيْهِ فِى ٱلْءَاخِرِينَ ﴿١٢٩﴾}\\
130.\  & \mytextarabic{سَلَـٰمٌ عَلَىٰٓ إِلْ يَاسِينَ ﴿١٣٠﴾}\\
131.\  & \mytextarabic{إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿١٣١﴾}\\
132.\  & \mytextarabic{إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿١٣٢﴾}\\
133.\  & \mytextarabic{وَإِنَّ لُوطًۭا لَّمِنَ ٱلْمُرْسَلِينَ ﴿١٣٣﴾}\\
134.\  & \mytextarabic{إِذْ نَجَّيْنَـٰهُ وَأَهْلَهُۥٓ أَجْمَعِينَ ﴿١٣٤﴾}\\
135.\  & \mytextarabic{إِلَّا عَجُوزًۭا فِى ٱلْغَٰبِرِينَ ﴿١٣٥﴾}\\
136.\  & \mytextarabic{ثُمَّ دَمَّرْنَا ٱلْءَاخَرِينَ ﴿١٣٦﴾}\\
137.\  & \mytextarabic{وَإِنَّكُمْ لَتَمُرُّونَ عَلَيْهِم مُّصْبِحِينَ ﴿١٣٧﴾}\\
138.\  & \mytextarabic{وَبِٱلَّيْلِ ۗ أَفَلَا تَعْقِلُونَ ﴿١٣٨﴾}\\
139.\  & \mytextarabic{وَإِنَّ يُونُسَ لَمِنَ ٱلْمُرْسَلِينَ ﴿١٣٩﴾}\\
140.\  & \mytextarabic{إِذْ أَبَقَ إِلَى ٱلْفُلْكِ ٱلْمَشْحُونِ ﴿١٤٠﴾}\\
141.\  & \mytextarabic{فَسَاهَمَ فَكَانَ مِنَ ٱلْمُدْحَضِينَ ﴿١٤١﴾}\\
142.\  & \mytextarabic{فَٱلْتَقَمَهُ ٱلْحُوتُ وَهُوَ مُلِيمٌۭ ﴿١٤٢﴾}\\
143.\  & \mytextarabic{فَلَوْلَآ أَنَّهُۥ كَانَ مِنَ ٱلْمُسَبِّحِينَ ﴿١٤٣﴾}\\
144.\  & \mytextarabic{لَلَبِثَ فِى بَطْنِهِۦٓ إِلَىٰ يَوْمِ يُبْعَثُونَ ﴿١٤٤﴾}\\
145.\  & \mytextarabic{۞ فَنَبَذْنَـٰهُ بِٱلْعَرَآءِ وَهُوَ سَقِيمٌۭ ﴿١٤٥﴾}\\
146.\  & \mytextarabic{وَأَنۢبَتْنَا عَلَيْهِ شَجَرَةًۭ مِّن يَقْطِينٍۢ ﴿١٤٦﴾}\\
147.\  & \mytextarabic{وَأَرْسَلْنَـٰهُ إِلَىٰ مِا۟ئَةِ أَلْفٍ أَوْ يَزِيدُونَ ﴿١٤٧﴾}\\
148.\  & \mytextarabic{فَـَٔامَنُوا۟ فَمَتَّعْنَـٰهُمْ إِلَىٰ حِينٍۢ ﴿١٤٨﴾}\\
149.\  & \mytextarabic{فَٱسْتَفْتِهِمْ أَلِرَبِّكَ ٱلْبَنَاتُ وَلَهُمُ ٱلْبَنُونَ ﴿١٤٩﴾}\\
150.\  & \mytextarabic{أَمْ خَلَقْنَا ٱلْمَلَـٰٓئِكَةَ إِنَـٰثًۭا وَهُمْ شَـٰهِدُونَ ﴿١٥٠﴾}\\
151.\  & \mytextarabic{أَلَآ إِنَّهُم مِّنْ إِفْكِهِمْ لَيَقُولُونَ ﴿١٥١﴾}\\
152.\  & \mytextarabic{وَلَدَ ٱللَّهُ وَإِنَّهُمْ لَكَـٰذِبُونَ ﴿١٥٢﴾}\\
153.\  & \mytextarabic{أَصْطَفَى ٱلْبَنَاتِ عَلَى ٱلْبَنِينَ ﴿١٥٣﴾}\\
154.\  & \mytextarabic{مَا لَكُمْ كَيْفَ تَحْكُمُونَ ﴿١٥٤﴾}\\
155.\  & \mytextarabic{أَفَلَا تَذَكَّرُونَ ﴿١٥٥﴾}\\
156.\  & \mytextarabic{أَمْ لَكُمْ سُلْطَٰنٌۭ مُّبِينٌۭ ﴿١٥٦﴾}\\
157.\  & \mytextarabic{فَأْتُوا۟ بِكِتَـٰبِكُمْ إِن كُنتُمْ صَـٰدِقِينَ ﴿١٥٧﴾}\\
158.\  & \mytextarabic{وَجَعَلُوا۟ بَيْنَهُۥ وَبَيْنَ ٱلْجِنَّةِ نَسَبًۭا ۚ وَلَقَدْ عَلِمَتِ ٱلْجِنَّةُ إِنَّهُمْ لَمُحْضَرُونَ ﴿١٥٨﴾}\\
159.\  & \mytextarabic{سُبْحَـٰنَ ٱللَّهِ عَمَّا يَصِفُونَ ﴿١٥٩﴾}\\
160.\  & \mytextarabic{إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿١٦٠﴾}\\
161.\  & \mytextarabic{فَإِنَّكُمْ وَمَا تَعْبُدُونَ ﴿١٦١﴾}\\
162.\  & \mytextarabic{مَآ أَنتُمْ عَلَيْهِ بِفَـٰتِنِينَ ﴿١٦٢﴾}\\
163.\  & \mytextarabic{إِلَّا مَنْ هُوَ صَالِ ٱلْجَحِيمِ ﴿١٦٣﴾}\\
164.\  & \mytextarabic{وَمَا مِنَّآ إِلَّا لَهُۥ مَقَامٌۭ مَّعْلُومٌۭ ﴿١٦٤﴾}\\
165.\  & \mytextarabic{وَإِنَّا لَنَحْنُ ٱلصَّآفُّونَ ﴿١٦٥﴾}\\
166.\  & \mytextarabic{وَإِنَّا لَنَحْنُ ٱلْمُسَبِّحُونَ ﴿١٦٦﴾}\\
167.\  & \mytextarabic{وَإِن كَانُوا۟ لَيَقُولُونَ ﴿١٦٧﴾}\\
168.\  & \mytextarabic{لَوْ أَنَّ عِندَنَا ذِكْرًۭا مِّنَ ٱلْأَوَّلِينَ ﴿١٦٨﴾}\\
169.\  & \mytextarabic{لَكُنَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿١٦٩﴾}\\
170.\  & \mytextarabic{فَكَفَرُوا۟ بِهِۦ ۖ فَسَوْفَ يَعْلَمُونَ ﴿١٧٠﴾}\\
171.\  & \mytextarabic{وَلَقَدْ سَبَقَتْ كَلِمَتُنَا لِعِبَادِنَا ٱلْمُرْسَلِينَ ﴿١٧١﴾}\\
172.\  & \mytextarabic{إِنَّهُمْ لَهُمُ ٱلْمَنصُورُونَ ﴿١٧٢﴾}\\
173.\  & \mytextarabic{وَإِنَّ جُندَنَا لَهُمُ ٱلْغَٰلِبُونَ ﴿١٧٣﴾}\\
174.\  & \mytextarabic{فَتَوَلَّ عَنْهُمْ حَتَّىٰ حِينٍۢ ﴿١٧٤﴾}\\
175.\  & \mytextarabic{وَأَبْصِرْهُمْ فَسَوْفَ يُبْصِرُونَ ﴿١٧٥﴾}\\
176.\  & \mytextarabic{أَفَبِعَذَابِنَا يَسْتَعْجِلُونَ ﴿١٧٦﴾}\\
177.\  & \mytextarabic{فَإِذَا نَزَلَ بِسَاحَتِهِمْ فَسَآءَ صَبَاحُ ٱلْمُنذَرِينَ ﴿١٧٧﴾}\\
178.\  & \mytextarabic{وَتَوَلَّ عَنْهُمْ حَتَّىٰ حِينٍۢ ﴿١٧٨﴾}\\
179.\  & \mytextarabic{وَأَبْصِرْ فَسَوْفَ يُبْصِرُونَ ﴿١٧٩﴾}\\
180.\  & \mytextarabic{سُبْحَـٰنَ رَبِّكَ رَبِّ ٱلْعِزَّةِ عَمَّا يَصِفُونَ ﴿١٨٠﴾}\\
181.\  & \mytextarabic{وَسَلَـٰمٌ عَلَى ٱلْمُرْسَلِينَ ﴿١٨١﴾}\\
182.\  & \mytextarabic{وَٱلْحَمْدُ لِلَّهِ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٨٢﴾}\\
\end{longtable}
