\begin{center}\section{ሱራቱ አልቀለም -  \textarabic{سوره  القلم}}\end{center}
\begin{longtable}{%
  @{}
    p{.5\textwidth}
  @{~~~}
    p{.5\textwidth}
    @{}
}
ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ نٓ ۚ وَٱلْقَلَمِ وَمَا يَسْطُرُونَ ﴿١﴾}\\
2.\  & \mytextarabic{مَآ أَنتَ بِنِعْمَةِ رَبِّكَ بِمَجْنُونٍۢ ﴿٢﴾}\\
3.\  & \mytextarabic{وَإِنَّ لَكَ لَأَجْرًا غَيْرَ مَمْنُونٍۢ ﴿٣﴾}\\
4.\  & \mytextarabic{وَإِنَّكَ لَعَلَىٰ خُلُقٍ عَظِيمٍۢ ﴿٤﴾}\\
5.\  & \mytextarabic{فَسَتُبْصِرُ وَيُبْصِرُونَ ﴿٥﴾}\\
6.\  & \mytextarabic{بِأَييِّكُمُ ٱلْمَفْتُونُ ﴿٦﴾}\\
7.\  & \mytextarabic{إِنَّ رَبَّكَ هُوَ أَعْلَمُ بِمَن ضَلَّ عَن سَبِيلِهِۦ وَهُوَ أَعْلَمُ بِٱلْمُهْتَدِينَ ﴿٧﴾}\\
8.\  & \mytextarabic{فَلَا تُطِعِ ٱلْمُكَذِّبِينَ ﴿٨﴾}\\
9.\  & \mytextarabic{وَدُّوا۟ لَوْ تُدْهِنُ فَيُدْهِنُونَ ﴿٩﴾}\\
10.\  & \mytextarabic{وَلَا تُطِعْ كُلَّ حَلَّافٍۢ مَّهِينٍ ﴿١٠﴾}\\
11.\  & \mytextarabic{هَمَّازٍۢ مَّشَّآءٍۭ بِنَمِيمٍۢ ﴿١١﴾}\\
12.\  & \mytextarabic{مَّنَّاعٍۢ لِّلْخَيْرِ مُعْتَدٍ أَثِيمٍ ﴿١٢﴾}\\
13.\  & \mytextarabic{عُتُلٍّۭ بَعْدَ ذَٟلِكَ زَنِيمٍ ﴿١٣﴾}\\
14.\  & \mytextarabic{أَن كَانَ ذَا مَالٍۢ وَبَنِينَ ﴿١٤﴾}\\
15.\  & \mytextarabic{إِذَا تُتْلَىٰ عَلَيْهِ ءَايَـٰتُنَا قَالَ أَسَـٰطِيرُ ٱلْأَوَّلِينَ ﴿١٥﴾}\\
16.\  & \mytextarabic{سَنَسِمُهُۥ عَلَى ٱلْخُرْطُومِ ﴿١٦﴾}\\
17.\  & \mytextarabic{إِنَّا بَلَوْنَـٰهُمْ كَمَا بَلَوْنَآ أَصْحَـٰبَ ٱلْجَنَّةِ إِذْ أَقْسَمُوا۟ لَيَصْرِمُنَّهَا مُصْبِحِينَ ﴿١٧﴾}\\
18.\  & \mytextarabic{وَلَا يَسْتَثْنُونَ ﴿١٨﴾}\\
19.\  & \mytextarabic{فَطَافَ عَلَيْهَا طَآئِفٌۭ مِّن رَّبِّكَ وَهُمْ نَآئِمُونَ ﴿١٩﴾}\\
20.\  & \mytextarabic{فَأَصْبَحَتْ كَٱلصَّرِيمِ ﴿٢٠﴾}\\
21.\  & \mytextarabic{فَتَنَادَوْا۟ مُصْبِحِينَ ﴿٢١﴾}\\
22.\  & \mytextarabic{أَنِ ٱغْدُوا۟ عَلَىٰ حَرْثِكُمْ إِن كُنتُمْ صَـٰرِمِينَ ﴿٢٢﴾}\\
23.\  & \mytextarabic{فَٱنطَلَقُوا۟ وَهُمْ يَتَخَـٰفَتُونَ ﴿٢٣﴾}\\
24.\  & \mytextarabic{أَن لَّا يَدْخُلَنَّهَا ٱلْيَوْمَ عَلَيْكُم مِّسْكِينٌۭ ﴿٢٤﴾}\\
25.\  & \mytextarabic{وَغَدَوْا۟ عَلَىٰ حَرْدٍۢ قَـٰدِرِينَ ﴿٢٥﴾}\\
26.\  & \mytextarabic{فَلَمَّا رَأَوْهَا قَالُوٓا۟ إِنَّا لَضَآلُّونَ ﴿٢٦﴾}\\
27.\  & \mytextarabic{بَلْ نَحْنُ مَحْرُومُونَ ﴿٢٧﴾}\\
28.\  & \mytextarabic{قَالَ أَوْسَطُهُمْ أَلَمْ أَقُل لَّكُمْ لَوْلَا تُسَبِّحُونَ ﴿٢٨﴾}\\
29.\  & \mytextarabic{قَالُوا۟ سُبْحَـٰنَ رَبِّنَآ إِنَّا كُنَّا ظَـٰلِمِينَ ﴿٢٩﴾}\\
30.\  & \mytextarabic{فَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٍۢ يَتَلَـٰوَمُونَ ﴿٣٠﴾}\\
31.\  & \mytextarabic{قَالُوا۟ يَـٰوَيْلَنَآ إِنَّا كُنَّا طَٰغِينَ ﴿٣١﴾}\\
32.\  & \mytextarabic{عَسَىٰ رَبُّنَآ أَن يُبْدِلَنَا خَيْرًۭا مِّنْهَآ إِنَّآ إِلَىٰ رَبِّنَا رَٰغِبُونَ ﴿٣٢﴾}\\
33.\  & \mytextarabic{كَذَٟلِكَ ٱلْعَذَابُ ۖ وَلَعَذَابُ ٱلْءَاخِرَةِ أَكْبَرُ ۚ لَوْ كَانُوا۟ يَعْلَمُونَ ﴿٣٣﴾}\\
34.\  & \mytextarabic{إِنَّ لِلْمُتَّقِينَ عِندَ رَبِّهِمْ جَنَّـٰتِ ٱلنَّعِيمِ ﴿٣٤﴾}\\
35.\  & \mytextarabic{أَفَنَجْعَلُ ٱلْمُسْلِمِينَ كَٱلْمُجْرِمِينَ ﴿٣٥﴾}\\
36.\  & \mytextarabic{مَا لَكُمْ كَيْفَ تَحْكُمُونَ ﴿٣٦﴾}\\
37.\  & \mytextarabic{أَمْ لَكُمْ كِتَـٰبٌۭ فِيهِ تَدْرُسُونَ ﴿٣٧﴾}\\
38.\  & \mytextarabic{إِنَّ لَكُمْ فِيهِ لَمَا تَخَيَّرُونَ ﴿٣٨﴾}\\
39.\  & \mytextarabic{أَمْ لَكُمْ أَيْمَـٰنٌ عَلَيْنَا بَٰلِغَةٌ إِلَىٰ يَوْمِ ٱلْقِيَـٰمَةِ ۙ إِنَّ لَكُمْ لَمَا تَحْكُمُونَ ﴿٣٩﴾}\\
40.\  & \mytextarabic{سَلْهُمْ أَيُّهُم بِذَٟلِكَ زَعِيمٌ ﴿٤٠﴾}\\
41.\  & \mytextarabic{أَمْ لَهُمْ شُرَكَآءُ فَلْيَأْتُوا۟ بِشُرَكَآئِهِمْ إِن كَانُوا۟ صَـٰدِقِينَ ﴿٤١﴾}\\
42.\  & \mytextarabic{يَوْمَ يُكْشَفُ عَن سَاقٍۢ وَيُدْعَوْنَ إِلَى ٱلسُّجُودِ فَلَا يَسْتَطِيعُونَ ﴿٤٢﴾}\\
43.\  & \mytextarabic{خَـٰشِعَةً أَبْصَـٰرُهُمْ تَرْهَقُهُمْ ذِلَّةٌۭ ۖ وَقَدْ كَانُوا۟ يُدْعَوْنَ إِلَى ٱلسُّجُودِ وَهُمْ سَـٰلِمُونَ ﴿٤٣﴾}\\
44.\  & \mytextarabic{فَذَرْنِى وَمَن يُكَذِّبُ بِهَـٰذَا ٱلْحَدِيثِ ۖ سَنَسْتَدْرِجُهُم مِّنْ حَيْثُ لَا يَعْلَمُونَ ﴿٤٤﴾}\\
45.\  & \mytextarabic{وَأُمْلِى لَهُمْ ۚ إِنَّ كَيْدِى مَتِينٌ ﴿٤٥﴾}\\
46.\  & \mytextarabic{أَمْ تَسْـَٔلُهُمْ أَجْرًۭا فَهُم مِّن مَّغْرَمٍۢ مُّثْقَلُونَ ﴿٤٦﴾}\\
47.\  & \mytextarabic{أَمْ عِندَهُمُ ٱلْغَيْبُ فَهُمْ يَكْتُبُونَ ﴿٤٧﴾}\\
48.\  & \mytextarabic{فَٱصْبِرْ لِحُكْمِ رَبِّكَ وَلَا تَكُن كَصَاحِبِ ٱلْحُوتِ إِذْ نَادَىٰ وَهُوَ مَكْظُومٌۭ ﴿٤٨﴾}\\
49.\  & \mytextarabic{لَّوْلَآ أَن تَدَٟرَكَهُۥ نِعْمَةٌۭ مِّن رَّبِّهِۦ لَنُبِذَ بِٱلْعَرَآءِ وَهُوَ مَذْمُومٌۭ ﴿٤٩﴾}\\
50.\  & \mytextarabic{فَٱجْتَبَٰهُ رَبُّهُۥ فَجَعَلَهُۥ مِنَ ٱلصَّـٰلِحِينَ ﴿٥٠﴾}\\
51.\  & \mytextarabic{وَإِن يَكَادُ ٱلَّذِينَ كَفَرُوا۟ لَيُزْلِقُونَكَ بِأَبْصَـٰرِهِمْ لَمَّا سَمِعُوا۟ ٱلذِّكْرَ وَيَقُولُونَ إِنَّهُۥ لَمَجْنُونٌۭ ﴿٥١﴾}\\
52.\  & \mytextarabic{وَمَا هُوَ إِلَّا ذِكْرٌۭ لِّلْعَـٰلَمِينَ ﴿٥٢﴾}\\
\end{longtable}
