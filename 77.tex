\begin{center}\section{ሱራቱ አልሙርሰላት -  \textarabic{سوره  المرسلات}}\end{center}
\begin{longtable}{%
  @{}
    p{.5\textwidth}
  @{~~~}
    p{.5\textwidth}
    @{}
}
ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ وَٱلْمُرْسَلَـٰتِ عُرْفًۭا ﴿١﴾}\\
2.\  & \mytextarabic{فَٱلْعَـٰصِفَـٰتِ عَصْفًۭا ﴿٢﴾}\\
3.\  & \mytextarabic{وَٱلنَّـٰشِرَٰتِ نَشْرًۭا ﴿٣﴾}\\
4.\  & \mytextarabic{فَٱلْفَـٰرِقَـٰتِ فَرْقًۭا ﴿٤﴾}\\
5.\  & \mytextarabic{فَٱلْمُلْقِيَـٰتِ ذِكْرًا ﴿٥﴾}\\
6.\  & \mytextarabic{عُذْرًا أَوْ نُذْرًا ﴿٦﴾}\\
7.\  & \mytextarabic{إِنَّمَا تُوعَدُونَ لَوَٟقِعٌۭ ﴿٧﴾}\\
8.\  & \mytextarabic{فَإِذَا ٱلنُّجُومُ طُمِسَتْ ﴿٨﴾}\\
9.\  & \mytextarabic{وَإِذَا ٱلسَّمَآءُ فُرِجَتْ ﴿٩﴾}\\
10.\  & \mytextarabic{وَإِذَا ٱلْجِبَالُ نُسِفَتْ ﴿١٠﴾}\\
11.\  & \mytextarabic{وَإِذَا ٱلرُّسُلُ أُقِّتَتْ ﴿١١﴾}\\
12.\  & \mytextarabic{لِأَىِّ يَوْمٍ أُجِّلَتْ ﴿١٢﴾}\\
13.\  & \mytextarabic{لِيَوْمِ ٱلْفَصْلِ ﴿١٣﴾}\\
14.\  & \mytextarabic{وَمَآ أَدْرَىٰكَ مَا يَوْمُ ٱلْفَصْلِ ﴿١٤﴾}\\
15.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿١٥﴾}\\
16.\  & \mytextarabic{أَلَمْ نُهْلِكِ ٱلْأَوَّلِينَ ﴿١٦﴾}\\
17.\  & \mytextarabic{ثُمَّ نُتْبِعُهُمُ ٱلْءَاخِرِينَ ﴿١٧﴾}\\
18.\  & \mytextarabic{كَذَٟلِكَ نَفْعَلُ بِٱلْمُجْرِمِينَ ﴿١٨﴾}\\
19.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿١٩﴾}\\
20.\  & \mytextarabic{أَلَمْ نَخْلُقكُّم مِّن مَّآءٍۢ مَّهِينٍۢ ﴿٢٠﴾}\\
21.\  & \mytextarabic{فَجَعَلْنَـٰهُ فِى قَرَارٍۢ مَّكِينٍ ﴿٢١﴾}\\
22.\  & \mytextarabic{إِلَىٰ قَدَرٍۢ مَّعْلُومٍۢ ﴿٢٢﴾}\\
23.\  & \mytextarabic{فَقَدَرْنَا فَنِعْمَ ٱلْقَـٰدِرُونَ ﴿٢٣﴾}\\
24.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿٢٤﴾}\\
25.\  & \mytextarabic{أَلَمْ نَجْعَلِ ٱلْأَرْضَ كِفَاتًا ﴿٢٥﴾}\\
26.\  & \mytextarabic{أَحْيَآءًۭ وَأَمْوَٟتًۭا ﴿٢٦﴾}\\
27.\  & \mytextarabic{وَجَعَلْنَا فِيهَا رَوَٟسِىَ شَـٰمِخَـٰتٍۢ وَأَسْقَيْنَـٰكُم مَّآءًۭ فُرَاتًۭا ﴿٢٧﴾}\\
28.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿٢٨﴾}\\
29.\  & \mytextarabic{ٱنطَلِقُوٓا۟ إِلَىٰ مَا كُنتُم بِهِۦ تُكَذِّبُونَ ﴿٢٩﴾}\\
30.\  & \mytextarabic{ٱنطَلِقُوٓا۟ إِلَىٰ ظِلٍّۢ ذِى ثَلَـٰثِ شُعَبٍۢ ﴿٣٠﴾}\\
31.\  & \mytextarabic{لَّا ظَلِيلٍۢ وَلَا يُغْنِى مِنَ ٱللَّهَبِ ﴿٣١﴾}\\
32.\  & \mytextarabic{إِنَّهَا تَرْمِى بِشَرَرٍۢ كَٱلْقَصْرِ ﴿٣٢﴾}\\
33.\  & \mytextarabic{كَأَنَّهُۥ جِمَـٰلَتٌۭ صُفْرٌۭ ﴿٣٣﴾}\\
34.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿٣٤﴾}\\
35.\  & \mytextarabic{هَـٰذَا يَوْمُ لَا يَنطِقُونَ ﴿٣٥﴾}\\
36.\  & \mytextarabic{وَلَا يُؤْذَنُ لَهُمْ فَيَعْتَذِرُونَ ﴿٣٦﴾}\\
37.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿٣٧﴾}\\
38.\  & \mytextarabic{هَـٰذَا يَوْمُ ٱلْفَصْلِ ۖ جَمَعْنَـٰكُمْ وَٱلْأَوَّلِينَ ﴿٣٨﴾}\\
39.\  & \mytextarabic{فَإِن كَانَ لَكُمْ كَيْدٌۭ فَكِيدُونِ ﴿٣٩﴾}\\
40.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿٤٠﴾}\\
41.\  & \mytextarabic{إِنَّ ٱلْمُتَّقِينَ فِى ظِلَـٰلٍۢ وَعُيُونٍۢ ﴿٤١﴾}\\
42.\  & \mytextarabic{وَفَوَٟكِهَ مِمَّا يَشْتَهُونَ ﴿٤٢﴾}\\
43.\  & \mytextarabic{كُلُوا۟ وَٱشْرَبُوا۟ هَنِيٓـًٔۢا بِمَا كُنتُمْ تَعْمَلُونَ ﴿٤٣﴾}\\
44.\  & \mytextarabic{إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿٤٤﴾}\\
45.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿٤٥﴾}\\
46.\  & \mytextarabic{كُلُوا۟ وَتَمَتَّعُوا۟ قَلِيلًا إِنَّكُم مُّجْرِمُونَ ﴿٤٦﴾}\\
47.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿٤٧﴾}\\
48.\  & \mytextarabic{وَإِذَا قِيلَ لَهُمُ ٱرْكَعُوا۟ لَا يَرْكَعُونَ ﴿٤٨﴾}\\
49.\  & \mytextarabic{وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿٤٩﴾}\\
50.\  & \mytextarabic{فَبِأَىِّ حَدِيثٍۭ بَعْدَهُۥ يُؤْمِنُونَ ﴿٥٠﴾}\\
\end{longtable}
