\begin{center}\section{ሱራቱ አንነባኢ -  \textarabic{سوره  النبإ}}\end{center}
\begin{longtable}{%
  @{}
    p{.5\textwidth}
  @{~~~}
    p{.5\textwidth}
    @{}
}
ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ عَمَّ يَتَسَآءَلُونَ ﴿١﴾}\\
2.\  & \mytextarabic{عَنِ ٱلنَّبَإِ ٱلْعَظِيمِ ﴿٢﴾}\\
3.\  & \mytextarabic{ٱلَّذِى هُمْ فِيهِ مُخْتَلِفُونَ ﴿٣﴾}\\
4.\  & \mytextarabic{كَلَّا سَيَعْلَمُونَ ﴿٤﴾}\\
5.\  & \mytextarabic{ثُمَّ كَلَّا سَيَعْلَمُونَ ﴿٥﴾}\\
6.\  & \mytextarabic{أَلَمْ نَجْعَلِ ٱلْأَرْضَ مِهَـٰدًۭا ﴿٦﴾}\\
7.\  & \mytextarabic{وَٱلْجِبَالَ أَوْتَادًۭا ﴿٧﴾}\\
8.\  & \mytextarabic{وَخَلَقْنَـٰكُمْ أَزْوَٟجًۭا ﴿٨﴾}\\
9.\  & \mytextarabic{وَجَعَلْنَا نَوْمَكُمْ سُبَاتًۭا ﴿٩﴾}\\
10.\  & \mytextarabic{وَجَعَلْنَا ٱلَّيْلَ لِبَاسًۭا ﴿١٠﴾}\\
11.\  & \mytextarabic{وَجَعَلْنَا ٱلنَّهَارَ مَعَاشًۭا ﴿١١﴾}\\
12.\  & \mytextarabic{وَبَنَيْنَا فَوْقَكُمْ سَبْعًۭا شِدَادًۭا ﴿١٢﴾}\\
13.\  & \mytextarabic{وَجَعَلْنَا سِرَاجًۭا وَهَّاجًۭا ﴿١٣﴾}\\
14.\  & \mytextarabic{وَأَنزَلْنَا مِنَ ٱلْمُعْصِرَٰتِ مَآءًۭ ثَجَّاجًۭا ﴿١٤﴾}\\
15.\  & \mytextarabic{لِّنُخْرِجَ بِهِۦ حَبًّۭا وَنَبَاتًۭا ﴿١٥﴾}\\
16.\  & \mytextarabic{وَجَنَّـٰتٍ أَلْفَافًا ﴿١٦﴾}\\
17.\  & \mytextarabic{إِنَّ يَوْمَ ٱلْفَصْلِ كَانَ مِيقَـٰتًۭا ﴿١٧﴾}\\
18.\  & \mytextarabic{يَوْمَ يُنفَخُ فِى ٱلصُّورِ فَتَأْتُونَ أَفْوَاجًۭا ﴿١٨﴾}\\
19.\  & \mytextarabic{وَفُتِحَتِ ٱلسَّمَآءُ فَكَانَتْ أَبْوَٟبًۭا ﴿١٩﴾}\\
20.\  & \mytextarabic{وَسُيِّرَتِ ٱلْجِبَالُ فَكَانَتْ سَرَابًا ﴿٢٠﴾}\\
21.\  & \mytextarabic{إِنَّ جَهَنَّمَ كَانَتْ مِرْصَادًۭا ﴿٢١﴾}\\
22.\  & \mytextarabic{لِّلطَّٰغِينَ مَـَٔابًۭا ﴿٢٢﴾}\\
23.\  & \mytextarabic{لَّٰبِثِينَ فِيهَآ أَحْقَابًۭا ﴿٢٣﴾}\\
24.\  & \mytextarabic{لَّا يَذُوقُونَ فِيهَا بَرْدًۭا وَلَا شَرَابًا ﴿٢٤﴾}\\
25.\  & \mytextarabic{إِلَّا حَمِيمًۭا وَغَسَّاقًۭا ﴿٢٥﴾}\\
26.\  & \mytextarabic{جَزَآءًۭ وِفَاقًا ﴿٢٦﴾}\\
27.\  & \mytextarabic{إِنَّهُمْ كَانُوا۟ لَا يَرْجُونَ حِسَابًۭا ﴿٢٧﴾}\\
28.\  & \mytextarabic{وَكَذَّبُوا۟ بِـَٔايَـٰتِنَا كِذَّابًۭا ﴿٢٨﴾}\\
29.\  & \mytextarabic{وَكُلَّ شَىْءٍ أَحْصَيْنَـٰهُ كِتَـٰبًۭا ﴿٢٩﴾}\\
30.\  & \mytextarabic{فَذُوقُوا۟ فَلَن نَّزِيدَكُمْ إِلَّا عَذَابًا ﴿٣٠﴾}\\
31.\  & \mytextarabic{إِنَّ لِلْمُتَّقِينَ مَفَازًا ﴿٣١﴾}\\
32.\  & \mytextarabic{حَدَآئِقَ وَأَعْنَـٰبًۭا ﴿٣٢﴾}\\
33.\  & \mytextarabic{وَكَوَاعِبَ أَتْرَابًۭا ﴿٣٣﴾}\\
34.\  & \mytextarabic{وَكَأْسًۭا دِهَاقًۭا ﴿٣٤﴾}\\
35.\  & \mytextarabic{لَّا يَسْمَعُونَ فِيهَا لَغْوًۭا وَلَا كِذَّٰبًۭا ﴿٣٥﴾}\\
36.\  & \mytextarabic{جَزَآءًۭ مِّن رَّبِّكَ عَطَآءً حِسَابًۭا ﴿٣٦﴾}\\
37.\  & \mytextarabic{رَّبِّ ٱلسَّمَـٰوَٟتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا ٱلرَّحْمَـٰنِ ۖ لَا يَمْلِكُونَ مِنْهُ خِطَابًۭا ﴿٣٧﴾}\\
38.\  & \mytextarabic{يَوْمَ يَقُومُ ٱلرُّوحُ وَٱلْمَلَـٰٓئِكَةُ صَفًّۭا ۖ لَّا يَتَكَلَّمُونَ إِلَّا مَنْ أَذِنَ لَهُ ٱلرَّحْمَـٰنُ وَقَالَ صَوَابًۭا ﴿٣٨﴾}\\
39.\  & \mytextarabic{ذَٟلِكَ ٱلْيَوْمُ ٱلْحَقُّ ۖ فَمَن شَآءَ ٱتَّخَذَ إِلَىٰ رَبِّهِۦ مَـَٔابًا ﴿٣٩﴾}\\
40.\  & \mytextarabic{إِنَّآ أَنذَرْنَـٰكُمْ عَذَابًۭا قَرِيبًۭا يَوْمَ يَنظُرُ ٱلْمَرْءُ مَا قَدَّمَتْ يَدَاهُ وَيَقُولُ ٱلْكَافِرُ يَـٰلَيْتَنِى كُنتُ تُرَٰبًۢا ﴿٤٠﴾}\\
\end{longtable}
