\begin{center}\section{ሱራቱ አልዱኻን -  \textarabic{سوره  الدخان}}\end{center}
\begin{longtable}{%
  @{}
    p{.5\textwidth}

}
ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ حمٓ ﴿١﴾}\\
2.\  & \mytextarabic{وَٱلْكِتَـٰبِ ٱلْمُبِينِ ﴿٢﴾}\\
3.\  & \mytextarabic{إِنَّآ أَنزَلْنَـٰهُ فِى لَيْلَةٍۢ مُّبَٰرَكَةٍ ۚ إِنَّا كُنَّا مُنذِرِينَ ﴿٣﴾}\\
4.\  & \mytextarabic{فِيهَا يُفْرَقُ كُلُّ أَمْرٍ حَكِيمٍ ﴿٤﴾}\\
5.\  & \mytextarabic{أَمْرًۭا مِّنْ عِندِنَآ ۚ إِنَّا كُنَّا مُرْسِلِينَ ﴿٥﴾}\\
6.\  & \mytextarabic{رَحْمَةًۭ مِّن رَّبِّكَ ۚ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿٦﴾}\\
7.\  & \mytextarabic{رَبِّ ٱلسَّمَـٰوَٟتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَآ ۖ إِن كُنتُم مُّوقِنِينَ ﴿٧﴾}\\
8.\  & \mytextarabic{لَآ إِلَـٰهَ إِلَّا هُوَ يُحْىِۦ وَيُمِيتُ ۖ رَبُّكُمْ وَرَبُّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿٨﴾}\\
9.\  & \mytextarabic{بَلْ هُمْ فِى شَكٍّۢ يَلْعَبُونَ ﴿٩﴾}\\
10.\  & \mytextarabic{فَٱرْتَقِبْ يَوْمَ تَأْتِى ٱلسَّمَآءُ بِدُخَانٍۢ مُّبِينٍۢ ﴿١٠﴾}\\
11.\  & \mytextarabic{يَغْشَى ٱلنَّاسَ ۖ هَـٰذَا عَذَابٌ أَلِيمٌۭ ﴿١١﴾}\\
12.\  & \mytextarabic{رَّبَّنَا ٱكْشِفْ عَنَّا ٱلْعَذَابَ إِنَّا مُؤْمِنُونَ ﴿١٢﴾}\\
13.\  & \mytextarabic{أَنَّىٰ لَهُمُ ٱلذِّكْرَىٰ وَقَدْ جَآءَهُمْ رَسُولٌۭ مُّبِينٌۭ ﴿١٣﴾}\\
14.\  & \mytextarabic{ثُمَّ تَوَلَّوْا۟ عَنْهُ وَقَالُوا۟ مُعَلَّمٌۭ مَّجْنُونٌ ﴿١٤﴾}\\
15.\  & \mytextarabic{إِنَّا كَاشِفُوا۟ ٱلْعَذَابِ قَلِيلًا ۚ إِنَّكُمْ عَآئِدُونَ ﴿١٥﴾}\\
16.\  & \mytextarabic{يَوْمَ نَبْطِشُ ٱلْبَطْشَةَ ٱلْكُبْرَىٰٓ إِنَّا مُنتَقِمُونَ ﴿١٦﴾}\\
17.\  & \mytextarabic{۞ وَلَقَدْ فَتَنَّا قَبْلَهُمْ قَوْمَ فِرْعَوْنَ وَجَآءَهُمْ رَسُولٌۭ كَرِيمٌ ﴿١٧﴾}\\
18.\  & \mytextarabic{أَنْ أَدُّوٓا۟ إِلَىَّ عِبَادَ ٱللَّهِ ۖ إِنِّى لَكُمْ رَسُولٌ أَمِينٌۭ ﴿١٨﴾}\\
19.\  & \mytextarabic{وَأَن لَّا تَعْلُوا۟ عَلَى ٱللَّهِ ۖ إِنِّىٓ ءَاتِيكُم بِسُلْطَٰنٍۢ مُّبِينٍۢ ﴿١٩﴾}\\
20.\  & \mytextarabic{وَإِنِّى عُذْتُ بِرَبِّى وَرَبِّكُمْ أَن تَرْجُمُونِ ﴿٢٠﴾}\\
21.\  & \mytextarabic{وَإِن لَّمْ تُؤْمِنُوا۟ لِى فَٱعْتَزِلُونِ ﴿٢١﴾}\\
22.\  & \mytextarabic{فَدَعَا رَبَّهُۥٓ أَنَّ هَـٰٓؤُلَآءِ قَوْمٌۭ مُّجْرِمُونَ ﴿٢٢﴾}\\
23.\  & \mytextarabic{فَأَسْرِ بِعِبَادِى لَيْلًا إِنَّكُم مُّتَّبَعُونَ ﴿٢٣﴾}\\
24.\  & \mytextarabic{وَٱتْرُكِ ٱلْبَحْرَ رَهْوًا ۖ إِنَّهُمْ جُندٌۭ مُّغْرَقُونَ ﴿٢٤﴾}\\
25.\  & \mytextarabic{كَمْ تَرَكُوا۟ مِن جَنَّـٰتٍۢ وَعُيُونٍۢ ﴿٢٥﴾}\\
26.\  & \mytextarabic{وَزُرُوعٍۢ وَمَقَامٍۢ كَرِيمٍۢ ﴿٢٦﴾}\\
27.\  & \mytextarabic{وَنَعْمَةٍۢ كَانُوا۟ فِيهَا فَـٰكِهِينَ ﴿٢٧﴾}\\
28.\  & \mytextarabic{كَذَٟلِكَ ۖ وَأَوْرَثْنَـٰهَا قَوْمًا ءَاخَرِينَ ﴿٢٨﴾}\\
29.\  & \mytextarabic{فَمَا بَكَتْ عَلَيْهِمُ ٱلسَّمَآءُ وَٱلْأَرْضُ وَمَا كَانُوا۟ مُنظَرِينَ ﴿٢٩﴾}\\
30.\  & \mytextarabic{وَلَقَدْ نَجَّيْنَا بَنِىٓ إِسْرَٰٓءِيلَ مِنَ ٱلْعَذَابِ ٱلْمُهِينِ ﴿٣٠﴾}\\
31.\  & \mytextarabic{مِن فِرْعَوْنَ ۚ إِنَّهُۥ كَانَ عَالِيًۭا مِّنَ ٱلْمُسْرِفِينَ ﴿٣١﴾}\\
32.\  & \mytextarabic{وَلَقَدِ ٱخْتَرْنَـٰهُمْ عَلَىٰ عِلْمٍ عَلَى ٱلْعَـٰلَمِينَ ﴿٣٢﴾}\\
33.\  & \mytextarabic{وَءَاتَيْنَـٰهُم مِّنَ ٱلْءَايَـٰتِ مَا فِيهِ بَلَـٰٓؤٌۭا۟ مُّبِينٌ ﴿٣٣﴾}\\
34.\  & \mytextarabic{إِنَّ هَـٰٓؤُلَآءِ لَيَقُولُونَ ﴿٣٤﴾}\\
35.\  & \mytextarabic{إِنْ هِىَ إِلَّا مَوْتَتُنَا ٱلْأُولَىٰ وَمَا نَحْنُ بِمُنشَرِينَ ﴿٣٥﴾}\\
36.\  & \mytextarabic{فَأْتُوا۟ بِـَٔابَآئِنَآ إِن كُنتُمْ صَـٰدِقِينَ ﴿٣٦﴾}\\
37.\  & \mytextarabic{أَهُمْ خَيْرٌ أَمْ قَوْمُ تُبَّعٍۢ وَٱلَّذِينَ مِن قَبْلِهِمْ ۚ أَهْلَكْنَـٰهُمْ ۖ إِنَّهُمْ كَانُوا۟ مُجْرِمِينَ ﴿٣٧﴾}\\
38.\  & \mytextarabic{وَمَا خَلَقْنَا ٱلسَّمَـٰوَٟتِ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا لَـٰعِبِينَ ﴿٣٨﴾}\\
39.\  & \mytextarabic{مَا خَلَقْنَـٰهُمَآ إِلَّا بِٱلْحَقِّ وَلَـٰكِنَّ أَكْثَرَهُمْ لَا يَعْلَمُونَ ﴿٣٩﴾}\\
40.\  & \mytextarabic{إِنَّ يَوْمَ ٱلْفَصْلِ مِيقَـٰتُهُمْ أَجْمَعِينَ ﴿٤٠﴾}\\
41.\  & \mytextarabic{يَوْمَ لَا يُغْنِى مَوْلًى عَن مَّوْلًۭى شَيْـًۭٔا وَلَا هُمْ يُنصَرُونَ ﴿٤١﴾}\\
42.\  & \mytextarabic{إِلَّا مَن رَّحِمَ ٱللَّهُ ۚ إِنَّهُۥ هُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿٤٢﴾}\\
43.\  & \mytextarabic{إِنَّ شَجَرَتَ ٱلزَّقُّومِ ﴿٤٣﴾}\\
44.\  & \mytextarabic{طَعَامُ ٱلْأَثِيمِ ﴿٤٤﴾}\\
45.\  & \mytextarabic{كَٱلْمُهْلِ يَغْلِى فِى ٱلْبُطُونِ ﴿٤٥﴾}\\
46.\  & \mytextarabic{كَغَلْىِ ٱلْحَمِيمِ ﴿٤٦﴾}\\
47.\  & \mytextarabic{خُذُوهُ فَٱعْتِلُوهُ إِلَىٰ سَوَآءِ ٱلْجَحِيمِ ﴿٤٧﴾}\\
48.\  & \mytextarabic{ثُمَّ صُبُّوا۟ فَوْقَ رَأْسِهِۦ مِنْ عَذَابِ ٱلْحَمِيمِ ﴿٤٨﴾}\\
49.\  & \mytextarabic{ذُقْ إِنَّكَ أَنتَ ٱلْعَزِيزُ ٱلْكَرِيمُ ﴿٤٩﴾}\\
50.\  & \mytextarabic{إِنَّ هَـٰذَا مَا كُنتُم بِهِۦ تَمْتَرُونَ ﴿٥٠﴾}\\
51.\  & \mytextarabic{إِنَّ ٱلْمُتَّقِينَ فِى مَقَامٍ أَمِينٍۢ ﴿٥١﴾}\\
52.\  & \mytextarabic{فِى جَنَّـٰتٍۢ وَعُيُونٍۢ ﴿٥٢﴾}\\
53.\  & \mytextarabic{يَلْبَسُونَ مِن سُندُسٍۢ وَإِسْتَبْرَقٍۢ مُّتَقَـٰبِلِينَ ﴿٥٣﴾}\\
54.\  & \mytextarabic{كَذَٟلِكَ وَزَوَّجْنَـٰهُم بِحُورٍ عِينٍۢ ﴿٥٤﴾}\\
55.\  & \mytextarabic{يَدْعُونَ فِيهَا بِكُلِّ فَـٰكِهَةٍ ءَامِنِينَ ﴿٥٥﴾}\\
56.\  & \mytextarabic{لَا يَذُوقُونَ فِيهَا ٱلْمَوْتَ إِلَّا ٱلْمَوْتَةَ ٱلْأُولَىٰ ۖ وَوَقَىٰهُمْ عَذَابَ ٱلْجَحِيمِ ﴿٥٦﴾}\\
57.\  & \mytextarabic{فَضْلًۭا مِّن رَّبِّكَ ۚ ذَٟلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿٥٧﴾}\\
58.\  & \mytextarabic{فَإِنَّمَا يَسَّرْنَـٰهُ بِلِسَانِكَ لَعَلَّهُمْ يَتَذَكَّرُونَ ﴿٥٨﴾}\\
59.\  & \mytextarabic{فَٱرْتَقِبْ إِنَّهُم مُّرْتَقِبُونَ ﴿٥٩﴾}\\
\end{longtable}
