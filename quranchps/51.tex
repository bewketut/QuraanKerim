ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ وَٱلذَّٰرِيَـٰتِ ذَرْوًۭا ﴿١﴾}\\
2.\  & \mytextarabic{فَٱلْحَـٰمِلَـٰتِ وِقْرًۭا ﴿٢﴾}\\
3.\  & \mytextarabic{فَٱلْجَٰرِيَـٰتِ يُسْرًۭا ﴿٣﴾}\\
4.\  & \mytextarabic{فَٱلْمُقَسِّمَـٰتِ أَمْرًا ﴿٤﴾}\\
5.\  & \mytextarabic{إِنَّمَا تُوعَدُونَ لَصَادِقٌۭ ﴿٥﴾}\\
6.\  & \mytextarabic{وَإِنَّ ٱلدِّينَ لَوَٟقِعٌۭ ﴿٦﴾}\\
7.\  & \mytextarabic{وَٱلسَّمَآءِ ذَاتِ ٱلْحُبُكِ ﴿٧﴾}\\
8.\  & \mytextarabic{إِنَّكُمْ لَفِى قَوْلٍۢ مُّخْتَلِفٍۢ ﴿٨﴾}\\
9.\  & \mytextarabic{يُؤْفَكُ عَنْهُ مَنْ أُفِكَ ﴿٩﴾}\\
10.\  & \mytextarabic{قُتِلَ ٱلْخَرَّٟصُونَ ﴿١٠﴾}\\
11.\  & \mytextarabic{ٱلَّذِينَ هُمْ فِى غَمْرَةٍۢ سَاهُونَ ﴿١١﴾}\\
12.\  & \mytextarabic{يَسْـَٔلُونَ أَيَّانَ يَوْمُ ٱلدِّينِ ﴿١٢﴾}\\
13.\  & \mytextarabic{يَوْمَ هُمْ عَلَى ٱلنَّارِ يُفْتَنُونَ ﴿١٣﴾}\\
14.\  & \mytextarabic{ذُوقُوا۟ فِتْنَتَكُمْ هَـٰذَا ٱلَّذِى كُنتُم بِهِۦ تَسْتَعْجِلُونَ ﴿١٤﴾}\\
15.\  & \mytextarabic{إِنَّ ٱلْمُتَّقِينَ فِى جَنَّـٰتٍۢ وَعُيُونٍ ﴿١٥﴾}\\
16.\  & \mytextarabic{ءَاخِذِينَ مَآ ءَاتَىٰهُمْ رَبُّهُمْ ۚ إِنَّهُمْ كَانُوا۟ قَبْلَ ذَٟلِكَ مُحْسِنِينَ ﴿١٦﴾}\\
17.\  & \mytextarabic{كَانُوا۟ قَلِيلًۭا مِّنَ ٱلَّيْلِ مَا يَهْجَعُونَ ﴿١٧﴾}\\
18.\  & \mytextarabic{وَبِٱلْأَسْحَارِ هُمْ يَسْتَغْفِرُونَ ﴿١٨﴾}\\
19.\  & \mytextarabic{وَفِىٓ أَمْوَٟلِهِمْ حَقٌّۭ لِّلسَّآئِلِ وَٱلْمَحْرُومِ ﴿١٩﴾}\\
20.\  & \mytextarabic{وَفِى ٱلْأَرْضِ ءَايَـٰتٌۭ لِّلْمُوقِنِينَ ﴿٢٠﴾}\\
21.\  & \mytextarabic{وَفِىٓ أَنفُسِكُمْ ۚ أَفَلَا تُبْصِرُونَ ﴿٢١﴾}\\
22.\  & \mytextarabic{وَفِى ٱلسَّمَآءِ رِزْقُكُمْ وَمَا تُوعَدُونَ ﴿٢٢﴾}\\
23.\  & \mytextarabic{فَوَرَبِّ ٱلسَّمَآءِ وَٱلْأَرْضِ إِنَّهُۥ لَحَقٌّۭ مِّثْلَ مَآ أَنَّكُمْ تَنطِقُونَ ﴿٢٣﴾}\\
24.\  & \mytextarabic{هَلْ أَتَىٰكَ حَدِيثُ ضَيْفِ إِبْرَٰهِيمَ ٱلْمُكْرَمِينَ ﴿٢٤﴾}\\
25.\  & \mytextarabic{إِذْ دَخَلُوا۟ عَلَيْهِ فَقَالُوا۟ سَلَـٰمًۭا ۖ قَالَ سَلَـٰمٌۭ قَوْمٌۭ مُّنكَرُونَ ﴿٢٥﴾}\\
26.\  & \mytextarabic{فَرَاغَ إِلَىٰٓ أَهْلِهِۦ فَجَآءَ بِعِجْلٍۢ سَمِينٍۢ ﴿٢٦﴾}\\
27.\  & \mytextarabic{فَقَرَّبَهُۥٓ إِلَيْهِمْ قَالَ أَلَا تَأْكُلُونَ ﴿٢٧﴾}\\
28.\  & \mytextarabic{فَأَوْجَسَ مِنْهُمْ خِيفَةًۭ ۖ قَالُوا۟ لَا تَخَفْ ۖ وَبَشَّرُوهُ بِغُلَـٰمٍ عَلِيمٍۢ ﴿٢٨﴾}\\
29.\  & \mytextarabic{فَأَقْبَلَتِ ٱمْرَأَتُهُۥ فِى صَرَّةٍۢ فَصَكَّتْ وَجْهَهَا وَقَالَتْ عَجُوزٌ عَقِيمٌۭ ﴿٢٩﴾}\\
30.\  & \mytextarabic{قَالُوا۟ كَذَٟلِكِ قَالَ رَبُّكِ ۖ إِنَّهُۥ هُوَ ٱلْحَكِيمُ ٱلْعَلِيمُ ﴿٣٠﴾}\\
31.\  & \mytextarabic{۞ قَالَ فَمَا خَطْبُكُمْ أَيُّهَا ٱلْمُرْسَلُونَ ﴿٣١﴾}\\
32.\  & \mytextarabic{قَالُوٓا۟ إِنَّآ أُرْسِلْنَآ إِلَىٰ قَوْمٍۢ مُّجْرِمِينَ ﴿٣٢﴾}\\
33.\  & \mytextarabic{لِنُرْسِلَ عَلَيْهِمْ حِجَارَةًۭ مِّن طِينٍۢ ﴿٣٣﴾}\\
34.\  & \mytextarabic{مُّسَوَّمَةً عِندَ رَبِّكَ لِلْمُسْرِفِينَ ﴿٣٤﴾}\\
35.\  & \mytextarabic{فَأَخْرَجْنَا مَن كَانَ فِيهَا مِنَ ٱلْمُؤْمِنِينَ ﴿٣٥﴾}\\
36.\  & \mytextarabic{فَمَا وَجَدْنَا فِيهَا غَيْرَ بَيْتٍۢ مِّنَ ٱلْمُسْلِمِينَ ﴿٣٦﴾}\\
37.\  & \mytextarabic{وَتَرَكْنَا فِيهَآ ءَايَةًۭ لِّلَّذِينَ يَخَافُونَ ٱلْعَذَابَ ٱلْأَلِيمَ ﴿٣٧﴾}\\
38.\  & \mytextarabic{وَفِى مُوسَىٰٓ إِذْ أَرْسَلْنَـٰهُ إِلَىٰ فِرْعَوْنَ بِسُلْطَٰنٍۢ مُّبِينٍۢ ﴿٣٨﴾}\\
39.\  & \mytextarabic{فَتَوَلَّىٰ بِرُكْنِهِۦ وَقَالَ سَـٰحِرٌ أَوْ مَجْنُونٌۭ ﴿٣٩﴾}\\
40.\  & \mytextarabic{فَأَخَذْنَـٰهُ وَجُنُودَهُۥ فَنَبَذْنَـٰهُمْ فِى ٱلْيَمِّ وَهُوَ مُلِيمٌۭ ﴿٤٠﴾}\\
41.\  & \mytextarabic{وَفِى عَادٍ إِذْ أَرْسَلْنَا عَلَيْهِمُ ٱلرِّيحَ ٱلْعَقِيمَ ﴿٤١﴾}\\
42.\  & \mytextarabic{مَا تَذَرُ مِن شَىْءٍ أَتَتْ عَلَيْهِ إِلَّا جَعَلَتْهُ كَٱلرَّمِيمِ ﴿٤٢﴾}\\
43.\  & \mytextarabic{وَفِى ثَمُودَ إِذْ قِيلَ لَهُمْ تَمَتَّعُوا۟ حَتَّىٰ حِينٍۢ ﴿٤٣﴾}\\
44.\  & \mytextarabic{فَعَتَوْا۟ عَنْ أَمْرِ رَبِّهِمْ فَأَخَذَتْهُمُ ٱلصَّـٰعِقَةُ وَهُمْ يَنظُرُونَ ﴿٤٤﴾}\\
45.\  & \mytextarabic{فَمَا ٱسْتَطَٰعُوا۟ مِن قِيَامٍۢ وَمَا كَانُوا۟ مُنتَصِرِينَ ﴿٤٥﴾}\\
46.\  & \mytextarabic{وَقَوْمَ نُوحٍۢ مِّن قَبْلُ ۖ إِنَّهُمْ كَانُوا۟ قَوْمًۭا فَـٰسِقِينَ ﴿٤٦﴾}\\
47.\  & \mytextarabic{وَٱلسَّمَآءَ بَنَيْنَـٰهَا بِأَيْي۟دٍۢ وَإِنَّا لَمُوسِعُونَ ﴿٤٧﴾}\\
48.\  & \mytextarabic{وَٱلْأَرْضَ فَرَشْنَـٰهَا فَنِعْمَ ٱلْمَـٰهِدُونَ ﴿٤٨﴾}\\
49.\  & \mytextarabic{وَمِن كُلِّ شَىْءٍ خَلَقْنَا زَوْجَيْنِ لَعَلَّكُمْ تَذَكَّرُونَ ﴿٤٩﴾}\\
50.\  & \mytextarabic{فَفِرُّوٓا۟ إِلَى ٱللَّهِ ۖ إِنِّى لَكُم مِّنْهُ نَذِيرٌۭ مُّبِينٌۭ ﴿٥٠﴾}\\
51.\  & \mytextarabic{وَلَا تَجْعَلُوا۟ مَعَ ٱللَّهِ إِلَـٰهًا ءَاخَرَ ۖ إِنِّى لَكُم مِّنْهُ نَذِيرٌۭ مُّبِينٌۭ ﴿٥١﴾}\\
52.\  & \mytextarabic{كَذَٟلِكَ مَآ أَتَى ٱلَّذِينَ مِن قَبْلِهِم مِّن رَّسُولٍ إِلَّا قَالُوا۟ سَاحِرٌ أَوْ مَجْنُونٌ ﴿٥٢﴾}\\
53.\  & \mytextarabic{أَتَوَاصَوْا۟ بِهِۦ ۚ بَلْ هُمْ قَوْمٌۭ طَاغُونَ ﴿٥٣﴾}\\
54.\  & \mytextarabic{فَتَوَلَّ عَنْهُمْ فَمَآ أَنتَ بِمَلُومٍۢ ﴿٥٤﴾}\\
55.\  & \mytextarabic{وَذَكِّرْ فَإِنَّ ٱلذِّكْرَىٰ تَنفَعُ ٱلْمُؤْمِنِينَ ﴿٥٥﴾}\\
56.\  & \mytextarabic{وَمَا خَلَقْتُ ٱلْجِنَّ وَٱلْإِنسَ إِلَّا لِيَعْبُدُونِ ﴿٥٦﴾}\\
57.\  & \mytextarabic{مَآ أُرِيدُ مِنْهُم مِّن رِّزْقٍۢ وَمَآ أُرِيدُ أَن يُطْعِمُونِ ﴿٥٧﴾}\\
58.\  & \mytextarabic{إِنَّ ٱللَّهَ هُوَ ٱلرَّزَّاقُ ذُو ٱلْقُوَّةِ ٱلْمَتِينُ ﴿٥٨﴾}\\
59.\  & \mytextarabic{فَإِنَّ لِلَّذِينَ ظَلَمُوا۟ ذَنُوبًۭا مِّثْلَ ذَنُوبِ أَصْحَـٰبِهِمْ فَلَا يَسْتَعْجِلُونِ ﴿٥٩﴾}\\
60.\  & \mytextarabic{فَوَيْلٌۭ لِّلَّذِينَ كَفَرُوا۟ مِن يَوْمِهِمُ ٱلَّذِى يُوعَدُونَ ﴿٦٠﴾}\\
\end{longtable}
