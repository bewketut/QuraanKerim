ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ ٱلرَّحْمَـٰنُ ﴿١﴾}\\
2.\  & \mytextarabic{عَلَّمَ ٱلْقُرْءَانَ ﴿٢﴾}\\
3.\  & \mytextarabic{خَلَقَ ٱلْإِنسَـٰنَ ﴿٣﴾}\\
4.\  & \mytextarabic{عَلَّمَهُ ٱلْبَيَانَ ﴿٤﴾}\\
5.\  & \mytextarabic{ٱلشَّمْسُ وَٱلْقَمَرُ بِحُسْبَانٍۢ ﴿٥﴾}\\
6.\  & \mytextarabic{وَٱلنَّجْمُ وَٱلشَّجَرُ يَسْجُدَانِ ﴿٦﴾}\\
7.\  & \mytextarabic{وَٱلسَّمَآءَ رَفَعَهَا وَوَضَعَ ٱلْمِيزَانَ ﴿٧﴾}\\
8.\  & \mytextarabic{أَلَّا تَطْغَوْا۟ فِى ٱلْمِيزَانِ ﴿٨﴾}\\
9.\  & \mytextarabic{وَأَقِيمُوا۟ ٱلْوَزْنَ بِٱلْقِسْطِ وَلَا تُخْسِرُوا۟ ٱلْمِيزَانَ ﴿٩﴾}\\
10.\  & \mytextarabic{وَٱلْأَرْضَ وَضَعَهَا لِلْأَنَامِ ﴿١٠﴾}\\
11.\  & \mytextarabic{فِيهَا فَـٰكِهَةٌۭ وَٱلنَّخْلُ ذَاتُ ٱلْأَكْمَامِ ﴿١١﴾}\\
12.\  & \mytextarabic{وَٱلْحَبُّ ذُو ٱلْعَصْفِ وَٱلرَّيْحَانُ ﴿١٢﴾}\\
13.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿١٣﴾}\\
14.\  & \mytextarabic{خَلَقَ ٱلْإِنسَـٰنَ مِن صَلْصَـٰلٍۢ كَٱلْفَخَّارِ ﴿١٤﴾}\\
15.\  & \mytextarabic{وَخَلَقَ ٱلْجَآنَّ مِن مَّارِجٍۢ مِّن نَّارٍۢ ﴿١٥﴾}\\
16.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿١٦﴾}\\
17.\  & \mytextarabic{رَبُّ ٱلْمَشْرِقَيْنِ وَرَبُّ ٱلْمَغْرِبَيْنِ ﴿١٧﴾}\\
18.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿١٨﴾}\\
19.\  & \mytextarabic{مَرَجَ ٱلْبَحْرَيْنِ يَلْتَقِيَانِ ﴿١٩﴾}\\
20.\  & \mytextarabic{بَيْنَهُمَا بَرْزَخٌۭ لَّا يَبْغِيَانِ ﴿٢٠﴾}\\
21.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٢١﴾}\\
22.\  & \mytextarabic{يَخْرُجُ مِنْهُمَا ٱللُّؤْلُؤُ وَٱلْمَرْجَانُ ﴿٢٢﴾}\\
23.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٢٣﴾}\\
24.\  & \mytextarabic{وَلَهُ ٱلْجَوَارِ ٱلْمُنشَـَٔاتُ فِى ٱلْبَحْرِ كَٱلْأَعْلَـٰمِ ﴿٢٤﴾}\\
25.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٢٥﴾}\\
26.\  & \mytextarabic{كُلُّ مَنْ عَلَيْهَا فَانٍۢ ﴿٢٦﴾}\\
27.\  & \mytextarabic{وَيَبْقَىٰ وَجْهُ رَبِّكَ ذُو ٱلْجَلَـٰلِ وَٱلْإِكْرَامِ ﴿٢٧﴾}\\
28.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٢٨﴾}\\
29.\  & \mytextarabic{يَسْـَٔلُهُۥ مَن فِى ٱلسَّمَـٰوَٟتِ وَٱلْأَرْضِ ۚ كُلَّ يَوْمٍ هُوَ فِى شَأْنٍۢ ﴿٢٩﴾}\\
30.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٣٠﴾}\\
31.\  & \mytextarabic{سَنَفْرُغُ لَكُمْ أَيُّهَ ٱلثَّقَلَانِ ﴿٣١﴾}\\
32.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٣٢﴾}\\
33.\  & \mytextarabic{يَـٰمَعْشَرَ ٱلْجِنِّ وَٱلْإِنسِ إِنِ ٱسْتَطَعْتُمْ أَن تَنفُذُوا۟ مِنْ أَقْطَارِ ٱلسَّمَـٰوَٟتِ وَٱلْأَرْضِ فَٱنفُذُوا۟ ۚ لَا تَنفُذُونَ إِلَّا بِسُلْطَٰنٍۢ ﴿٣٣﴾}\\
34.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٣٤﴾}\\
35.\  & \mytextarabic{يُرْسَلُ عَلَيْكُمَا شُوَاظٌۭ مِّن نَّارٍۢ وَنُحَاسٌۭ فَلَا تَنتَصِرَانِ ﴿٣٥﴾}\\
36.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٣٦﴾}\\
37.\  & \mytextarabic{فَإِذَا ٱنشَقَّتِ ٱلسَّمَآءُ فَكَانَتْ وَرْدَةًۭ كَٱلدِّهَانِ ﴿٣٧﴾}\\
38.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٣٨﴾}\\
39.\  & \mytextarabic{فَيَوْمَئِذٍۢ لَّا يُسْـَٔلُ عَن ذَنۢبِهِۦٓ إِنسٌۭ وَلَا جَآنٌّۭ ﴿٣٩﴾}\\
40.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٤٠﴾}\\
41.\  & \mytextarabic{يُعْرَفُ ٱلْمُجْرِمُونَ بِسِيمَـٰهُمْ فَيُؤْخَذُ بِٱلنَّوَٟصِى وَٱلْأَقْدَامِ ﴿٤١﴾}\\
42.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٤٢﴾}\\
43.\  & \mytextarabic{هَـٰذِهِۦ جَهَنَّمُ ٱلَّتِى يُكَذِّبُ بِهَا ٱلْمُجْرِمُونَ ﴿٤٣﴾}\\
44.\  & \mytextarabic{يَطُوفُونَ بَيْنَهَا وَبَيْنَ حَمِيمٍ ءَانٍۢ ﴿٤٤﴾}\\
45.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٤٥﴾}\\
46.\  & \mytextarabic{وَلِمَنْ خَافَ مَقَامَ رَبِّهِۦ جَنَّتَانِ ﴿٤٦﴾}\\
47.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٤٧﴾}\\
48.\  & \mytextarabic{ذَوَاتَآ أَفْنَانٍۢ ﴿٤٨﴾}\\
49.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٤٩﴾}\\
50.\  & \mytextarabic{فِيهِمَا عَيْنَانِ تَجْرِيَانِ ﴿٥٠﴾}\\
51.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٥١﴾}\\
52.\  & \mytextarabic{فِيهِمَا مِن كُلِّ فَـٰكِهَةٍۢ زَوْجَانِ ﴿٥٢﴾}\\
53.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٥٣﴾}\\
54.\  & \mytextarabic{مُتَّكِـِٔينَ عَلَىٰ فُرُشٍۭ بَطَآئِنُهَا مِنْ إِسْتَبْرَقٍۢ ۚ وَجَنَى ٱلْجَنَّتَيْنِ دَانٍۢ ﴿٥٤﴾}\\
55.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٥٥﴾}\\
56.\  & \mytextarabic{فِيهِنَّ قَـٰصِرَٰتُ ٱلطَّرْفِ لَمْ يَطْمِثْهُنَّ إِنسٌۭ قَبْلَهُمْ وَلَا جَآنٌّۭ ﴿٥٦﴾}\\
57.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٥٧﴾}\\
58.\  & \mytextarabic{كَأَنَّهُنَّ ٱلْيَاقُوتُ وَٱلْمَرْجَانُ ﴿٥٨﴾}\\
59.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٥٩﴾}\\
60.\  & \mytextarabic{هَلْ جَزَآءُ ٱلْإِحْسَـٰنِ إِلَّا ٱلْإِحْسَـٰنُ ﴿٦٠﴾}\\
61.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٦١﴾}\\
62.\  & \mytextarabic{وَمِن دُونِهِمَا جَنَّتَانِ ﴿٦٢﴾}\\
63.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٦٣﴾}\\
64.\  & \mytextarabic{مُدْهَآمَّتَانِ ﴿٦٤﴾}\\
65.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٦٥﴾}\\
66.\  & \mytextarabic{فِيهِمَا عَيْنَانِ نَضَّاخَتَانِ ﴿٦٦﴾}\\
67.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٦٧﴾}\\
68.\  & \mytextarabic{فِيهِمَا فَـٰكِهَةٌۭ وَنَخْلٌۭ وَرُمَّانٌۭ ﴿٦٨﴾}\\
69.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٦٩﴾}\\
70.\  & \mytextarabic{فِيهِنَّ خَيْرَٰتٌ حِسَانٌۭ ﴿٧٠﴾}\\
71.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٧١﴾}\\
72.\  & \mytextarabic{حُورٌۭ مَّقْصُورَٰتٌۭ فِى ٱلْخِيَامِ ﴿٧٢﴾}\\
73.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٧٣﴾}\\
74.\  & \mytextarabic{لَمْ يَطْمِثْهُنَّ إِنسٌۭ قَبْلَهُمْ وَلَا جَآنٌّۭ ﴿٧٤﴾}\\
75.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٧٥﴾}\\
76.\  & \mytextarabic{مُتَّكِـِٔينَ عَلَىٰ رَفْرَفٍ خُضْرٍۢ وَعَبْقَرِىٍّ حِسَانٍۢ ﴿٧٦﴾}\\
77.\  & \mytextarabic{فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿٧٧﴾}\\
78.\  & \mytextarabic{تَبَٰرَكَ ٱسْمُ رَبِّكَ ذِى ٱلْجَلَـٰلِ وَٱلْإِكْرَامِ ﴿٧٨﴾}\\
\end{longtable}
