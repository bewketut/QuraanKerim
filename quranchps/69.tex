\begin{center}\section{ሱራቱ አልሀቃ -  \textarabic{سوره  الحاقة}}\end{center}
\begin{longtable}{%
  @{}
    p{.5\textwidth}
  @{~~~}
    p{.5\textwidth}
    @{}
}
ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ ٱلْحَآقَّةُ ﴿١﴾}\\
2.\  & \mytextarabic{مَا ٱلْحَآقَّةُ ﴿٢﴾}\\
3.\  & \mytextarabic{وَمَآ أَدْرَىٰكَ مَا ٱلْحَآقَّةُ ﴿٣﴾}\\
4.\  & \mytextarabic{كَذَّبَتْ ثَمُودُ وَعَادٌۢ بِٱلْقَارِعَةِ ﴿٤﴾}\\
5.\  & \mytextarabic{فَأَمَّا ثَمُودُ فَأُهْلِكُوا۟ بِٱلطَّاغِيَةِ ﴿٥﴾}\\
6.\  & \mytextarabic{وَأَمَّا عَادٌۭ فَأُهْلِكُوا۟ بِرِيحٍۢ صَرْصَرٍ عَاتِيَةٍۢ ﴿٦﴾}\\
7.\  & \mytextarabic{سَخَّرَهَا عَلَيْهِمْ سَبْعَ لَيَالٍۢ وَثَمَـٰنِيَةَ أَيَّامٍ حُسُومًۭا فَتَرَى ٱلْقَوْمَ فِيهَا صَرْعَىٰ كَأَنَّهُمْ أَعْجَازُ نَخْلٍ خَاوِيَةٍۢ ﴿٧﴾}\\
8.\  & \mytextarabic{فَهَلْ تَرَىٰ لَهُم مِّنۢ بَاقِيَةٍۢ ﴿٨﴾}\\
9.\  & \mytextarabic{وَجَآءَ فِرْعَوْنُ وَمَن قَبْلَهُۥ وَٱلْمُؤْتَفِكَـٰتُ بِٱلْخَاطِئَةِ ﴿٩﴾}\\
10.\  & \mytextarabic{فَعَصَوْا۟ رَسُولَ رَبِّهِمْ فَأَخَذَهُمْ أَخْذَةًۭ رَّابِيَةً ﴿١٠﴾}\\
11.\  & \mytextarabic{إِنَّا لَمَّا طَغَا ٱلْمَآءُ حَمَلْنَـٰكُمْ فِى ٱلْجَارِيَةِ ﴿١١﴾}\\
12.\  & \mytextarabic{لِنَجْعَلَهَا لَكُمْ تَذْكِرَةًۭ وَتَعِيَهَآ أُذُنٌۭ وَٟعِيَةٌۭ ﴿١٢﴾}\\
13.\  & \mytextarabic{فَإِذَا نُفِخَ فِى ٱلصُّورِ نَفْخَةٌۭ وَٟحِدَةٌۭ ﴿١٣﴾}\\
14.\  & \mytextarabic{وَحُمِلَتِ ٱلْأَرْضُ وَٱلْجِبَالُ فَدُكَّتَا دَكَّةًۭ وَٟحِدَةًۭ ﴿١٤﴾}\\
15.\  & \mytextarabic{فَيَوْمَئِذٍۢ وَقَعَتِ ٱلْوَاقِعَةُ ﴿١٥﴾}\\
16.\  & \mytextarabic{وَٱنشَقَّتِ ٱلسَّمَآءُ فَهِىَ يَوْمَئِذٍۢ وَاهِيَةٌۭ ﴿١٦﴾}\\
17.\  & \mytextarabic{وَٱلْمَلَكُ عَلَىٰٓ أَرْجَآئِهَا ۚ وَيَحْمِلُ عَرْشَ رَبِّكَ فَوْقَهُمْ يَوْمَئِذٍۢ ثَمَـٰنِيَةٌۭ ﴿١٧﴾}\\
18.\  & \mytextarabic{يَوْمَئِذٍۢ تُعْرَضُونَ لَا تَخْفَىٰ مِنكُمْ خَافِيَةٌۭ ﴿١٨﴾}\\
19.\  & \mytextarabic{فَأَمَّا مَنْ أُوتِىَ كِتَـٰبَهُۥ بِيَمِينِهِۦ فَيَقُولُ هَآؤُمُ ٱقْرَءُوا۟ كِتَـٰبِيَهْ ﴿١٩﴾}\\
20.\  & \mytextarabic{إِنِّى ظَنَنتُ أَنِّى مُلَـٰقٍ حِسَابِيَهْ ﴿٢٠﴾}\\
21.\  & \mytextarabic{فَهُوَ فِى عِيشَةٍۢ رَّاضِيَةٍۢ ﴿٢١﴾}\\
22.\  & \mytextarabic{فِى جَنَّةٍ عَالِيَةٍۢ ﴿٢٢﴾}\\
23.\  & \mytextarabic{قُطُوفُهَا دَانِيَةٌۭ ﴿٢٣﴾}\\
24.\  & \mytextarabic{كُلُوا۟ وَٱشْرَبُوا۟ هَنِيٓـًٔۢا بِمَآ أَسْلَفْتُمْ فِى ٱلْأَيَّامِ ٱلْخَالِيَةِ ﴿٢٤﴾}\\
25.\  & \mytextarabic{وَأَمَّا مَنْ أُوتِىَ كِتَـٰبَهُۥ بِشِمَالِهِۦ فَيَقُولُ يَـٰلَيْتَنِى لَمْ أُوتَ كِتَـٰبِيَهْ ﴿٢٥﴾}\\
26.\  & \mytextarabic{وَلَمْ أَدْرِ مَا حِسَابِيَهْ ﴿٢٦﴾}\\
27.\  & \mytextarabic{يَـٰلَيْتَهَا كَانَتِ ٱلْقَاضِيَةَ ﴿٢٧﴾}\\
28.\  & \mytextarabic{مَآ أَغْنَىٰ عَنِّى مَالِيَهْ ۜ ﴿٢٨﴾}\\
29.\  & \mytextarabic{هَلَكَ عَنِّى سُلْطَٰنِيَهْ ﴿٢٩﴾}\\
30.\  & \mytextarabic{خُذُوهُ فَغُلُّوهُ ﴿٣٠﴾}\\
31.\  & \mytextarabic{ثُمَّ ٱلْجَحِيمَ صَلُّوهُ ﴿٣١﴾}\\
32.\  & \mytextarabic{ثُمَّ فِى سِلْسِلَةٍۢ ذَرْعُهَا سَبْعُونَ ذِرَاعًۭا فَٱسْلُكُوهُ ﴿٣٢﴾}\\
33.\  & \mytextarabic{إِنَّهُۥ كَانَ لَا يُؤْمِنُ بِٱللَّهِ ٱلْعَظِيمِ ﴿٣٣﴾}\\
34.\  & \mytextarabic{وَلَا يَحُضُّ عَلَىٰ طَعَامِ ٱلْمِسْكِينِ ﴿٣٤﴾}\\
35.\  & \mytextarabic{فَلَيْسَ لَهُ ٱلْيَوْمَ هَـٰهُنَا حَمِيمٌۭ ﴿٣٥﴾}\\
36.\  & \mytextarabic{وَلَا طَعَامٌ إِلَّا مِنْ غِسْلِينٍۢ ﴿٣٦﴾}\\
37.\  & \mytextarabic{لَّا يَأْكُلُهُۥٓ إِلَّا ٱلْخَـٰطِـُٔونَ ﴿٣٧﴾}\\
38.\  & \mytextarabic{فَلَآ أُقْسِمُ بِمَا تُبْصِرُونَ ﴿٣٨﴾}\\
39.\  & \mytextarabic{وَمَا لَا تُبْصِرُونَ ﴿٣٩﴾}\\
40.\  & \mytextarabic{إِنَّهُۥ لَقَوْلُ رَسُولٍۢ كَرِيمٍۢ ﴿٤٠﴾}\\
41.\  & \mytextarabic{وَمَا هُوَ بِقَوْلِ شَاعِرٍۢ ۚ قَلِيلًۭا مَّا تُؤْمِنُونَ ﴿٤١﴾}\\
42.\  & \mytextarabic{وَلَا بِقَوْلِ كَاهِنٍۢ ۚ قَلِيلًۭا مَّا تَذَكَّرُونَ ﴿٤٢﴾}\\
43.\  & \mytextarabic{تَنزِيلٌۭ مِّن رَّبِّ ٱلْعَـٰلَمِينَ ﴿٤٣﴾}\\
44.\  & \mytextarabic{وَلَوْ تَقَوَّلَ عَلَيْنَا بَعْضَ ٱلْأَقَاوِيلِ ﴿٤٤﴾}\\
45.\  & \mytextarabic{لَأَخَذْنَا مِنْهُ بِٱلْيَمِينِ ﴿٤٥﴾}\\
46.\  & \mytextarabic{ثُمَّ لَقَطَعْنَا مِنْهُ ٱلْوَتِينَ ﴿٤٦﴾}\\
47.\  & \mytextarabic{فَمَا مِنكُم مِّنْ أَحَدٍ عَنْهُ حَـٰجِزِينَ ﴿٤٧﴾}\\
48.\  & \mytextarabic{وَإِنَّهُۥ لَتَذْكِرَةٌۭ لِّلْمُتَّقِينَ ﴿٤٨﴾}\\
49.\  & \mytextarabic{وَإِنَّا لَنَعْلَمُ أَنَّ مِنكُم مُّكَذِّبِينَ ﴿٤٩﴾}\\
50.\  & \mytextarabic{وَإِنَّهُۥ لَحَسْرَةٌ عَلَى ٱلْكَـٰفِرِينَ ﴿٥٠﴾}\\
51.\  & \mytextarabic{وَإِنَّهُۥ لَحَقُّ ٱلْيَقِينِ ﴿٥١﴾}\\
52.\  & \mytextarabic{فَسَبِّحْ بِٱسْمِ رَبِّكَ ٱلْعَظِيمِ ﴿٥٢﴾}\\
\end{longtable}
