\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ وَٱلْعَـٰدِيَـٰتِ ضَبْحًۭا ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{فَٱلْمُورِيَـٰتِ قَدْحًۭا ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{فَٱلْمُغِيرَٰتِ صُبْحًۭا ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{فَأَثَرْنَ بِهِۦ نَقْعًۭا ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{فَوَسَطْنَ بِهِۦ جَمْعًا ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{إِنَّ ٱلْإِنسَـٰنَ لِرَبِّهِۦ لَكَنُودٌۭ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{وَإِنَّهُۥ عَلَىٰ ذَٟلِكَ لَشَهِيدٌۭ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{وَإِنَّهُۥ لِحُبِّ ٱلْخَيْرِ لَشَدِيدٌ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{۞ أَفَلَا يَعْلَمُ إِذَا بُعْثِرَ مَا فِى ٱلْقُبُورِ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{وَحُصِّلَ مَا فِى ٱلصُّدُورِ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{إِنَّ رَبَّهُم بِهِمْ يَوْمَئِذٍۢ لَّخَبِيرٌۢ ﴿١١﴾}\\
\end{longtable}
