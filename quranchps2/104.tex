\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ وَيْلٌۭ لِّكُلِّ هُمَزَةٍۢ لُّمَزَةٍ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{ٱلَّذِى جَمَعَ مَالًۭا وَعَدَّدَهُۥ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{يَحْسَبُ أَنَّ مَالَهُۥٓ أَخْلَدَهُۥ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{كَلَّا ۖ لَيُنۢبَذَنَّ فِى ٱلْحُطَمَةِ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{وَمَآ أَدْرَىٰكَ مَا ٱلْحُطَمَةُ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{نَارُ ٱللَّهِ ٱلْمُوقَدَةُ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{ٱلَّتِى تَطَّلِعُ عَلَى ٱلْأَفْـِٔدَةِ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{إِنَّهَا عَلَيْهِم مُّؤْصَدَةٌۭ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{فِى عَمَدٍۢ مُّمَدَّدَةٍۭ ﴿٩﴾}\\
\end{longtable}
