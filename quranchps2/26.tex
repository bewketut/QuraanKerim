\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ طسٓمٓ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{تِلْكَ ءَايَـٰتُ ٱلْكِتَـٰبِ ٱلْمُبِينِ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{لَعَلَّكَ بَٰخِعٌۭ نَّفْسَكَ أَلَّا يَكُونُوا۟ مُؤْمِنِينَ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{إِن نَّشَأْ نُنَزِّلْ عَلَيْهِم مِّنَ ٱلسَّمَآءِ ءَايَةًۭ فَظَلَّتْ أَعْنَـٰقُهُمْ لَهَا خَـٰضِعِينَ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{وَمَا يَأْتِيهِم مِّن ذِكْرٍۢ مِّنَ ٱلرَّحْمَـٰنِ مُحْدَثٍ إِلَّا كَانُوا۟ عَنْهُ مُعْرِضِينَ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{فَقَدْ كَذَّبُوا۟ فَسَيَأْتِيهِمْ أَنۢبَٰٓؤُا۟ مَا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{أَوَلَمْ يَرَوْا۟ إِلَى ٱلْأَرْضِ كَمْ أَنۢبَتْنَا فِيهَا مِن كُلِّ زَوْجٍۢ كَرِيمٍ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{إِنَّ فِى ذَٟلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{وَإِذْ نَادَىٰ رَبُّكَ مُوسَىٰٓ أَنِ ٱئْتِ ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{قَوْمَ فِرْعَوْنَ ۚ أَلَا يَتَّقُونَ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{قَالَ رَبِّ إِنِّىٓ أَخَافُ أَن يُكَذِّبُونِ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{وَيَضِيقُ صَدْرِى وَلَا يَنطَلِقُ لِسَانِى فَأَرْسِلْ إِلَىٰ هَـٰرُونَ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{وَلَهُمْ عَلَىَّ ذَنۢبٌۭ فَأَخَافُ أَن يَقْتُلُونِ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{قَالَ كَلَّا ۖ فَٱذْهَبَا بِـَٔايَـٰتِنَآ ۖ إِنَّا مَعَكُم مُّسْتَمِعُونَ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{فَأْتِيَا فِرْعَوْنَ فَقُولَآ إِنَّا رَسُولُ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{أَنْ أَرْسِلْ مَعَنَا بَنِىٓ إِسْرَٰٓءِيلَ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{قَالَ أَلَمْ نُرَبِّكَ فِينَا وَلِيدًۭا وَلَبِثْتَ فِينَا مِنْ عُمُرِكَ سِنِينَ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{وَفَعَلْتَ فَعْلَتَكَ ٱلَّتِى فَعَلْتَ وَأَنتَ مِنَ ٱلْكَـٰفِرِينَ ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{قَالَ فَعَلْتُهَآ إِذًۭا وَأَنَا۠ مِنَ ٱلضَّآلِّينَ ﴿٢٠﴾}\\
\textamh{21.\  } & \mytextarabic{فَفَرَرْتُ مِنكُمْ لَمَّا خِفْتُكُمْ فَوَهَبَ لِى رَبِّى حُكْمًۭا وَجَعَلَنِى مِنَ ٱلْمُرْسَلِينَ ﴿٢١﴾}\\
\textamh{22.\  } & \mytextarabic{وَتِلْكَ نِعْمَةٌۭ تَمُنُّهَا عَلَىَّ أَنْ عَبَّدتَّ بَنِىٓ إِسْرَٰٓءِيلَ ﴿٢٢﴾}\\
\textamh{23.\  } & \mytextarabic{قَالَ فِرْعَوْنُ وَمَا رَبُّ ٱلْعَـٰلَمِينَ ﴿٢٣﴾}\\
\textamh{24.\  } & \mytextarabic{قَالَ رَبُّ ٱلسَّمَـٰوَٟتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَآ ۖ إِن كُنتُم مُّوقِنِينَ ﴿٢٤﴾}\\
\textamh{25.\  } & \mytextarabic{قَالَ لِمَنْ حَوْلَهُۥٓ أَلَا تَسْتَمِعُونَ ﴿٢٥﴾}\\
\textamh{26.\  } & \mytextarabic{قَالَ رَبُّكُمْ وَرَبُّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿٢٦﴾}\\
\textamh{27.\  } & \mytextarabic{قَالَ إِنَّ رَسُولَكُمُ ٱلَّذِىٓ أُرْسِلَ إِلَيْكُمْ لَمَجْنُونٌۭ ﴿٢٧﴾}\\
\textamh{28.\  } & \mytextarabic{قَالَ رَبُّ ٱلْمَشْرِقِ وَٱلْمَغْرِبِ وَمَا بَيْنَهُمَآ ۖ إِن كُنتُمْ تَعْقِلُونَ ﴿٢٨﴾}\\
\textamh{29.\  } & \mytextarabic{قَالَ لَئِنِ ٱتَّخَذْتَ إِلَـٰهًا غَيْرِى لَأَجْعَلَنَّكَ مِنَ ٱلْمَسْجُونِينَ ﴿٢٩﴾}\\
\textamh{30.\  } & \mytextarabic{قَالَ أَوَلَوْ جِئْتُكَ بِشَىْءٍۢ مُّبِينٍۢ ﴿٣٠﴾}\\
\textamh{31.\  } & \mytextarabic{قَالَ فَأْتِ بِهِۦٓ إِن كُنتَ مِنَ ٱلصَّـٰدِقِينَ ﴿٣١﴾}\\
\textamh{32.\  } & \mytextarabic{فَأَلْقَىٰ عَصَاهُ فَإِذَا هِىَ ثُعْبَانٌۭ مُّبِينٌۭ ﴿٣٢﴾}\\
\textamh{33.\  } & \mytextarabic{وَنَزَعَ يَدَهُۥ فَإِذَا هِىَ بَيْضَآءُ لِلنَّـٰظِرِينَ ﴿٣٣﴾}\\
\textamh{34.\  } & \mytextarabic{قَالَ لِلْمَلَإِ حَوْلَهُۥٓ إِنَّ هَـٰذَا لَسَـٰحِرٌ عَلِيمٌۭ ﴿٣٤﴾}\\
\textamh{35.\  } & \mytextarabic{يُرِيدُ أَن يُخْرِجَكُم مِّنْ أَرْضِكُم بِسِحْرِهِۦ فَمَاذَا تَأْمُرُونَ ﴿٣٥﴾}\\
\textamh{36.\  } & \mytextarabic{قَالُوٓا۟ أَرْجِهْ وَأَخَاهُ وَٱبْعَثْ فِى ٱلْمَدَآئِنِ حَـٰشِرِينَ ﴿٣٦﴾}\\
\textamh{37.\  } & \mytextarabic{يَأْتُوكَ بِكُلِّ سَحَّارٍ عَلِيمٍۢ ﴿٣٧﴾}\\
\textamh{38.\  } & \mytextarabic{فَجُمِعَ ٱلسَّحَرَةُ لِمِيقَـٰتِ يَوْمٍۢ مَّعْلُومٍۢ ﴿٣٨﴾}\\
\textamh{39.\  } & \mytextarabic{وَقِيلَ لِلنَّاسِ هَلْ أَنتُم مُّجْتَمِعُونَ ﴿٣٩﴾}\\
\textamh{40.\  } & \mytextarabic{لَعَلَّنَا نَتَّبِعُ ٱلسَّحَرَةَ إِن كَانُوا۟ هُمُ ٱلْغَٰلِبِينَ ﴿٤٠﴾}\\
\textamh{41.\  } & \mytextarabic{فَلَمَّا جَآءَ ٱلسَّحَرَةُ قَالُوا۟ لِفِرْعَوْنَ أَئِنَّ لَنَا لَأَجْرًا إِن كُنَّا نَحْنُ ٱلْغَٰلِبِينَ ﴿٤١﴾}\\
\textamh{42.\  } & \mytextarabic{قَالَ نَعَمْ وَإِنَّكُمْ إِذًۭا لَّمِنَ ٱلْمُقَرَّبِينَ ﴿٤٢﴾}\\
\textamh{43.\  } & \mytextarabic{قَالَ لَهُم مُّوسَىٰٓ أَلْقُوا۟ مَآ أَنتُم مُّلْقُونَ ﴿٤٣﴾}\\
\textamh{44.\  } & \mytextarabic{فَأَلْقَوْا۟ حِبَالَهُمْ وَعِصِيَّهُمْ وَقَالُوا۟ بِعِزَّةِ فِرْعَوْنَ إِنَّا لَنَحْنُ ٱلْغَٰلِبُونَ ﴿٤٤﴾}\\
\textamh{45.\  } & \mytextarabic{فَأَلْقَىٰ مُوسَىٰ عَصَاهُ فَإِذَا هِىَ تَلْقَفُ مَا يَأْفِكُونَ ﴿٤٥﴾}\\
\textamh{46.\  } & \mytextarabic{فَأُلْقِىَ ٱلسَّحَرَةُ سَـٰجِدِينَ ﴿٤٦﴾}\\
\textamh{47.\  } & \mytextarabic{قَالُوٓا۟ ءَامَنَّا بِرَبِّ ٱلْعَـٰلَمِينَ ﴿٤٧﴾}\\
\textamh{48.\  } & \mytextarabic{رَبِّ مُوسَىٰ وَهَـٰرُونَ ﴿٤٨﴾}\\
\textamh{49.\  } & \mytextarabic{قَالَ ءَامَنتُمْ لَهُۥ قَبْلَ أَنْ ءَاذَنَ لَكُمْ ۖ إِنَّهُۥ لَكَبِيرُكُمُ ٱلَّذِى عَلَّمَكُمُ ٱلسِّحْرَ فَلَسَوْفَ تَعْلَمُونَ ۚ لَأُقَطِّعَنَّ أَيْدِيَكُمْ وَأَرْجُلَكُم مِّنْ خِلَـٰفٍۢ وَلَأُصَلِّبَنَّكُمْ أَجْمَعِينَ ﴿٤٩﴾}\\
\textamh{50.\  } & \mytextarabic{قَالُوا۟ لَا ضَيْرَ ۖ إِنَّآ إِلَىٰ رَبِّنَا مُنقَلِبُونَ ﴿٥٠﴾}\\
\textamh{51.\  } & \mytextarabic{إِنَّا نَطْمَعُ أَن يَغْفِرَ لَنَا رَبُّنَا خَطَٰيَـٰنَآ أَن كُنَّآ أَوَّلَ ٱلْمُؤْمِنِينَ ﴿٥١﴾}\\
\textamh{52.\  } & \mytextarabic{۞ وَأَوْحَيْنَآ إِلَىٰ مُوسَىٰٓ أَنْ أَسْرِ بِعِبَادِىٓ إِنَّكُم مُّتَّبَعُونَ ﴿٥٢﴾}\\
\textamh{53.\  } & \mytextarabic{فَأَرْسَلَ فِرْعَوْنُ فِى ٱلْمَدَآئِنِ حَـٰشِرِينَ ﴿٥٣﴾}\\
\textamh{54.\  } & \mytextarabic{إِنَّ هَـٰٓؤُلَآءِ لَشِرْذِمَةٌۭ قَلِيلُونَ ﴿٥٤﴾}\\
\textamh{55.\  } & \mytextarabic{وَإِنَّهُمْ لَنَا لَغَآئِظُونَ ﴿٥٥﴾}\\
\textamh{56.\  } & \mytextarabic{وَإِنَّا لَجَمِيعٌ حَـٰذِرُونَ ﴿٥٦﴾}\\
\textamh{57.\  } & \mytextarabic{فَأَخْرَجْنَـٰهُم مِّن جَنَّـٰتٍۢ وَعُيُونٍۢ ﴿٥٧﴾}\\
\textamh{58.\  } & \mytextarabic{وَكُنُوزٍۢ وَمَقَامٍۢ كَرِيمٍۢ ﴿٥٨﴾}\\
\textamh{59.\  } & \mytextarabic{كَذَٟلِكَ وَأَوْرَثْنَـٰهَا بَنِىٓ إِسْرَٰٓءِيلَ ﴿٥٩﴾}\\
\textamh{60.\  } & \mytextarabic{فَأَتْبَعُوهُم مُّشْرِقِينَ ﴿٦٠﴾}\\
\textamh{61.\  } & \mytextarabic{فَلَمَّا تَرَٰٓءَا ٱلْجَمْعَانِ قَالَ أَصْحَـٰبُ مُوسَىٰٓ إِنَّا لَمُدْرَكُونَ ﴿٦١﴾}\\
\textamh{62.\  } & \mytextarabic{قَالَ كَلَّآ ۖ إِنَّ مَعِىَ رَبِّى سَيَهْدِينِ ﴿٦٢﴾}\\
\textamh{63.\  } & \mytextarabic{فَأَوْحَيْنَآ إِلَىٰ مُوسَىٰٓ أَنِ ٱضْرِب بِّعَصَاكَ ٱلْبَحْرَ ۖ فَٱنفَلَقَ فَكَانَ كُلُّ فِرْقٍۢ كَٱلطَّوْدِ ٱلْعَظِيمِ ﴿٦٣﴾}\\
\textamh{64.\  } & \mytextarabic{وَأَزْلَفْنَا ثَمَّ ٱلْءَاخَرِينَ ﴿٦٤﴾}\\
\textamh{65.\  } & \mytextarabic{وَأَنجَيْنَا مُوسَىٰ وَمَن مَّعَهُۥٓ أَجْمَعِينَ ﴿٦٥﴾}\\
\textamh{66.\  } & \mytextarabic{ثُمَّ أَغْرَقْنَا ٱلْءَاخَرِينَ ﴿٦٦﴾}\\
\textamh{67.\  } & \mytextarabic{إِنَّ فِى ذَٟلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿٦٧﴾}\\
\textamh{68.\  } & \mytextarabic{وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿٦٨﴾}\\
\textamh{69.\  } & \mytextarabic{وَٱتْلُ عَلَيْهِمْ نَبَأَ إِبْرَٰهِيمَ ﴿٦٩﴾}\\
\textamh{70.\  } & \mytextarabic{إِذْ قَالَ لِأَبِيهِ وَقَوْمِهِۦ مَا تَعْبُدُونَ ﴿٧٠﴾}\\
\textamh{71.\  } & \mytextarabic{قَالُوا۟ نَعْبُدُ أَصْنَامًۭا فَنَظَلُّ لَهَا عَـٰكِفِينَ ﴿٧١﴾}\\
\textamh{72.\  } & \mytextarabic{قَالَ هَلْ يَسْمَعُونَكُمْ إِذْ تَدْعُونَ ﴿٧٢﴾}\\
\textamh{73.\  } & \mytextarabic{أَوْ يَنفَعُونَكُمْ أَوْ يَضُرُّونَ ﴿٧٣﴾}\\
\textamh{74.\  } & \mytextarabic{قَالُوا۟ بَلْ وَجَدْنَآ ءَابَآءَنَا كَذَٟلِكَ يَفْعَلُونَ ﴿٧٤﴾}\\
\textamh{75.\  } & \mytextarabic{قَالَ أَفَرَءَيْتُم مَّا كُنتُمْ تَعْبُدُونَ ﴿٧٥﴾}\\
\textamh{76.\  } & \mytextarabic{أَنتُمْ وَءَابَآؤُكُمُ ٱلْأَقْدَمُونَ ﴿٧٦﴾}\\
\textamh{77.\  } & \mytextarabic{فَإِنَّهُمْ عَدُوٌّۭ لِّىٓ إِلَّا رَبَّ ٱلْعَـٰلَمِينَ ﴿٧٧﴾}\\
\textamh{78.\  } & \mytextarabic{ٱلَّذِى خَلَقَنِى فَهُوَ يَهْدِينِ ﴿٧٨﴾}\\
\textamh{79.\  } & \mytextarabic{وَٱلَّذِى هُوَ يُطْعِمُنِى وَيَسْقِينِ ﴿٧٩﴾}\\
\textamh{80.\  } & \mytextarabic{وَإِذَا مَرِضْتُ فَهُوَ يَشْفِينِ ﴿٨٠﴾}\\
\textamh{81.\  } & \mytextarabic{وَٱلَّذِى يُمِيتُنِى ثُمَّ يُحْيِينِ ﴿٨١﴾}\\
\textamh{82.\  } & \mytextarabic{وَٱلَّذِىٓ أَطْمَعُ أَن يَغْفِرَ لِى خَطِيٓـَٔتِى يَوْمَ ٱلدِّينِ ﴿٨٢﴾}\\
\textamh{83.\  } & \mytextarabic{رَبِّ هَبْ لِى حُكْمًۭا وَأَلْحِقْنِى بِٱلصَّـٰلِحِينَ ﴿٨٣﴾}\\
\textamh{84.\  } & \mytextarabic{وَٱجْعَل لِّى لِسَانَ صِدْقٍۢ فِى ٱلْءَاخِرِينَ ﴿٨٤﴾}\\
\textamh{85.\  } & \mytextarabic{وَٱجْعَلْنِى مِن وَرَثَةِ جَنَّةِ ٱلنَّعِيمِ ﴿٨٥﴾}\\
\textamh{86.\  } & \mytextarabic{وَٱغْفِرْ لِأَبِىٓ إِنَّهُۥ كَانَ مِنَ ٱلضَّآلِّينَ ﴿٨٦﴾}\\
\textamh{87.\  } & \mytextarabic{وَلَا تُخْزِنِى يَوْمَ يُبْعَثُونَ ﴿٨٧﴾}\\
\textamh{88.\  } & \mytextarabic{يَوْمَ لَا يَنفَعُ مَالٌۭ وَلَا بَنُونَ ﴿٨٨﴾}\\
\textamh{89.\  } & \mytextarabic{إِلَّا مَنْ أَتَى ٱللَّهَ بِقَلْبٍۢ سَلِيمٍۢ ﴿٨٩﴾}\\
\textamh{90.\  } & \mytextarabic{وَأُزْلِفَتِ ٱلْجَنَّةُ لِلْمُتَّقِينَ ﴿٩٠﴾}\\
\textamh{91.\  } & \mytextarabic{وَبُرِّزَتِ ٱلْجَحِيمُ لِلْغَاوِينَ ﴿٩١﴾}\\
\textamh{92.\  } & \mytextarabic{وَقِيلَ لَهُمْ أَيْنَ مَا كُنتُمْ تَعْبُدُونَ ﴿٩٢﴾}\\
\textamh{93.\  } & \mytextarabic{مِن دُونِ ٱللَّهِ هَلْ يَنصُرُونَكُمْ أَوْ يَنتَصِرُونَ ﴿٩٣﴾}\\
\textamh{94.\  } & \mytextarabic{فَكُبْكِبُوا۟ فِيهَا هُمْ وَٱلْغَاوُۥنَ ﴿٩٤﴾}\\
\textamh{95.\  } & \mytextarabic{وَجُنُودُ إِبْلِيسَ أَجْمَعُونَ ﴿٩٥﴾}\\
\textamh{96.\  } & \mytextarabic{قَالُوا۟ وَهُمْ فِيهَا يَخْتَصِمُونَ ﴿٩٦﴾}\\
\textamh{97.\  } & \mytextarabic{تَٱللَّهِ إِن كُنَّا لَفِى ضَلَـٰلٍۢ مُّبِينٍ ﴿٩٧﴾}\\
\textamh{98.\  } & \mytextarabic{إِذْ نُسَوِّيكُم بِرَبِّ ٱلْعَـٰلَمِينَ ﴿٩٨﴾}\\
\textamh{99.\  } & \mytextarabic{وَمَآ أَضَلَّنَآ إِلَّا ٱلْمُجْرِمُونَ ﴿٩٩﴾}\\
\textamh{100.\  } & \mytextarabic{فَمَا لَنَا مِن شَـٰفِعِينَ ﴿١٠٠﴾}\\
\textamh{101.\  } & \mytextarabic{وَلَا صَدِيقٍ حَمِيمٍۢ ﴿١٠١﴾}\\
\textamh{102.\  } & \mytextarabic{فَلَوْ أَنَّ لَنَا كَرَّةًۭ فَنَكُونَ مِنَ ٱلْمُؤْمِنِينَ ﴿١٠٢﴾}\\
\textamh{103.\  } & \mytextarabic{إِنَّ فِى ذَٟلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿١٠٣﴾}\\
\textamh{104.\  } & \mytextarabic{وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿١٠٤﴾}\\
\textamh{105.\  } & \mytextarabic{كَذَّبَتْ قَوْمُ نُوحٍ ٱلْمُرْسَلِينَ ﴿١٠٥﴾}\\
\textamh{106.\  } & \mytextarabic{إِذْ قَالَ لَهُمْ أَخُوهُمْ نُوحٌ أَلَا تَتَّقُونَ ﴿١٠٦﴾}\\
\textamh{107.\  } & \mytextarabic{إِنِّى لَكُمْ رَسُولٌ أَمِينٌۭ ﴿١٠٧﴾}\\
\textamh{108.\  } & \mytextarabic{فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿١٠٨﴾}\\
\textamh{109.\  } & \mytextarabic{وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍ ۖ إِنْ أَجْرِىَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٠٩﴾}\\
\textamh{110.\  } & \mytextarabic{فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿١١٠﴾}\\
\textamh{111.\  } & \mytextarabic{۞ قَالُوٓا۟ أَنُؤْمِنُ لَكَ وَٱتَّبَعَكَ ٱلْأَرْذَلُونَ ﴿١١١﴾}\\
\textamh{112.\  } & \mytextarabic{قَالَ وَمَا عِلْمِى بِمَا كَانُوا۟ يَعْمَلُونَ ﴿١١٢﴾}\\
\textamh{113.\  } & \mytextarabic{إِنْ حِسَابُهُمْ إِلَّا عَلَىٰ رَبِّى ۖ لَوْ تَشْعُرُونَ ﴿١١٣﴾}\\
\textamh{114.\  } & \mytextarabic{وَمَآ أَنَا۠ بِطَارِدِ ٱلْمُؤْمِنِينَ ﴿١١٤﴾}\\
\textamh{115.\  } & \mytextarabic{إِنْ أَنَا۠ إِلَّا نَذِيرٌۭ مُّبِينٌۭ ﴿١١٥﴾}\\
\textamh{116.\  } & \mytextarabic{قَالُوا۟ لَئِن لَّمْ تَنتَهِ يَـٰنُوحُ لَتَكُونَنَّ مِنَ ٱلْمَرْجُومِينَ ﴿١١٦﴾}\\
\textamh{117.\  } & \mytextarabic{قَالَ رَبِّ إِنَّ قَوْمِى كَذَّبُونِ ﴿١١٧﴾}\\
\textamh{118.\  } & \mytextarabic{فَٱفْتَحْ بَيْنِى وَبَيْنَهُمْ فَتْحًۭا وَنَجِّنِى وَمَن مَّعِىَ مِنَ ٱلْمُؤْمِنِينَ ﴿١١٨﴾}\\
\textamh{119.\  } & \mytextarabic{فَأَنجَيْنَـٰهُ وَمَن مَّعَهُۥ فِى ٱلْفُلْكِ ٱلْمَشْحُونِ ﴿١١٩﴾}\\
\textamh{120.\  } & \mytextarabic{ثُمَّ أَغْرَقْنَا بَعْدُ ٱلْبَاقِينَ ﴿١٢٠﴾}\\
\textamh{121.\  } & \mytextarabic{إِنَّ فِى ذَٟلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿١٢١﴾}\\
\textamh{122.\  } & \mytextarabic{وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿١٢٢﴾}\\
\textamh{123.\  } & \mytextarabic{كَذَّبَتْ عَادٌ ٱلْمُرْسَلِينَ ﴿١٢٣﴾}\\
\textamh{124.\  } & \mytextarabic{إِذْ قَالَ لَهُمْ أَخُوهُمْ هُودٌ أَلَا تَتَّقُونَ ﴿١٢٤﴾}\\
\textamh{125.\  } & \mytextarabic{إِنِّى لَكُمْ رَسُولٌ أَمِينٌۭ ﴿١٢٥﴾}\\
\textamh{126.\  } & \mytextarabic{فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿١٢٦﴾}\\
\textamh{127.\  } & \mytextarabic{وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍ ۖ إِنْ أَجْرِىَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٢٧﴾}\\
\textamh{128.\  } & \mytextarabic{أَتَبْنُونَ بِكُلِّ رِيعٍ ءَايَةًۭ تَعْبَثُونَ ﴿١٢٨﴾}\\
\textamh{129.\  } & \mytextarabic{وَتَتَّخِذُونَ مَصَانِعَ لَعَلَّكُمْ تَخْلُدُونَ ﴿١٢٩﴾}\\
\textamh{130.\  } & \mytextarabic{وَإِذَا بَطَشْتُم بَطَشْتُمْ جَبَّارِينَ ﴿١٣٠﴾}\\
\textamh{131.\  } & \mytextarabic{فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿١٣١﴾}\\
\textamh{132.\  } & \mytextarabic{وَٱتَّقُوا۟ ٱلَّذِىٓ أَمَدَّكُم بِمَا تَعْلَمُونَ ﴿١٣٢﴾}\\
\textamh{133.\  } & \mytextarabic{أَمَدَّكُم بِأَنْعَـٰمٍۢ وَبَنِينَ ﴿١٣٣﴾}\\
\textamh{134.\  } & \mytextarabic{وَجَنَّـٰتٍۢ وَعُيُونٍ ﴿١٣٤﴾}\\
\textamh{135.\  } & \mytextarabic{إِنِّىٓ أَخَافُ عَلَيْكُمْ عَذَابَ يَوْمٍ عَظِيمٍۢ ﴿١٣٥﴾}\\
\textamh{136.\  } & \mytextarabic{قَالُوا۟ سَوَآءٌ عَلَيْنَآ أَوَعَظْتَ أَمْ لَمْ تَكُن مِّنَ ٱلْوَٟعِظِينَ ﴿١٣٦﴾}\\
\textamh{137.\  } & \mytextarabic{إِنْ هَـٰذَآ إِلَّا خُلُقُ ٱلْأَوَّلِينَ ﴿١٣٧﴾}\\
\textamh{138.\  } & \mytextarabic{وَمَا نَحْنُ بِمُعَذَّبِينَ ﴿١٣٨﴾}\\
\textamh{139.\  } & \mytextarabic{فَكَذَّبُوهُ فَأَهْلَكْنَـٰهُمْ ۗ إِنَّ فِى ذَٟلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿١٣٩﴾}\\
\textamh{140.\  } & \mytextarabic{وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿١٤٠﴾}\\
\textamh{141.\  } & \mytextarabic{كَذَّبَتْ ثَمُودُ ٱلْمُرْسَلِينَ ﴿١٤١﴾}\\
\textamh{142.\  } & \mytextarabic{إِذْ قَالَ لَهُمْ أَخُوهُمْ صَـٰلِحٌ أَلَا تَتَّقُونَ ﴿١٤٢﴾}\\
\textamh{143.\  } & \mytextarabic{إِنِّى لَكُمْ رَسُولٌ أَمِينٌۭ ﴿١٤٣﴾}\\
\textamh{144.\  } & \mytextarabic{فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿١٤٤﴾}\\
\textamh{145.\  } & \mytextarabic{وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍ ۖ إِنْ أَجْرِىَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٤٥﴾}\\
\textamh{146.\  } & \mytextarabic{أَتُتْرَكُونَ فِى مَا هَـٰهُنَآ ءَامِنِينَ ﴿١٤٦﴾}\\
\textamh{147.\  } & \mytextarabic{فِى جَنَّـٰتٍۢ وَعُيُونٍۢ ﴿١٤٧﴾}\\
\textamh{148.\  } & \mytextarabic{وَزُرُوعٍۢ وَنَخْلٍۢ طَلْعُهَا هَضِيمٌۭ ﴿١٤٨﴾}\\
\textamh{149.\  } & \mytextarabic{وَتَنْحِتُونَ مِنَ ٱلْجِبَالِ بُيُوتًۭا فَـٰرِهِينَ ﴿١٤٩﴾}\\
\textamh{150.\  } & \mytextarabic{فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿١٥٠﴾}\\
\textamh{151.\  } & \mytextarabic{وَلَا تُطِيعُوٓا۟ أَمْرَ ٱلْمُسْرِفِينَ ﴿١٥١﴾}\\
\textamh{152.\  } & \mytextarabic{ٱلَّذِينَ يُفْسِدُونَ فِى ٱلْأَرْضِ وَلَا يُصْلِحُونَ ﴿١٥٢﴾}\\
\textamh{153.\  } & \mytextarabic{قَالُوٓا۟ إِنَّمَآ أَنتَ مِنَ ٱلْمُسَحَّرِينَ ﴿١٥٣﴾}\\
\textamh{154.\  } & \mytextarabic{مَآ أَنتَ إِلَّا بَشَرٌۭ مِّثْلُنَا فَأْتِ بِـَٔايَةٍ إِن كُنتَ مِنَ ٱلصَّـٰدِقِينَ ﴿١٥٤﴾}\\
\textamh{155.\  } & \mytextarabic{قَالَ هَـٰذِهِۦ نَاقَةٌۭ لَّهَا شِرْبٌۭ وَلَكُمْ شِرْبُ يَوْمٍۢ مَّعْلُومٍۢ ﴿١٥٥﴾}\\
\textamh{156.\  } & \mytextarabic{وَلَا تَمَسُّوهَا بِسُوٓءٍۢ فَيَأْخُذَكُمْ عَذَابُ يَوْمٍ عَظِيمٍۢ ﴿١٥٦﴾}\\
\textamh{157.\  } & \mytextarabic{فَعَقَرُوهَا فَأَصْبَحُوا۟ نَـٰدِمِينَ ﴿١٥٧﴾}\\
\textamh{158.\  } & \mytextarabic{فَأَخَذَهُمُ ٱلْعَذَابُ ۗ إِنَّ فِى ذَٟلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿١٥٨﴾}\\
\textamh{159.\  } & \mytextarabic{وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿١٥٩﴾}\\
\textamh{160.\  } & \mytextarabic{كَذَّبَتْ قَوْمُ لُوطٍ ٱلْمُرْسَلِينَ ﴿١٦٠﴾}\\
\textamh{161.\  } & \mytextarabic{إِذْ قَالَ لَهُمْ أَخُوهُمْ لُوطٌ أَلَا تَتَّقُونَ ﴿١٦١﴾}\\
\textamh{162.\  } & \mytextarabic{إِنِّى لَكُمْ رَسُولٌ أَمِينٌۭ ﴿١٦٢﴾}\\
\textamh{163.\  } & \mytextarabic{فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿١٦٣﴾}\\
\textamh{164.\  } & \mytextarabic{وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍ ۖ إِنْ أَجْرِىَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٦٤﴾}\\
\textamh{165.\  } & \mytextarabic{أَتَأْتُونَ ٱلذُّكْرَانَ مِنَ ٱلْعَـٰلَمِينَ ﴿١٦٥﴾}\\
\textamh{166.\  } & \mytextarabic{وَتَذَرُونَ مَا خَلَقَ لَكُمْ رَبُّكُم مِّنْ أَزْوَٟجِكُم ۚ بَلْ أَنتُمْ قَوْمٌ عَادُونَ ﴿١٦٦﴾}\\
\textamh{167.\  } & \mytextarabic{قَالُوا۟ لَئِن لَّمْ تَنتَهِ يَـٰلُوطُ لَتَكُونَنَّ مِنَ ٱلْمُخْرَجِينَ ﴿١٦٧﴾}\\
\textamh{168.\  } & \mytextarabic{قَالَ إِنِّى لِعَمَلِكُم مِّنَ ٱلْقَالِينَ ﴿١٦٨﴾}\\
\textamh{169.\  } & \mytextarabic{رَبِّ نَجِّنِى وَأَهْلِى مِمَّا يَعْمَلُونَ ﴿١٦٩﴾}\\
\textamh{170.\  } & \mytextarabic{فَنَجَّيْنَـٰهُ وَأَهْلَهُۥٓ أَجْمَعِينَ ﴿١٧٠﴾}\\
\textamh{171.\  } & \mytextarabic{إِلَّا عَجُوزًۭا فِى ٱلْغَٰبِرِينَ ﴿١٧١﴾}\\
\textamh{172.\  } & \mytextarabic{ثُمَّ دَمَّرْنَا ٱلْءَاخَرِينَ ﴿١٧٢﴾}\\
\textamh{173.\  } & \mytextarabic{وَأَمْطَرْنَا عَلَيْهِم مَّطَرًۭا ۖ فَسَآءَ مَطَرُ ٱلْمُنذَرِينَ ﴿١٧٣﴾}\\
\textamh{174.\  } & \mytextarabic{إِنَّ فِى ذَٟلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿١٧٤﴾}\\
\textamh{175.\  } & \mytextarabic{وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿١٧٥﴾}\\
\textamh{176.\  } & \mytextarabic{كَذَّبَ أَصْحَـٰبُ لْـَٔيْكَةِ ٱلْمُرْسَلِينَ ﴿١٧٦﴾}\\
\textamh{177.\  } & \mytextarabic{إِذْ قَالَ لَهُمْ شُعَيْبٌ أَلَا تَتَّقُونَ ﴿١٧٧﴾}\\
\textamh{178.\  } & \mytextarabic{إِنِّى لَكُمْ رَسُولٌ أَمِينٌۭ ﴿١٧٨﴾}\\
\textamh{179.\  } & \mytextarabic{فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿١٧٩﴾}\\
\textamh{180.\  } & \mytextarabic{وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍ ۖ إِنْ أَجْرِىَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٨٠﴾}\\
\textamh{181.\  } & \mytextarabic{۞ أَوْفُوا۟ ٱلْكَيْلَ وَلَا تَكُونُوا۟ مِنَ ٱلْمُخْسِرِينَ ﴿١٨١﴾}\\
\textamh{182.\  } & \mytextarabic{وَزِنُوا۟ بِٱلْقِسْطَاسِ ٱلْمُسْتَقِيمِ ﴿١٨٢﴾}\\
\textamh{183.\  } & \mytextarabic{وَلَا تَبْخَسُوا۟ ٱلنَّاسَ أَشْيَآءَهُمْ وَلَا تَعْثَوْا۟ فِى ٱلْأَرْضِ مُفْسِدِينَ ﴿١٨٣﴾}\\
\textamh{184.\  } & \mytextarabic{وَٱتَّقُوا۟ ٱلَّذِى خَلَقَكُمْ وَٱلْجِبِلَّةَ ٱلْأَوَّلِينَ ﴿١٨٤﴾}\\
\textamh{185.\  } & \mytextarabic{قَالُوٓا۟ إِنَّمَآ أَنتَ مِنَ ٱلْمُسَحَّرِينَ ﴿١٨٥﴾}\\
\textamh{186.\  } & \mytextarabic{وَمَآ أَنتَ إِلَّا بَشَرٌۭ مِّثْلُنَا وَإِن نَّظُنُّكَ لَمِنَ ٱلْكَـٰذِبِينَ ﴿١٨٦﴾}\\
\textamh{187.\  } & \mytextarabic{فَأَسْقِطْ عَلَيْنَا كِسَفًۭا مِّنَ ٱلسَّمَآءِ إِن كُنتَ مِنَ ٱلصَّـٰدِقِينَ ﴿١٨٧﴾}\\
\textamh{188.\  } & \mytextarabic{قَالَ رَبِّىٓ أَعْلَمُ بِمَا تَعْمَلُونَ ﴿١٨٨﴾}\\
\textamh{189.\  } & \mytextarabic{فَكَذَّبُوهُ فَأَخَذَهُمْ عَذَابُ يَوْمِ ٱلظُّلَّةِ ۚ إِنَّهُۥ كَانَ عَذَابَ يَوْمٍ عَظِيمٍ ﴿١٨٩﴾}\\
\textamh{190.\  } & \mytextarabic{إِنَّ فِى ذَٟلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿١٩٠﴾}\\
\textamh{191.\  } & \mytextarabic{وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿١٩١﴾}\\
\textamh{192.\  } & \mytextarabic{وَإِنَّهُۥ لَتَنزِيلُ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٩٢﴾}\\
\textamh{193.\  } & \mytextarabic{نَزَلَ بِهِ ٱلرُّوحُ ٱلْأَمِينُ ﴿١٩٣﴾}\\
\textamh{194.\  } & \mytextarabic{عَلَىٰ قَلْبِكَ لِتَكُونَ مِنَ ٱلْمُنذِرِينَ ﴿١٩٤﴾}\\
\textamh{195.\  } & \mytextarabic{بِلِسَانٍ عَرَبِىٍّۢ مُّبِينٍۢ ﴿١٩٥﴾}\\
\textamh{196.\  } & \mytextarabic{وَإِنَّهُۥ لَفِى زُبُرِ ٱلْأَوَّلِينَ ﴿١٩٦﴾}\\
\textamh{197.\  } & \mytextarabic{أَوَلَمْ يَكُن لَّهُمْ ءَايَةً أَن يَعْلَمَهُۥ عُلَمَـٰٓؤُا۟ بَنِىٓ إِسْرَٰٓءِيلَ ﴿١٩٧﴾}\\
\textamh{198.\  } & \mytextarabic{وَلَوْ نَزَّلْنَـٰهُ عَلَىٰ بَعْضِ ٱلْأَعْجَمِينَ ﴿١٩٨﴾}\\
\textamh{199.\  } & \mytextarabic{فَقَرَأَهُۥ عَلَيْهِم مَّا كَانُوا۟ بِهِۦ مُؤْمِنِينَ ﴿١٩٩﴾}\\
\textamh{200.\  } & \mytextarabic{كَذَٟلِكَ سَلَكْنَـٰهُ فِى قُلُوبِ ٱلْمُجْرِمِينَ ﴿٢٠٠﴾}\\
\textamh{201.\  } & \mytextarabic{لَا يُؤْمِنُونَ بِهِۦ حَتَّىٰ يَرَوُا۟ ٱلْعَذَابَ ٱلْأَلِيمَ ﴿٢٠١﴾}\\
\textamh{202.\  } & \mytextarabic{فَيَأْتِيَهُم بَغْتَةًۭ وَهُمْ لَا يَشْعُرُونَ ﴿٢٠٢﴾}\\
\textamh{203.\  } & \mytextarabic{فَيَقُولُوا۟ هَلْ نَحْنُ مُنظَرُونَ ﴿٢٠٣﴾}\\
\textamh{204.\  } & \mytextarabic{أَفَبِعَذَابِنَا يَسْتَعْجِلُونَ ﴿٢٠٤﴾}\\
\textamh{205.\  } & \mytextarabic{أَفَرَءَيْتَ إِن مَّتَّعْنَـٰهُمْ سِنِينَ ﴿٢٠٥﴾}\\
\textamh{206.\  } & \mytextarabic{ثُمَّ جَآءَهُم مَّا كَانُوا۟ يُوعَدُونَ ﴿٢٠٦﴾}\\
\textamh{207.\  } & \mytextarabic{مَآ أَغْنَىٰ عَنْهُم مَّا كَانُوا۟ يُمَتَّعُونَ ﴿٢٠٧﴾}\\
\textamh{208.\  } & \mytextarabic{وَمَآ أَهْلَكْنَا مِن قَرْيَةٍ إِلَّا لَهَا مُنذِرُونَ ﴿٢٠٨﴾}\\
\textamh{209.\  } & \mytextarabic{ذِكْرَىٰ وَمَا كُنَّا ظَـٰلِمِينَ ﴿٢٠٩﴾}\\
\textamh{210.\  } & \mytextarabic{وَمَا تَنَزَّلَتْ بِهِ ٱلشَّيَـٰطِينُ ﴿٢١٠﴾}\\
\textamh{211.\  } & \mytextarabic{وَمَا يَنۢبَغِى لَهُمْ وَمَا يَسْتَطِيعُونَ ﴿٢١١﴾}\\
\textamh{212.\  } & \mytextarabic{إِنَّهُمْ عَنِ ٱلسَّمْعِ لَمَعْزُولُونَ ﴿٢١٢﴾}\\
\textamh{213.\  } & \mytextarabic{فَلَا تَدْعُ مَعَ ٱللَّهِ إِلَـٰهًا ءَاخَرَ فَتَكُونَ مِنَ ٱلْمُعَذَّبِينَ ﴿٢١٣﴾}\\
\textamh{214.\  } & \mytextarabic{وَأَنذِرْ عَشِيرَتَكَ ٱلْأَقْرَبِينَ ﴿٢١٤﴾}\\
\textamh{215.\  } & \mytextarabic{وَٱخْفِضْ جَنَاحَكَ لِمَنِ ٱتَّبَعَكَ مِنَ ٱلْمُؤْمِنِينَ ﴿٢١٥﴾}\\
\textamh{216.\  } & \mytextarabic{فَإِنْ عَصَوْكَ فَقُلْ إِنِّى بَرِىٓءٌۭ مِّمَّا تَعْمَلُونَ ﴿٢١٦﴾}\\
\textamh{217.\  } & \mytextarabic{وَتَوَكَّلْ عَلَى ٱلْعَزِيزِ ٱلرَّحِيمِ ﴿٢١٧﴾}\\
\textamh{218.\  } & \mytextarabic{ٱلَّذِى يَرَىٰكَ حِينَ تَقُومُ ﴿٢١٨﴾}\\
\textamh{219.\  } & \mytextarabic{وَتَقَلُّبَكَ فِى ٱلسَّٰجِدِينَ ﴿٢١٩﴾}\\
\textamh{220.\  } & \mytextarabic{إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿٢٢٠﴾}\\
\textamh{221.\  } & \mytextarabic{هَلْ أُنَبِّئُكُمْ عَلَىٰ مَن تَنَزَّلُ ٱلشَّيَـٰطِينُ ﴿٢٢١﴾}\\
\textamh{222.\  } & \mytextarabic{تَنَزَّلُ عَلَىٰ كُلِّ أَفَّاكٍ أَثِيمٍۢ ﴿٢٢٢﴾}\\
\textamh{223.\  } & \mytextarabic{يُلْقُونَ ٱلسَّمْعَ وَأَكْثَرُهُمْ كَـٰذِبُونَ ﴿٢٢٣﴾}\\
\textamh{224.\  } & \mytextarabic{وَٱلشُّعَرَآءُ يَتَّبِعُهُمُ ٱلْغَاوُۥنَ ﴿٢٢٤﴾}\\
\textamh{225.\  } & \mytextarabic{أَلَمْ تَرَ أَنَّهُمْ فِى كُلِّ وَادٍۢ يَهِيمُونَ ﴿٢٢٥﴾}\\
\textamh{226.\  } & \mytextarabic{وَأَنَّهُمْ يَقُولُونَ مَا لَا يَفْعَلُونَ ﴿٢٢٦﴾}\\
\textamh{227.\  } & \mytextarabic{إِلَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ وَذَكَرُوا۟ ٱللَّهَ كَثِيرًۭا وَٱنتَصَرُوا۟ مِنۢ بَعْدِ مَا ظُلِمُوا۟ ۗ وَسَيَعْلَمُ ٱلَّذِينَ ظَلَمُوٓا۟ أَىَّ مُنقَلَبٍۢ يَنقَلِبُونَ ﴿٢٢٧﴾}\\
\end{longtable}
