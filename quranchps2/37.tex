\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ وَٱلصَّـٰٓفَّٰتِ صَفًّۭا ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{فَٱلزَّٰجِرَٰتِ زَجْرًۭا ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{فَٱلتَّٰلِيَـٰتِ ذِكْرًا ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{إِنَّ إِلَـٰهَكُمْ لَوَٟحِدٌۭ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{رَّبُّ ٱلسَّمَـٰوَٟتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا وَرَبُّ ٱلْمَشَـٰرِقِ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{إِنَّا زَيَّنَّا ٱلسَّمَآءَ ٱلدُّنْيَا بِزِينَةٍ ٱلْكَوَاكِبِ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{وَحِفْظًۭا مِّن كُلِّ شَيْطَٰنٍۢ مَّارِدٍۢ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{لَّا يَسَّمَّعُونَ إِلَى ٱلْمَلَإِ ٱلْأَعْلَىٰ وَيُقْذَفُونَ مِن كُلِّ جَانِبٍۢ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{دُحُورًۭا ۖ وَلَهُمْ عَذَابٌۭ وَاصِبٌ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{إِلَّا مَنْ خَطِفَ ٱلْخَطْفَةَ فَأَتْبَعَهُۥ شِهَابٌۭ ثَاقِبٌۭ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{فَٱسْتَفْتِهِمْ أَهُمْ أَشَدُّ خَلْقًا أَم مَّنْ خَلَقْنَآ ۚ إِنَّا خَلَقْنَـٰهُم مِّن طِينٍۢ لَّازِبٍۭ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{بَلْ عَجِبْتَ وَيَسْخَرُونَ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{وَإِذَا ذُكِّرُوا۟ لَا يَذْكُرُونَ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{وَإِذَا رَأَوْا۟ ءَايَةًۭ يَسْتَسْخِرُونَ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{وَقَالُوٓا۟ إِنْ هَـٰذَآ إِلَّا سِحْرٌۭ مُّبِينٌ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{أَءِذَا مِتْنَا وَكُنَّا تُرَابًۭا وَعِظَـٰمًا أَءِنَّا لَمَبْعُوثُونَ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{أَوَءَابَآؤُنَا ٱلْأَوَّلُونَ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{قُلْ نَعَمْ وَأَنتُمْ دَٟخِرُونَ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{فَإِنَّمَا هِىَ زَجْرَةٌۭ وَٟحِدَةٌۭ فَإِذَا هُمْ يَنظُرُونَ ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{وَقَالُوا۟ يَـٰوَيْلَنَا هَـٰذَا يَوْمُ ٱلدِّينِ ﴿٢٠﴾}\\
\textamh{21.\  } & \mytextarabic{هَـٰذَا يَوْمُ ٱلْفَصْلِ ٱلَّذِى كُنتُم بِهِۦ تُكَذِّبُونَ ﴿٢١﴾}\\
\textamh{22.\  } & \mytextarabic{۞ ٱحْشُرُوا۟ ٱلَّذِينَ ظَلَمُوا۟ وَأَزْوَٟجَهُمْ وَمَا كَانُوا۟ يَعْبُدُونَ ﴿٢٢﴾}\\
\textamh{23.\  } & \mytextarabic{مِن دُونِ ٱللَّهِ فَٱهْدُوهُمْ إِلَىٰ صِرَٰطِ ٱلْجَحِيمِ ﴿٢٣﴾}\\
\textamh{24.\  } & \mytextarabic{وَقِفُوهُمْ ۖ إِنَّهُم مَّسْـُٔولُونَ ﴿٢٤﴾}\\
\textamh{25.\  } & \mytextarabic{مَا لَكُمْ لَا تَنَاصَرُونَ ﴿٢٥﴾}\\
\textamh{26.\  } & \mytextarabic{بَلْ هُمُ ٱلْيَوْمَ مُسْتَسْلِمُونَ ﴿٢٦﴾}\\
\textamh{27.\  } & \mytextarabic{وَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٍۢ يَتَسَآءَلُونَ ﴿٢٧﴾}\\
\textamh{28.\  } & \mytextarabic{قَالُوٓا۟ إِنَّكُمْ كُنتُمْ تَأْتُونَنَا عَنِ ٱلْيَمِينِ ﴿٢٨﴾}\\
\textamh{29.\  } & \mytextarabic{قَالُوا۟ بَل لَّمْ تَكُونُوا۟ مُؤْمِنِينَ ﴿٢٩﴾}\\
\textamh{30.\  } & \mytextarabic{وَمَا كَانَ لَنَا عَلَيْكُم مِّن سُلْطَٰنٍۭ ۖ بَلْ كُنتُمْ قَوْمًۭا طَٰغِينَ ﴿٣٠﴾}\\
\textamh{31.\  } & \mytextarabic{فَحَقَّ عَلَيْنَا قَوْلُ رَبِّنَآ ۖ إِنَّا لَذَآئِقُونَ ﴿٣١﴾}\\
\textamh{32.\  } & \mytextarabic{فَأَغْوَيْنَـٰكُمْ إِنَّا كُنَّا غَٰوِينَ ﴿٣٢﴾}\\
\textamh{33.\  } & \mytextarabic{فَإِنَّهُمْ يَوْمَئِذٍۢ فِى ٱلْعَذَابِ مُشْتَرِكُونَ ﴿٣٣﴾}\\
\textamh{34.\  } & \mytextarabic{إِنَّا كَذَٟلِكَ نَفْعَلُ بِٱلْمُجْرِمِينَ ﴿٣٤﴾}\\
\textamh{35.\  } & \mytextarabic{إِنَّهُمْ كَانُوٓا۟ إِذَا قِيلَ لَهُمْ لَآ إِلَـٰهَ إِلَّا ٱللَّهُ يَسْتَكْبِرُونَ ﴿٣٥﴾}\\
\textamh{36.\  } & \mytextarabic{وَيَقُولُونَ أَئِنَّا لَتَارِكُوٓا۟ ءَالِهَتِنَا لِشَاعِرٍۢ مَّجْنُونٍۭ ﴿٣٦﴾}\\
\textamh{37.\  } & \mytextarabic{بَلْ جَآءَ بِٱلْحَقِّ وَصَدَّقَ ٱلْمُرْسَلِينَ ﴿٣٧﴾}\\
\textamh{38.\  } & \mytextarabic{إِنَّكُمْ لَذَآئِقُوا۟ ٱلْعَذَابِ ٱلْأَلِيمِ ﴿٣٨﴾}\\
\textamh{39.\  } & \mytextarabic{وَمَا تُجْزَوْنَ إِلَّا مَا كُنتُمْ تَعْمَلُونَ ﴿٣٩﴾}\\
\textamh{40.\  } & \mytextarabic{إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿٤٠﴾}\\
\textamh{41.\  } & \mytextarabic{أُو۟لَـٰٓئِكَ لَهُمْ رِزْقٌۭ مَّعْلُومٌۭ ﴿٤١﴾}\\
\textamh{42.\  } & \mytextarabic{فَوَٟكِهُ ۖ وَهُم مُّكْرَمُونَ ﴿٤٢﴾}\\
\textamh{43.\  } & \mytextarabic{فِى جَنَّـٰتِ ٱلنَّعِيمِ ﴿٤٣﴾}\\
\textamh{44.\  } & \mytextarabic{عَلَىٰ سُرُرٍۢ مُّتَقَـٰبِلِينَ ﴿٤٤﴾}\\
\textamh{45.\  } & \mytextarabic{يُطَافُ عَلَيْهِم بِكَأْسٍۢ مِّن مَّعِينٍۭ ﴿٤٥﴾}\\
\textamh{46.\  } & \mytextarabic{بَيْضَآءَ لَذَّةٍۢ لِّلشَّـٰرِبِينَ ﴿٤٦﴾}\\
\textamh{47.\  } & \mytextarabic{لَا فِيهَا غَوْلٌۭ وَلَا هُمْ عَنْهَا يُنزَفُونَ ﴿٤٧﴾}\\
\textamh{48.\  } & \mytextarabic{وَعِندَهُمْ قَـٰصِرَٰتُ ٱلطَّرْفِ عِينٌۭ ﴿٤٨﴾}\\
\textamh{49.\  } & \mytextarabic{كَأَنَّهُنَّ بَيْضٌۭ مَّكْنُونٌۭ ﴿٤٩﴾}\\
\textamh{50.\  } & \mytextarabic{فَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٍۢ يَتَسَآءَلُونَ ﴿٥٠﴾}\\
\textamh{51.\  } & \mytextarabic{قَالَ قَآئِلٌۭ مِّنْهُمْ إِنِّى كَانَ لِى قَرِينٌۭ ﴿٥١﴾}\\
\textamh{52.\  } & \mytextarabic{يَقُولُ أَءِنَّكَ لَمِنَ ٱلْمُصَدِّقِينَ ﴿٥٢﴾}\\
\textamh{53.\  } & \mytextarabic{أَءِذَا مِتْنَا وَكُنَّا تُرَابًۭا وَعِظَـٰمًا أَءِنَّا لَمَدِينُونَ ﴿٥٣﴾}\\
\textamh{54.\  } & \mytextarabic{قَالَ هَلْ أَنتُم مُّطَّلِعُونَ ﴿٥٤﴾}\\
\textamh{55.\  } & \mytextarabic{فَٱطَّلَعَ فَرَءَاهُ فِى سَوَآءِ ٱلْجَحِيمِ ﴿٥٥﴾}\\
\textamh{56.\  } & \mytextarabic{قَالَ تَٱللَّهِ إِن كِدتَّ لَتُرْدِينِ ﴿٥٦﴾}\\
\textamh{57.\  } & \mytextarabic{وَلَوْلَا نِعْمَةُ رَبِّى لَكُنتُ مِنَ ٱلْمُحْضَرِينَ ﴿٥٧﴾}\\
\textamh{58.\  } & \mytextarabic{أَفَمَا نَحْنُ بِمَيِّتِينَ ﴿٥٨﴾}\\
\textamh{59.\  } & \mytextarabic{إِلَّا مَوْتَتَنَا ٱلْأُولَىٰ وَمَا نَحْنُ بِمُعَذَّبِينَ ﴿٥٩﴾}\\
\textamh{60.\  } & \mytextarabic{إِنَّ هَـٰذَا لَهُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿٦٠﴾}\\
\textamh{61.\  } & \mytextarabic{لِمِثْلِ هَـٰذَا فَلْيَعْمَلِ ٱلْعَـٰمِلُونَ ﴿٦١﴾}\\
\textamh{62.\  } & \mytextarabic{أَذَٟلِكَ خَيْرٌۭ نُّزُلًا أَمْ شَجَرَةُ ٱلزَّقُّومِ ﴿٦٢﴾}\\
\textamh{63.\  } & \mytextarabic{إِنَّا جَعَلْنَـٰهَا فِتْنَةًۭ لِّلظَّـٰلِمِينَ ﴿٦٣﴾}\\
\textamh{64.\  } & \mytextarabic{إِنَّهَا شَجَرَةٌۭ تَخْرُجُ فِىٓ أَصْلِ ٱلْجَحِيمِ ﴿٦٤﴾}\\
\textamh{65.\  } & \mytextarabic{طَلْعُهَا كَأَنَّهُۥ رُءُوسُ ٱلشَّيَـٰطِينِ ﴿٦٥﴾}\\
\textamh{66.\  } & \mytextarabic{فَإِنَّهُمْ لَءَاكِلُونَ مِنْهَا فَمَالِـُٔونَ مِنْهَا ٱلْبُطُونَ ﴿٦٦﴾}\\
\textamh{67.\  } & \mytextarabic{ثُمَّ إِنَّ لَهُمْ عَلَيْهَا لَشَوْبًۭا مِّنْ حَمِيمٍۢ ﴿٦٧﴾}\\
\textamh{68.\  } & \mytextarabic{ثُمَّ إِنَّ مَرْجِعَهُمْ لَإِلَى ٱلْجَحِيمِ ﴿٦٨﴾}\\
\textamh{69.\  } & \mytextarabic{إِنَّهُمْ أَلْفَوْا۟ ءَابَآءَهُمْ ضَآلِّينَ ﴿٦٩﴾}\\
\textamh{70.\  } & \mytextarabic{فَهُمْ عَلَىٰٓ ءَاثَـٰرِهِمْ يُهْرَعُونَ ﴿٧٠﴾}\\
\textamh{71.\  } & \mytextarabic{وَلَقَدْ ضَلَّ قَبْلَهُمْ أَكْثَرُ ٱلْأَوَّلِينَ ﴿٧١﴾}\\
\textamh{72.\  } & \mytextarabic{وَلَقَدْ أَرْسَلْنَا فِيهِم مُّنذِرِينَ ﴿٧٢﴾}\\
\textamh{73.\  } & \mytextarabic{فَٱنظُرْ كَيْفَ كَانَ عَـٰقِبَةُ ٱلْمُنذَرِينَ ﴿٧٣﴾}\\
\textamh{74.\  } & \mytextarabic{إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿٧٤﴾}\\
\textamh{75.\  } & \mytextarabic{وَلَقَدْ نَادَىٰنَا نُوحٌۭ فَلَنِعْمَ ٱلْمُجِيبُونَ ﴿٧٥﴾}\\
\textamh{76.\  } & \mytextarabic{وَنَجَّيْنَـٰهُ وَأَهْلَهُۥ مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿٧٦﴾}\\
\textamh{77.\  } & \mytextarabic{وَجَعَلْنَا ذُرِّيَّتَهُۥ هُمُ ٱلْبَاقِينَ ﴿٧٧﴾}\\
\textamh{78.\  } & \mytextarabic{وَتَرَكْنَا عَلَيْهِ فِى ٱلْءَاخِرِينَ ﴿٧٨﴾}\\
\textamh{79.\  } & \mytextarabic{سَلَـٰمٌ عَلَىٰ نُوحٍۢ فِى ٱلْعَـٰلَمِينَ ﴿٧٩﴾}\\
\textamh{80.\  } & \mytextarabic{إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿٨٠﴾}\\
\textamh{81.\  } & \mytextarabic{إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿٨١﴾}\\
\textamh{82.\  } & \mytextarabic{ثُمَّ أَغْرَقْنَا ٱلْءَاخَرِينَ ﴿٨٢﴾}\\
\textamh{83.\  } & \mytextarabic{۞ وَإِنَّ مِن شِيعَتِهِۦ لَإِبْرَٰهِيمَ ﴿٨٣﴾}\\
\textamh{84.\  } & \mytextarabic{إِذْ جَآءَ رَبَّهُۥ بِقَلْبٍۢ سَلِيمٍ ﴿٨٤﴾}\\
\textamh{85.\  } & \mytextarabic{إِذْ قَالَ لِأَبِيهِ وَقَوْمِهِۦ مَاذَا تَعْبُدُونَ ﴿٨٥﴾}\\
\textamh{86.\  } & \mytextarabic{أَئِفْكًا ءَالِهَةًۭ دُونَ ٱللَّهِ تُرِيدُونَ ﴿٨٦﴾}\\
\textamh{87.\  } & \mytextarabic{فَمَا ظَنُّكُم بِرَبِّ ٱلْعَـٰلَمِينَ ﴿٨٧﴾}\\
\textamh{88.\  } & \mytextarabic{فَنَظَرَ نَظْرَةًۭ فِى ٱلنُّجُومِ ﴿٨٨﴾}\\
\textamh{89.\  } & \mytextarabic{فَقَالَ إِنِّى سَقِيمٌۭ ﴿٨٩﴾}\\
\textamh{90.\  } & \mytextarabic{فَتَوَلَّوْا۟ عَنْهُ مُدْبِرِينَ ﴿٩٠﴾}\\
\textamh{91.\  } & \mytextarabic{فَرَاغَ إِلَىٰٓ ءَالِهَتِهِمْ فَقَالَ أَلَا تَأْكُلُونَ ﴿٩١﴾}\\
\textamh{92.\  } & \mytextarabic{مَا لَكُمْ لَا تَنطِقُونَ ﴿٩٢﴾}\\
\textamh{93.\  } & \mytextarabic{فَرَاغَ عَلَيْهِمْ ضَرْبًۢا بِٱلْيَمِينِ ﴿٩٣﴾}\\
\textamh{94.\  } & \mytextarabic{فَأَقْبَلُوٓا۟ إِلَيْهِ يَزِفُّونَ ﴿٩٤﴾}\\
\textamh{95.\  } & \mytextarabic{قَالَ أَتَعْبُدُونَ مَا تَنْحِتُونَ ﴿٩٥﴾}\\
\textamh{96.\  } & \mytextarabic{وَٱللَّهُ خَلَقَكُمْ وَمَا تَعْمَلُونَ ﴿٩٦﴾}\\
\textamh{97.\  } & \mytextarabic{قَالُوا۟ ٱبْنُوا۟ لَهُۥ بُنْيَـٰنًۭا فَأَلْقُوهُ فِى ٱلْجَحِيمِ ﴿٩٧﴾}\\
\textamh{98.\  } & \mytextarabic{فَأَرَادُوا۟ بِهِۦ كَيْدًۭا فَجَعَلْنَـٰهُمُ ٱلْأَسْفَلِينَ ﴿٩٨﴾}\\
\textamh{99.\  } & \mytextarabic{وَقَالَ إِنِّى ذَاهِبٌ إِلَىٰ رَبِّى سَيَهْدِينِ ﴿٩٩﴾}\\
\textamh{100.\  } & \mytextarabic{رَبِّ هَبْ لِى مِنَ ٱلصَّـٰلِحِينَ ﴿١٠٠﴾}\\
\textamh{101.\  } & \mytextarabic{فَبَشَّرْنَـٰهُ بِغُلَـٰمٍ حَلِيمٍۢ ﴿١٠١﴾}\\
\textamh{102.\  } & \mytextarabic{فَلَمَّا بَلَغَ مَعَهُ ٱلسَّعْىَ قَالَ يَـٰبُنَىَّ إِنِّىٓ أَرَىٰ فِى ٱلْمَنَامِ أَنِّىٓ أَذْبَحُكَ فَٱنظُرْ مَاذَا تَرَىٰ ۚ قَالَ يَـٰٓأَبَتِ ٱفْعَلْ مَا تُؤْمَرُ ۖ سَتَجِدُنِىٓ إِن شَآءَ ٱللَّهُ مِنَ ٱلصَّـٰبِرِينَ ﴿١٠٢﴾}\\
\textamh{103.\  } & \mytextarabic{فَلَمَّآ أَسْلَمَا وَتَلَّهُۥ لِلْجَبِينِ ﴿١٠٣﴾}\\
\textamh{104.\  } & \mytextarabic{وَنَـٰدَيْنَـٰهُ أَن يَـٰٓإِبْرَٰهِيمُ ﴿١٠٤﴾}\\
\textamh{105.\  } & \mytextarabic{قَدْ صَدَّقْتَ ٱلرُّءْيَآ ۚ إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿١٠٥﴾}\\
\textamh{106.\  } & \mytextarabic{إِنَّ هَـٰذَا لَهُوَ ٱلْبَلَـٰٓؤُا۟ ٱلْمُبِينُ ﴿١٠٦﴾}\\
\textamh{107.\  } & \mytextarabic{وَفَدَيْنَـٰهُ بِذِبْحٍ عَظِيمٍۢ ﴿١٠٧﴾}\\
\textamh{108.\  } & \mytextarabic{وَتَرَكْنَا عَلَيْهِ فِى ٱلْءَاخِرِينَ ﴿١٠٨﴾}\\
\textamh{109.\  } & \mytextarabic{سَلَـٰمٌ عَلَىٰٓ إِبْرَٰهِيمَ ﴿١٠٩﴾}\\
\textamh{110.\  } & \mytextarabic{كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿١١٠﴾}\\
\textamh{111.\  } & \mytextarabic{إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿١١١﴾}\\
\textamh{112.\  } & \mytextarabic{وَبَشَّرْنَـٰهُ بِإِسْحَـٰقَ نَبِيًّۭا مِّنَ ٱلصَّـٰلِحِينَ ﴿١١٢﴾}\\
\textamh{113.\  } & \mytextarabic{وَبَٰرَكْنَا عَلَيْهِ وَعَلَىٰٓ إِسْحَـٰقَ ۚ وَمِن ذُرِّيَّتِهِمَا مُحْسِنٌۭ وَظَالِمٌۭ لِّنَفْسِهِۦ مُبِينٌۭ ﴿١١٣﴾}\\
\textamh{114.\  } & \mytextarabic{وَلَقَدْ مَنَنَّا عَلَىٰ مُوسَىٰ وَهَـٰرُونَ ﴿١١٤﴾}\\
\textamh{115.\  } & \mytextarabic{وَنَجَّيْنَـٰهُمَا وَقَوْمَهُمَا مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿١١٥﴾}\\
\textamh{116.\  } & \mytextarabic{وَنَصَرْنَـٰهُمْ فَكَانُوا۟ هُمُ ٱلْغَٰلِبِينَ ﴿١١٦﴾}\\
\textamh{117.\  } & \mytextarabic{وَءَاتَيْنَـٰهُمَا ٱلْكِتَـٰبَ ٱلْمُسْتَبِينَ ﴿١١٧﴾}\\
\textamh{118.\  } & \mytextarabic{وَهَدَيْنَـٰهُمَا ٱلصِّرَٰطَ ٱلْمُسْتَقِيمَ ﴿١١٨﴾}\\
\textamh{119.\  } & \mytextarabic{وَتَرَكْنَا عَلَيْهِمَا فِى ٱلْءَاخِرِينَ ﴿١١٩﴾}\\
\textamh{120.\  } & \mytextarabic{سَلَـٰمٌ عَلَىٰ مُوسَىٰ وَهَـٰرُونَ ﴿١٢٠﴾}\\
\textamh{121.\  } & \mytextarabic{إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿١٢١﴾}\\
\textamh{122.\  } & \mytextarabic{إِنَّهُمَا مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿١٢٢﴾}\\
\textamh{123.\  } & \mytextarabic{وَإِنَّ إِلْيَاسَ لَمِنَ ٱلْمُرْسَلِينَ ﴿١٢٣﴾}\\
\textamh{124.\  } & \mytextarabic{إِذْ قَالَ لِقَوْمِهِۦٓ أَلَا تَتَّقُونَ ﴿١٢٤﴾}\\
\textamh{125.\  } & \mytextarabic{أَتَدْعُونَ بَعْلًۭا وَتَذَرُونَ أَحْسَنَ ٱلْخَـٰلِقِينَ ﴿١٢٥﴾}\\
\textamh{126.\  } & \mytextarabic{ٱللَّهَ رَبَّكُمْ وَرَبَّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿١٢٦﴾}\\
\textamh{127.\  } & \mytextarabic{فَكَذَّبُوهُ فَإِنَّهُمْ لَمُحْضَرُونَ ﴿١٢٧﴾}\\
\textamh{128.\  } & \mytextarabic{إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿١٢٨﴾}\\
\textamh{129.\  } & \mytextarabic{وَتَرَكْنَا عَلَيْهِ فِى ٱلْءَاخِرِينَ ﴿١٢٩﴾}\\
\textamh{130.\  } & \mytextarabic{سَلَـٰمٌ عَلَىٰٓ إِلْ يَاسِينَ ﴿١٣٠﴾}\\
\textamh{131.\  } & \mytextarabic{إِنَّا كَذَٟلِكَ نَجْزِى ٱلْمُحْسِنِينَ ﴿١٣١﴾}\\
\textamh{132.\  } & \mytextarabic{إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿١٣٢﴾}\\
\textamh{133.\  } & \mytextarabic{وَإِنَّ لُوطًۭا لَّمِنَ ٱلْمُرْسَلِينَ ﴿١٣٣﴾}\\
\textamh{134.\  } & \mytextarabic{إِذْ نَجَّيْنَـٰهُ وَأَهْلَهُۥٓ أَجْمَعِينَ ﴿١٣٤﴾}\\
\textamh{135.\  } & \mytextarabic{إِلَّا عَجُوزًۭا فِى ٱلْغَٰبِرِينَ ﴿١٣٥﴾}\\
\textamh{136.\  } & \mytextarabic{ثُمَّ دَمَّرْنَا ٱلْءَاخَرِينَ ﴿١٣٦﴾}\\
\textamh{137.\  } & \mytextarabic{وَإِنَّكُمْ لَتَمُرُّونَ عَلَيْهِم مُّصْبِحِينَ ﴿١٣٧﴾}\\
\textamh{138.\  } & \mytextarabic{وَبِٱلَّيْلِ ۗ أَفَلَا تَعْقِلُونَ ﴿١٣٨﴾}\\
\textamh{139.\  } & \mytextarabic{وَإِنَّ يُونُسَ لَمِنَ ٱلْمُرْسَلِينَ ﴿١٣٩﴾}\\
\textamh{140.\  } & \mytextarabic{إِذْ أَبَقَ إِلَى ٱلْفُلْكِ ٱلْمَشْحُونِ ﴿١٤٠﴾}\\
\textamh{141.\  } & \mytextarabic{فَسَاهَمَ فَكَانَ مِنَ ٱلْمُدْحَضِينَ ﴿١٤١﴾}\\
\textamh{142.\  } & \mytextarabic{فَٱلْتَقَمَهُ ٱلْحُوتُ وَهُوَ مُلِيمٌۭ ﴿١٤٢﴾}\\
\textamh{143.\  } & \mytextarabic{فَلَوْلَآ أَنَّهُۥ كَانَ مِنَ ٱلْمُسَبِّحِينَ ﴿١٤٣﴾}\\
\textamh{144.\  } & \mytextarabic{لَلَبِثَ فِى بَطْنِهِۦٓ إِلَىٰ يَوْمِ يُبْعَثُونَ ﴿١٤٤﴾}\\
\textamh{145.\  } & \mytextarabic{۞ فَنَبَذْنَـٰهُ بِٱلْعَرَآءِ وَهُوَ سَقِيمٌۭ ﴿١٤٥﴾}\\
\textamh{146.\  } & \mytextarabic{وَأَنۢبَتْنَا عَلَيْهِ شَجَرَةًۭ مِّن يَقْطِينٍۢ ﴿١٤٦﴾}\\
\textamh{147.\  } & \mytextarabic{وَأَرْسَلْنَـٰهُ إِلَىٰ مِا۟ئَةِ أَلْفٍ أَوْ يَزِيدُونَ ﴿١٤٧﴾}\\
\textamh{148.\  } & \mytextarabic{فَـَٔامَنُوا۟ فَمَتَّعْنَـٰهُمْ إِلَىٰ حِينٍۢ ﴿١٤٨﴾}\\
\textamh{149.\  } & \mytextarabic{فَٱسْتَفْتِهِمْ أَلِرَبِّكَ ٱلْبَنَاتُ وَلَهُمُ ٱلْبَنُونَ ﴿١٤٩﴾}\\
\textamh{150.\  } & \mytextarabic{أَمْ خَلَقْنَا ٱلْمَلَـٰٓئِكَةَ إِنَـٰثًۭا وَهُمْ شَـٰهِدُونَ ﴿١٥٠﴾}\\
\textamh{151.\  } & \mytextarabic{أَلَآ إِنَّهُم مِّنْ إِفْكِهِمْ لَيَقُولُونَ ﴿١٥١﴾}\\
\textamh{152.\  } & \mytextarabic{وَلَدَ ٱللَّهُ وَإِنَّهُمْ لَكَـٰذِبُونَ ﴿١٥٢﴾}\\
\textamh{153.\  } & \mytextarabic{أَصْطَفَى ٱلْبَنَاتِ عَلَى ٱلْبَنِينَ ﴿١٥٣﴾}\\
\textamh{154.\  } & \mytextarabic{مَا لَكُمْ كَيْفَ تَحْكُمُونَ ﴿١٥٤﴾}\\
\textamh{155.\  } & \mytextarabic{أَفَلَا تَذَكَّرُونَ ﴿١٥٥﴾}\\
\textamh{156.\  } & \mytextarabic{أَمْ لَكُمْ سُلْطَٰنٌۭ مُّبِينٌۭ ﴿١٥٦﴾}\\
\textamh{157.\  } & \mytextarabic{فَأْتُوا۟ بِكِتَـٰبِكُمْ إِن كُنتُمْ صَـٰدِقِينَ ﴿١٥٧﴾}\\
\textamh{158.\  } & \mytextarabic{وَجَعَلُوا۟ بَيْنَهُۥ وَبَيْنَ ٱلْجِنَّةِ نَسَبًۭا ۚ وَلَقَدْ عَلِمَتِ ٱلْجِنَّةُ إِنَّهُمْ لَمُحْضَرُونَ ﴿١٥٨﴾}\\
\textamh{159.\  } & \mytextarabic{سُبْحَـٰنَ ٱللَّهِ عَمَّا يَصِفُونَ ﴿١٥٩﴾}\\
\textamh{160.\  } & \mytextarabic{إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿١٦٠﴾}\\
\textamh{161.\  } & \mytextarabic{فَإِنَّكُمْ وَمَا تَعْبُدُونَ ﴿١٦١﴾}\\
\textamh{162.\  } & \mytextarabic{مَآ أَنتُمْ عَلَيْهِ بِفَـٰتِنِينَ ﴿١٦٢﴾}\\
\textamh{163.\  } & \mytextarabic{إِلَّا مَنْ هُوَ صَالِ ٱلْجَحِيمِ ﴿١٦٣﴾}\\
\textamh{164.\  } & \mytextarabic{وَمَا مِنَّآ إِلَّا لَهُۥ مَقَامٌۭ مَّعْلُومٌۭ ﴿١٦٤﴾}\\
\textamh{165.\  } & \mytextarabic{وَإِنَّا لَنَحْنُ ٱلصَّآفُّونَ ﴿١٦٥﴾}\\
\textamh{166.\  } & \mytextarabic{وَإِنَّا لَنَحْنُ ٱلْمُسَبِّحُونَ ﴿١٦٦﴾}\\
\textamh{167.\  } & \mytextarabic{وَإِن كَانُوا۟ لَيَقُولُونَ ﴿١٦٧﴾}\\
\textamh{168.\  } & \mytextarabic{لَوْ أَنَّ عِندَنَا ذِكْرًۭا مِّنَ ٱلْأَوَّلِينَ ﴿١٦٨﴾}\\
\textamh{169.\  } & \mytextarabic{لَكُنَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿١٦٩﴾}\\
\textamh{170.\  } & \mytextarabic{فَكَفَرُوا۟ بِهِۦ ۖ فَسَوْفَ يَعْلَمُونَ ﴿١٧٠﴾}\\
\textamh{171.\  } & \mytextarabic{وَلَقَدْ سَبَقَتْ كَلِمَتُنَا لِعِبَادِنَا ٱلْمُرْسَلِينَ ﴿١٧١﴾}\\
\textamh{172.\  } & \mytextarabic{إِنَّهُمْ لَهُمُ ٱلْمَنصُورُونَ ﴿١٧٢﴾}\\
\textamh{173.\  } & \mytextarabic{وَإِنَّ جُندَنَا لَهُمُ ٱلْغَٰلِبُونَ ﴿١٧٣﴾}\\
\textamh{174.\  } & \mytextarabic{فَتَوَلَّ عَنْهُمْ حَتَّىٰ حِينٍۢ ﴿١٧٤﴾}\\
\textamh{175.\  } & \mytextarabic{وَأَبْصِرْهُمْ فَسَوْفَ يُبْصِرُونَ ﴿١٧٥﴾}\\
\textamh{176.\  } & \mytextarabic{أَفَبِعَذَابِنَا يَسْتَعْجِلُونَ ﴿١٧٦﴾}\\
\textamh{177.\  } & \mytextarabic{فَإِذَا نَزَلَ بِسَاحَتِهِمْ فَسَآءَ صَبَاحُ ٱلْمُنذَرِينَ ﴿١٧٧﴾}\\
\textamh{178.\  } & \mytextarabic{وَتَوَلَّ عَنْهُمْ حَتَّىٰ حِينٍۢ ﴿١٧٨﴾}\\
\textamh{179.\  } & \mytextarabic{وَأَبْصِرْ فَسَوْفَ يُبْصِرُونَ ﴿١٧٩﴾}\\
\textamh{180.\  } & \mytextarabic{سُبْحَـٰنَ رَبِّكَ رَبِّ ٱلْعِزَّةِ عَمَّا يَصِفُونَ ﴿١٨٠﴾}\\
\textamh{181.\  } & \mytextarabic{وَسَلَـٰمٌ عَلَى ٱلْمُرْسَلِينَ ﴿١٨١﴾}\\
\textamh{182.\  } & \mytextarabic{وَٱلْحَمْدُ لِلَّهِ رَبِّ ٱلْعَـٰلَمِينَ ﴿١٨٢﴾}\\
\end{longtable}
