\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ سَأَلَ سَآئِلٌۢ بِعَذَابٍۢ وَاقِعٍۢ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{لِّلْكَـٰفِرِينَ لَيْسَ لَهُۥ دَافِعٌۭ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{مِّنَ ٱللَّهِ ذِى ٱلْمَعَارِجِ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{تَعْرُجُ ٱلْمَلَـٰٓئِكَةُ وَٱلرُّوحُ إِلَيْهِ فِى يَوْمٍۢ كَانَ مِقْدَارُهُۥ خَمْسِينَ أَلْفَ سَنَةٍۢ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{فَٱصْبِرْ صَبْرًۭا جَمِيلًا ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{إِنَّهُمْ يَرَوْنَهُۥ بَعِيدًۭا ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{وَنَرَىٰهُ قَرِيبًۭا ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{يَوْمَ تَكُونُ ٱلسَّمَآءُ كَٱلْمُهْلِ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{وَتَكُونُ ٱلْجِبَالُ كَٱلْعِهْنِ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{وَلَا يَسْـَٔلُ حَمِيمٌ حَمِيمًۭا ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{يُبَصَّرُونَهُمْ ۚ يَوَدُّ ٱلْمُجْرِمُ لَوْ يَفْتَدِى مِنْ عَذَابِ يَوْمِئِذٍۭ بِبَنِيهِ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{وَصَـٰحِبَتِهِۦ وَأَخِيهِ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{وَفَصِيلَتِهِ ٱلَّتِى تُـْٔوِيهِ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{وَمَن فِى ٱلْأَرْضِ جَمِيعًۭا ثُمَّ يُنجِيهِ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{كَلَّآ ۖ إِنَّهَا لَظَىٰ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{نَزَّاعَةًۭ لِّلشَّوَىٰ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{تَدْعُوا۟ مَنْ أَدْبَرَ وَتَوَلَّىٰ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{وَجَمَعَ فَأَوْعَىٰٓ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{۞ إِنَّ ٱلْإِنسَـٰنَ خُلِقَ هَلُوعًا ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{إِذَا مَسَّهُ ٱلشَّرُّ جَزُوعًۭا ﴿٢٠﴾}\\
\textamh{21.\  } & \mytextarabic{وَإِذَا مَسَّهُ ٱلْخَيْرُ مَنُوعًا ﴿٢١﴾}\\
\textamh{22.\  } & \mytextarabic{إِلَّا ٱلْمُصَلِّينَ ﴿٢٢﴾}\\
\textamh{23.\  } & \mytextarabic{ٱلَّذِينَ هُمْ عَلَىٰ صَلَاتِهِمْ دَآئِمُونَ ﴿٢٣﴾}\\
\textamh{24.\  } & \mytextarabic{وَٱلَّذِينَ فِىٓ أَمْوَٟلِهِمْ حَقٌّۭ مَّعْلُومٌۭ ﴿٢٤﴾}\\
\textamh{25.\  } & \mytextarabic{لِّلسَّآئِلِ وَٱلْمَحْرُومِ ﴿٢٥﴾}\\
\textamh{26.\  } & \mytextarabic{وَٱلَّذِينَ يُصَدِّقُونَ بِيَوْمِ ٱلدِّينِ ﴿٢٦﴾}\\
\textamh{27.\  } & \mytextarabic{وَٱلَّذِينَ هُم مِّنْ عَذَابِ رَبِّهِم مُّشْفِقُونَ ﴿٢٧﴾}\\
\textamh{28.\  } & \mytextarabic{إِنَّ عَذَابَ رَبِّهِمْ غَيْرُ مَأْمُونٍۢ ﴿٢٨﴾}\\
\textamh{29.\  } & \mytextarabic{وَٱلَّذِينَ هُمْ لِفُرُوجِهِمْ حَـٰفِظُونَ ﴿٢٩﴾}\\
\textamh{30.\  } & \mytextarabic{إِلَّا عَلَىٰٓ أَزْوَٟجِهِمْ أَوْ مَا مَلَكَتْ أَيْمَـٰنُهُمْ فَإِنَّهُمْ غَيْرُ مَلُومِينَ ﴿٣٠﴾}\\
\textamh{31.\  } & \mytextarabic{فَمَنِ ٱبْتَغَىٰ وَرَآءَ ذَٟلِكَ فَأُو۟لَـٰٓئِكَ هُمُ ٱلْعَادُونَ ﴿٣١﴾}\\
\textamh{32.\  } & \mytextarabic{وَٱلَّذِينَ هُمْ لِأَمَـٰنَـٰتِهِمْ وَعَهْدِهِمْ رَٰعُونَ ﴿٣٢﴾}\\
\textamh{33.\  } & \mytextarabic{وَٱلَّذِينَ هُم بِشَهَـٰدَٟتِهِمْ قَآئِمُونَ ﴿٣٣﴾}\\
\textamh{34.\  } & \mytextarabic{وَٱلَّذِينَ هُمْ عَلَىٰ صَلَاتِهِمْ يُحَافِظُونَ ﴿٣٤﴾}\\
\textamh{35.\  } & \mytextarabic{أُو۟لَـٰٓئِكَ فِى جَنَّـٰتٍۢ مُّكْرَمُونَ ﴿٣٥﴾}\\
\textamh{36.\  } & \mytextarabic{فَمَالِ ٱلَّذِينَ كَفَرُوا۟ قِبَلَكَ مُهْطِعِينَ ﴿٣٦﴾}\\
\textamh{37.\  } & \mytextarabic{عَنِ ٱلْيَمِينِ وَعَنِ ٱلشِّمَالِ عِزِينَ ﴿٣٧﴾}\\
\textamh{38.\  } & \mytextarabic{أَيَطْمَعُ كُلُّ ٱمْرِئٍۢ مِّنْهُمْ أَن يُدْخَلَ جَنَّةَ نَعِيمٍۢ ﴿٣٨﴾}\\
\textamh{39.\  } & \mytextarabic{كَلَّآ ۖ إِنَّا خَلَقْنَـٰهُم مِّمَّا يَعْلَمُونَ ﴿٣٩﴾}\\
\textamh{40.\  } & \mytextarabic{فَلَآ أُقْسِمُ بِرَبِّ ٱلْمَشَـٰرِقِ وَٱلْمَغَٰرِبِ إِنَّا لَقَـٰدِرُونَ ﴿٤٠﴾}\\
\textamh{41.\  } & \mytextarabic{عَلَىٰٓ أَن نُّبَدِّلَ خَيْرًۭا مِّنْهُمْ وَمَا نَحْنُ بِمَسْبُوقِينَ ﴿٤١﴾}\\
\textamh{42.\  } & \mytextarabic{فَذَرْهُمْ يَخُوضُوا۟ وَيَلْعَبُوا۟ حَتَّىٰ يُلَـٰقُوا۟ يَوْمَهُمُ ٱلَّذِى يُوعَدُونَ ﴿٤٢﴾}\\
\textamh{43.\  } & \mytextarabic{يَوْمَ يَخْرُجُونَ مِنَ ٱلْأَجْدَاثِ سِرَاعًۭا كَأَنَّهُمْ إِلَىٰ نُصُبٍۢ يُوفِضُونَ ﴿٤٣﴾}\\
\textamh{44.\  } & \mytextarabic{خَـٰشِعَةً أَبْصَـٰرُهُمْ تَرْهَقُهُمْ ذِلَّةٌۭ ۚ ذَٟلِكَ ٱلْيَوْمُ ٱلَّذِى كَانُوا۟ يُوعَدُونَ ﴿٤٤﴾}\\
\end{longtable}
