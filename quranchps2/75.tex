\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ لَآ أُقْسِمُ بِيَوْمِ ٱلْقِيَـٰمَةِ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَلَآ أُقْسِمُ بِٱلنَّفْسِ ٱللَّوَّامَةِ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{أَيَحْسَبُ ٱلْإِنسَـٰنُ أَلَّن نَّجْمَعَ عِظَامَهُۥ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{بَلَىٰ قَـٰدِرِينَ عَلَىٰٓ أَن نُّسَوِّىَ بَنَانَهُۥ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{بَلْ يُرِيدُ ٱلْإِنسَـٰنُ لِيَفْجُرَ أَمَامَهُۥ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{يَسْـَٔلُ أَيَّانَ يَوْمُ ٱلْقِيَـٰمَةِ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{فَإِذَا بَرِقَ ٱلْبَصَرُ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{وَخَسَفَ ٱلْقَمَرُ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{وَجُمِعَ ٱلشَّمْسُ وَٱلْقَمَرُ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{يَقُولُ ٱلْإِنسَـٰنُ يَوْمَئِذٍ أَيْنَ ٱلْمَفَرُّ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{كَلَّا لَا وَزَرَ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{إِلَىٰ رَبِّكَ يَوْمَئِذٍ ٱلْمُسْتَقَرُّ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{يُنَبَّؤُا۟ ٱلْإِنسَـٰنُ يَوْمَئِذٍۭ بِمَا قَدَّمَ وَأَخَّرَ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{بَلِ ٱلْإِنسَـٰنُ عَلَىٰ نَفْسِهِۦ بَصِيرَةٌۭ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{وَلَوْ أَلْقَىٰ مَعَاذِيرَهُۥ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{لَا تُحَرِّكْ بِهِۦ لِسَانَكَ لِتَعْجَلَ بِهِۦٓ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{إِنَّ عَلَيْنَا جَمْعَهُۥ وَقُرْءَانَهُۥ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{فَإِذَا قَرَأْنَـٰهُ فَٱتَّبِعْ قُرْءَانَهُۥ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{ثُمَّ إِنَّ عَلَيْنَا بَيَانَهُۥ ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{كَلَّا بَلْ تُحِبُّونَ ٱلْعَاجِلَةَ ﴿٢٠﴾}\\
\textamh{21.\  } & \mytextarabic{وَتَذَرُونَ ٱلْءَاخِرَةَ ﴿٢١﴾}\\
\textamh{22.\  } & \mytextarabic{وُجُوهٌۭ يَوْمَئِذٍۢ نَّاضِرَةٌ ﴿٢٢﴾}\\
\textamh{23.\  } & \mytextarabic{إِلَىٰ رَبِّهَا نَاظِرَةٌۭ ﴿٢٣﴾}\\
\textamh{24.\  } & \mytextarabic{وَوُجُوهٌۭ يَوْمَئِذٍۭ بَاسِرَةٌۭ ﴿٢٤﴾}\\
\textamh{25.\  } & \mytextarabic{تَظُنُّ أَن يُفْعَلَ بِهَا فَاقِرَةٌۭ ﴿٢٥﴾}\\
\textamh{26.\  } & \mytextarabic{كَلَّآ إِذَا بَلَغَتِ ٱلتَّرَاقِىَ ﴿٢٦﴾}\\
\textamh{27.\  } & \mytextarabic{وَقِيلَ مَنْ ۜ رَاقٍۢ ﴿٢٧﴾}\\
\textamh{28.\  } & \mytextarabic{وَظَنَّ أَنَّهُ ٱلْفِرَاقُ ﴿٢٨﴾}\\
\textamh{29.\  } & \mytextarabic{وَٱلْتَفَّتِ ٱلسَّاقُ بِٱلسَّاقِ ﴿٢٩﴾}\\
\textamh{30.\  } & \mytextarabic{إِلَىٰ رَبِّكَ يَوْمَئِذٍ ٱلْمَسَاقُ ﴿٣٠﴾}\\
\textamh{31.\  } & \mytextarabic{فَلَا صَدَّقَ وَلَا صَلَّىٰ ﴿٣١﴾}\\
\textamh{32.\  } & \mytextarabic{وَلَـٰكِن كَذَّبَ وَتَوَلَّىٰ ﴿٣٢﴾}\\
\textamh{33.\  } & \mytextarabic{ثُمَّ ذَهَبَ إِلَىٰٓ أَهْلِهِۦ يَتَمَطَّىٰٓ ﴿٣٣﴾}\\
\textamh{34.\  } & \mytextarabic{أَوْلَىٰ لَكَ فَأَوْلَىٰ ﴿٣٤﴾}\\
\textamh{35.\  } & \mytextarabic{ثُمَّ أَوْلَىٰ لَكَ فَأَوْلَىٰٓ ﴿٣٥﴾}\\
\textamh{36.\  } & \mytextarabic{أَيَحْسَبُ ٱلْإِنسَـٰنُ أَن يُتْرَكَ سُدًى ﴿٣٦﴾}\\
\textamh{37.\  } & \mytextarabic{أَلَمْ يَكُ نُطْفَةًۭ مِّن مَّنِىٍّۢ يُمْنَىٰ ﴿٣٧﴾}\\
\textamh{38.\  } & \mytextarabic{ثُمَّ كَانَ عَلَقَةًۭ فَخَلَقَ فَسَوَّىٰ ﴿٣٨﴾}\\
\textamh{39.\  } & \mytextarabic{فَجَعَلَ مِنْهُ ٱلزَّوْجَيْنِ ٱلذَّكَرَ وَٱلْأُنثَىٰٓ ﴿٣٩﴾}\\
\textamh{40.\  } & \mytextarabic{أَلَيْسَ ذَٟلِكَ بِقَـٰدِرٍ عَلَىٰٓ أَن يُحْۦِىَ ٱلْمَوْتَىٰ ﴿٤٠﴾}\\
\end{longtable}
