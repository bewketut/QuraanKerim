\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ وَٱلنَّـٰزِعَـٰتِ غَرْقًۭا ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَٱلنَّـٰشِطَٰتِ نَشْطًۭا ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{وَٱلسَّٰبِحَـٰتِ سَبْحًۭا ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{فَٱلسَّٰبِقَـٰتِ سَبْقًۭا ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{فَٱلْمُدَبِّرَٰتِ أَمْرًۭا ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{يَوْمَ تَرْجُفُ ٱلرَّاجِفَةُ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{تَتْبَعُهَا ٱلرَّادِفَةُ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{قُلُوبٌۭ يَوْمَئِذٍۢ وَاجِفَةٌ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{أَبْصَـٰرُهَا خَـٰشِعَةٌۭ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{يَقُولُونَ أَءِنَّا لَمَرْدُودُونَ فِى ٱلْحَافِرَةِ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{أَءِذَا كُنَّا عِظَـٰمًۭا نَّخِرَةًۭ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{قَالُوا۟ تِلْكَ إِذًۭا كَرَّةٌ خَاسِرَةٌۭ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{فَإِنَّمَا هِىَ زَجْرَةٌۭ وَٟحِدَةٌۭ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{فَإِذَا هُم بِٱلسَّاهِرَةِ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{هَلْ أَتَىٰكَ حَدِيثُ مُوسَىٰٓ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{إِذْ نَادَىٰهُ رَبُّهُۥ بِٱلْوَادِ ٱلْمُقَدَّسِ طُوًى ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{ٱذْهَبْ إِلَىٰ فِرْعَوْنَ إِنَّهُۥ طَغَىٰ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{فَقُلْ هَل لَّكَ إِلَىٰٓ أَن تَزَكَّىٰ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{وَأَهْدِيَكَ إِلَىٰ رَبِّكَ فَتَخْشَىٰ ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{فَأَرَىٰهُ ٱلْءَايَةَ ٱلْكُبْرَىٰ ﴿٢٠﴾}\\
\textamh{21.\  } & \mytextarabic{فَكَذَّبَ وَعَصَىٰ ﴿٢١﴾}\\
\textamh{22.\  } & \mytextarabic{ثُمَّ أَدْبَرَ يَسْعَىٰ ﴿٢٢﴾}\\
\textamh{23.\  } & \mytextarabic{فَحَشَرَ فَنَادَىٰ ﴿٢٣﴾}\\
\textamh{24.\  } & \mytextarabic{فَقَالَ أَنَا۠ رَبُّكُمُ ٱلْأَعْلَىٰ ﴿٢٤﴾}\\
\textamh{25.\  } & \mytextarabic{فَأَخَذَهُ ٱللَّهُ نَكَالَ ٱلْءَاخِرَةِ وَٱلْأُولَىٰٓ ﴿٢٥﴾}\\
\textamh{26.\  } & \mytextarabic{إِنَّ فِى ذَٟلِكَ لَعِبْرَةًۭ لِّمَن يَخْشَىٰٓ ﴿٢٦﴾}\\
\textamh{27.\  } & \mytextarabic{ءَأَنتُمْ أَشَدُّ خَلْقًا أَمِ ٱلسَّمَآءُ ۚ بَنَىٰهَا ﴿٢٧﴾}\\
\textamh{28.\  } & \mytextarabic{رَفَعَ سَمْكَهَا فَسَوَّىٰهَا ﴿٢٨﴾}\\
\textamh{29.\  } & \mytextarabic{وَأَغْطَشَ لَيْلَهَا وَأَخْرَجَ ضُحَىٰهَا ﴿٢٩﴾}\\
\textamh{30.\  } & \mytextarabic{وَٱلْأَرْضَ بَعْدَ ذَٟلِكَ دَحَىٰهَآ ﴿٣٠﴾}\\
\textamh{31.\  } & \mytextarabic{أَخْرَجَ مِنْهَا مَآءَهَا وَمَرْعَىٰهَا ﴿٣١﴾}\\
\textamh{32.\  } & \mytextarabic{وَٱلْجِبَالَ أَرْسَىٰهَا ﴿٣٢﴾}\\
\textamh{33.\  } & \mytextarabic{مَتَـٰعًۭا لَّكُمْ وَلِأَنْعَـٰمِكُمْ ﴿٣٣﴾}\\
\textamh{34.\  } & \mytextarabic{فَإِذَا جَآءَتِ ٱلطَّآمَّةُ ٱلْكُبْرَىٰ ﴿٣٤﴾}\\
\textamh{35.\  } & \mytextarabic{يَوْمَ يَتَذَكَّرُ ٱلْإِنسَـٰنُ مَا سَعَىٰ ﴿٣٥﴾}\\
\textamh{36.\  } & \mytextarabic{وَبُرِّزَتِ ٱلْجَحِيمُ لِمَن يَرَىٰ ﴿٣٦﴾}\\
\textamh{37.\  } & \mytextarabic{فَأَمَّا مَن طَغَىٰ ﴿٣٧﴾}\\
\textamh{38.\  } & \mytextarabic{وَءَاثَرَ ٱلْحَيَوٰةَ ٱلدُّنْيَا ﴿٣٨﴾}\\
\textamh{39.\  } & \mytextarabic{فَإِنَّ ٱلْجَحِيمَ هِىَ ٱلْمَأْوَىٰ ﴿٣٩﴾}\\
\textamh{40.\  } & \mytextarabic{وَأَمَّا مَنْ خَافَ مَقَامَ رَبِّهِۦ وَنَهَى ٱلنَّفْسَ عَنِ ٱلْهَوَىٰ ﴿٤٠﴾}\\
\textamh{41.\  } & \mytextarabic{فَإِنَّ ٱلْجَنَّةَ هِىَ ٱلْمَأْوَىٰ ﴿٤١﴾}\\
\textamh{42.\  } & \mytextarabic{يَسْـَٔلُونَكَ عَنِ ٱلسَّاعَةِ أَيَّانَ مُرْسَىٰهَا ﴿٤٢﴾}\\
\textamh{43.\  } & \mytextarabic{فِيمَ أَنتَ مِن ذِكْرَىٰهَآ ﴿٤٣﴾}\\
\textamh{44.\  } & \mytextarabic{إِلَىٰ رَبِّكَ مُنتَهَىٰهَآ ﴿٤٤﴾}\\
\textamh{45.\  } & \mytextarabic{إِنَّمَآ أَنتَ مُنذِرُ مَن يَخْشَىٰهَا ﴿٤٥﴾}\\
\textamh{46.\  } & \mytextarabic{كَأَنَّهُمْ يَوْمَ يَرَوْنَهَا لَمْ يَلْبَثُوٓا۟ إِلَّا عَشِيَّةً أَوْ ضُحَىٰهَا ﴿٤٦﴾}\\
\end{longtable}
