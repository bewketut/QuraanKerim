\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ إِذَا ٱلشَّمْسُ كُوِّرَتْ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَإِذَا ٱلنُّجُومُ ٱنكَدَرَتْ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{وَإِذَا ٱلْجِبَالُ سُيِّرَتْ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{وَإِذَا ٱلْعِشَارُ عُطِّلَتْ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{وَإِذَا ٱلْوُحُوشُ حُشِرَتْ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{وَإِذَا ٱلْبِحَارُ سُجِّرَتْ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{وَإِذَا ٱلنُّفُوسُ زُوِّجَتْ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{وَإِذَا ٱلْمَوْءُۥدَةُ سُئِلَتْ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{بِأَىِّ ذَنۢبٍۢ قُتِلَتْ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{وَإِذَا ٱلصُّحُفُ نُشِرَتْ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{وَإِذَا ٱلسَّمَآءُ كُشِطَتْ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{وَإِذَا ٱلْجَحِيمُ سُعِّرَتْ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{وَإِذَا ٱلْجَنَّةُ أُزْلِفَتْ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{عَلِمَتْ نَفْسٌۭ مَّآ أَحْضَرَتْ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{فَلَآ أُقْسِمُ بِٱلْخُنَّسِ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{ٱلْجَوَارِ ٱلْكُنَّسِ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{وَٱلَّيْلِ إِذَا عَسْعَسَ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{وَٱلصُّبْحِ إِذَا تَنَفَّسَ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{إِنَّهُۥ لَقَوْلُ رَسُولٍۢ كَرِيمٍۢ ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{ذِى قُوَّةٍ عِندَ ذِى ٱلْعَرْشِ مَكِينٍۢ ﴿٢٠﴾}\\
\textamh{21.\  } & \mytextarabic{مُّطَاعٍۢ ثَمَّ أَمِينٍۢ ﴿٢١﴾}\\
\textamh{22.\  } & \mytextarabic{وَمَا صَاحِبُكُم بِمَجْنُونٍۢ ﴿٢٢﴾}\\
\textamh{23.\  } & \mytextarabic{وَلَقَدْ رَءَاهُ بِٱلْأُفُقِ ٱلْمُبِينِ ﴿٢٣﴾}\\
\textamh{24.\  } & \mytextarabic{وَمَا هُوَ عَلَى ٱلْغَيْبِ بِضَنِينٍۢ ﴿٢٤﴾}\\
\textamh{25.\  } & \mytextarabic{وَمَا هُوَ بِقَوْلِ شَيْطَٰنٍۢ رَّجِيمٍۢ ﴿٢٥﴾}\\
\textamh{26.\  } & \mytextarabic{فَأَيْنَ تَذْهَبُونَ ﴿٢٦﴾}\\
\textamh{27.\  } & \mytextarabic{إِنْ هُوَ إِلَّا ذِكْرٌۭ لِّلْعَـٰلَمِينَ ﴿٢٧﴾}\\
\textamh{28.\  } & \mytextarabic{لِمَن شَآءَ مِنكُمْ أَن يَسْتَقِيمَ ﴿٢٨﴾}\\
\textamh{29.\  } & \mytextarabic{وَمَا تَشَآءُونَ إِلَّآ أَن يَشَآءَ ٱللَّهُ رَبُّ ٱلْعَـٰلَمِينَ ﴿٢٩﴾}\\
\end{longtable}
