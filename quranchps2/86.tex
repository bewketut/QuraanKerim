\begin{center}\section{\LR{\textamhsec{ሱራቱ አጥጣሪቂ -}  \textarabic{سوره  الطارق}}}\end{center}
\begin{longtable}{%
  @{}
    p{.5\textwidth}
  @{~~~}
    p{.5\textwidth}
    @{}
}
\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ وَٱلسَّمَآءِ وَٱلطَّارِقِ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَمَآ أَدْرَىٰكَ مَا ٱلطَّارِقُ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{ٱلنَّجْمُ ٱلثَّاقِبُ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{إِن كُلُّ نَفْسٍۢ لَّمَّا عَلَيْهَا حَافِظٌۭ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{فَلْيَنظُرِ ٱلْإِنسَـٰنُ مِمَّ خُلِقَ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{خُلِقَ مِن مَّآءٍۢ دَافِقٍۢ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{يَخْرُجُ مِنۢ بَيْنِ ٱلصُّلْبِ وَٱلتَّرَآئِبِ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{إِنَّهُۥ عَلَىٰ رَجْعِهِۦ لَقَادِرٌۭ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{يَوْمَ تُبْلَى ٱلسَّرَآئِرُ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{فَمَا لَهُۥ مِن قُوَّةٍۢ وَلَا نَاصِرٍۢ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{وَٱلسَّمَآءِ ذَاتِ ٱلرَّجْعِ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{وَٱلْأَرْضِ ذَاتِ ٱلصَّدْعِ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{إِنَّهُۥ لَقَوْلٌۭ فَصْلٌۭ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{وَمَا هُوَ بِٱلْهَزْلِ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{إِنَّهُمْ يَكِيدُونَ كَيْدًۭا ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{وَأَكِيدُ كَيْدًۭا ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{فَمَهِّلِ ٱلْكَـٰفِرِينَ أَمْهِلْهُمْ رُوَيْدًۢا ﴿١٧﴾}\\
\end{longtable}
