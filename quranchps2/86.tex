\nopagebreak
\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ\\
\textamh{1.\  } &  وَٱلسَّمَآءِ وَٱلطَّارِقِ ﴿١﴾\\
\textamh{2.\  } & وَمَآ أَدْرَىٰكَ مَا ٱلطَّارِقُ ﴿٢﴾\\
\textamh{3.\  } & ٱلنَّجْمُ ٱلثَّاقِبُ ﴿٣﴾\\
\textamh{4.\  } & إِن كُلُّ نَفْسٍۢ لَّمَّا عَلَيْهَا حَافِظٌۭ ﴿٤﴾\\
\textamh{5.\  } & فَلْيَنظُرِ ٱلْإِنسَـٰنُ مِمَّ خُلِقَ ﴿٥﴾\\
\textamh{6.\  } & خُلِقَ مِن مَّآءٍۢ دَافِقٍۢ ﴿٦﴾\\
\textamh{7.\  } & يَخْرُجُ مِنۢ بَيْنِ ٱلصُّلْبِ وَٱلتَّرَآئِبِ ﴿٧﴾\\
\textamh{8.\  } & إِنَّهُۥ عَلَىٰ رَجْعِهِۦ لَقَادِرٌۭ ﴿٨﴾\\
\textamh{9.\  } & يَوْمَ تُبْلَى ٱلسَّرَآئِرُ ﴿٩﴾\\
\textamh{10.\  } & فَمَا لَهُۥ مِن قُوَّةٍۢ وَلَا نَاصِرٍۢ ﴿١٠﴾\\
\textamh{11.\  } & وَٱلسَّمَآءِ ذَاتِ ٱلرَّجْعِ ﴿١١﴾\\
\textamh{12.\  } & وَٱلْأَرْضِ ذَاتِ ٱلصَّدْعِ ﴿١٢﴾\\
\textamh{13.\  } & إِنَّهُۥ لَقَوْلٌۭ فَصْلٌۭ ﴿١٣﴾\\
\textamh{14.\  } & وَمَا هُوَ بِٱلْهَزْلِ ﴿١٤﴾\\
\textamh{15.\  } & إِنَّهُمْ يَكِيدُونَ كَيْدًۭا ﴿١٥﴾\\
\textamh{16.\  } & وَأَكِيدُ كَيْدًۭا ﴿١٦﴾\\
\textamh{17.\  } & فَمَهِّلِ ٱلْكَـٰفِرِينَ أَمْهِلْهُمْ رُوَيْدًۢا ﴿١٧﴾\\
\end{longtable}
