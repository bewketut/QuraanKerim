  @{~~~}

}
\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ سَبِّحِ ٱسْمَ رَبِّكَ ٱلْأَعْلَى ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{ٱلَّذِى خَلَقَ فَسَوَّىٰ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{وَٱلَّذِى قَدَّرَ فَهَدَىٰ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{وَٱلَّذِىٓ أَخْرَجَ ٱلْمَرْعَىٰ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{فَجَعَلَهُۥ غُثَآءً أَحْوَىٰ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{سَنُقْرِئُكَ فَلَا تَنسَىٰٓ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{إِلَّا مَا شَآءَ ٱللَّهُ ۚ إِنَّهُۥ يَعْلَمُ ٱلْجَهْرَ وَمَا يَخْفَىٰ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{وَنُيَسِّرُكَ لِلْيُسْرَىٰ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{فَذَكِّرْ إِن نَّفَعَتِ ٱلذِّكْرَىٰ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{سَيَذَّكَّرُ مَن يَخْشَىٰ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{وَيَتَجَنَّبُهَا ٱلْأَشْقَى ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{ٱلَّذِى يَصْلَى ٱلنَّارَ ٱلْكُبْرَىٰ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{ثُمَّ لَا يَمُوتُ فِيهَا وَلَا يَحْيَىٰ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{قَدْ أَفْلَحَ مَن تَزَكَّىٰ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{وَذَكَرَ ٱسْمَ رَبِّهِۦ فَصَلَّىٰ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{بَلْ تُؤْثِرُونَ ٱلْحَيَوٰةَ ٱلدُّنْيَا ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{وَٱلْءَاخِرَةُ خَيْرٌۭ وَأَبْقَىٰٓ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{إِنَّ هَـٰذَا لَفِى ٱلصُّحُفِ ٱلْأُولَىٰ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{صُحُفِ إِبْرَٰهِيمَ وَمُوسَىٰ ﴿١٩﴾}\\
\end{longtable}
