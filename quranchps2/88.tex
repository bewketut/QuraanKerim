\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ هَلْ أَتَىٰكَ حَدِيثُ ٱلْغَٰشِيَةِ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وُجُوهٌۭ يَوْمَئِذٍ خَـٰشِعَةٌ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{عَامِلَةٌۭ نَّاصِبَةٌۭ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{تَصْلَىٰ نَارًا حَامِيَةًۭ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{تُسْقَىٰ مِنْ عَيْنٍ ءَانِيَةٍۢ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{لَّيْسَ لَهُمْ طَعَامٌ إِلَّا مِن ضَرِيعٍۢ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{لَّا يُسْمِنُ وَلَا يُغْنِى مِن جُوعٍۢ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{وُجُوهٌۭ يَوْمَئِذٍۢ نَّاعِمَةٌۭ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{لِّسَعْيِهَا رَاضِيَةٌۭ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{فِى جَنَّةٍ عَالِيَةٍۢ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{لَّا تَسْمَعُ فِيهَا لَـٰغِيَةًۭ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{فِيهَا عَيْنٌۭ جَارِيَةٌۭ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{فِيهَا سُرُرٌۭ مَّرْفُوعَةٌۭ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{وَأَكْوَابٌۭ مَّوْضُوعَةٌۭ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{وَنَمَارِقُ مَصْفُوفَةٌۭ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{وَزَرَابِىُّ مَبْثُوثَةٌ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{أَفَلَا يَنظُرُونَ إِلَى ٱلْإِبِلِ كَيْفَ خُلِقَتْ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{وَإِلَى ٱلسَّمَآءِ كَيْفَ رُفِعَتْ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{وَإِلَى ٱلْجِبَالِ كَيْفَ نُصِبَتْ ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{وَإِلَى ٱلْأَرْضِ كَيْفَ سُطِحَتْ ﴿٢٠﴾}\\
\textamh{21.\  } & \mytextarabic{فَذَكِّرْ إِنَّمَآ أَنتَ مُذَكِّرٌۭ ﴿٢١﴾}\\
\textamh{22.\  } & \mytextarabic{لَّسْتَ عَلَيْهِم بِمُصَيْطِرٍ ﴿٢٢﴾}\\
\textamh{23.\  } & \mytextarabic{إِلَّا مَن تَوَلَّىٰ وَكَفَرَ ﴿٢٣﴾}\\
\textamh{24.\  } & \mytextarabic{فَيُعَذِّبُهُ ٱللَّهُ ٱلْعَذَابَ ٱلْأَكْبَرَ ﴿٢٤﴾}\\
\textamh{25.\  } & \mytextarabic{إِنَّ إِلَيْنَآ إِيَابَهُمْ ﴿٢٥﴾}\\
\textamh{26.\  } & \mytextarabic{ثُمَّ إِنَّ عَلَيْنَا حِسَابَهُم ﴿٢٦﴾}\\
\end{longtable}
