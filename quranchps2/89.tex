\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ وَٱلْفَجْرِ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَلَيَالٍ عَشْرٍۢ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{وَٱلشَّفْعِ وَٱلْوَتْرِ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{وَٱلَّيْلِ إِذَا يَسْرِ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{هَلْ فِى ذَٟلِكَ قَسَمٌۭ لِّذِى حِجْرٍ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{أَلَمْ تَرَ كَيْفَ فَعَلَ رَبُّكَ بِعَادٍ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{إِرَمَ ذَاتِ ٱلْعِمَادِ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{ٱلَّتِى لَمْ يُخْلَقْ مِثْلُهَا فِى ٱلْبِلَـٰدِ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{وَثَمُودَ ٱلَّذِينَ جَابُوا۟ ٱلصَّخْرَ بِٱلْوَادِ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{وَفِرْعَوْنَ ذِى ٱلْأَوْتَادِ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{ٱلَّذِينَ طَغَوْا۟ فِى ٱلْبِلَـٰدِ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{فَأَكْثَرُوا۟ فِيهَا ٱلْفَسَادَ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{فَصَبَّ عَلَيْهِمْ رَبُّكَ سَوْطَ عَذَابٍ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{إِنَّ رَبَّكَ لَبِٱلْمِرْصَادِ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{فَأَمَّا ٱلْإِنسَـٰنُ إِذَا مَا ٱبْتَلَىٰهُ رَبُّهُۥ فَأَكْرَمَهُۥ وَنَعَّمَهُۥ فَيَقُولُ رَبِّىٓ أَكْرَمَنِ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{وَأَمَّآ إِذَا مَا ٱبْتَلَىٰهُ فَقَدَرَ عَلَيْهِ رِزْقَهُۥ فَيَقُولُ رَبِّىٓ أَهَـٰنَنِ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{كَلَّا ۖ بَل لَّا تُكْرِمُونَ ٱلْيَتِيمَ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{وَلَا تَحَـٰٓضُّونَ عَلَىٰ طَعَامِ ٱلْمِسْكِينِ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{وَتَأْكُلُونَ ٱلتُّرَاثَ أَكْلًۭا لَّمًّۭا ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{وَتُحِبُّونَ ٱلْمَالَ حُبًّۭا جَمًّۭا ﴿٢٠﴾}\\
\textamh{21.\  } & \mytextarabic{كَلَّآ إِذَا دُكَّتِ ٱلْأَرْضُ دَكًّۭا دَكًّۭا ﴿٢١﴾}\\
\textamh{22.\  } & \mytextarabic{وَجَآءَ رَبُّكَ وَٱلْمَلَكُ صَفًّۭا صَفًّۭا ﴿٢٢﴾}\\
\textamh{23.\  } & \mytextarabic{وَجِا۟ىٓءَ يَوْمَئِذٍۭ بِجَهَنَّمَ ۚ يَوْمَئِذٍۢ يَتَذَكَّرُ ٱلْإِنسَـٰنُ وَأَنَّىٰ لَهُ ٱلذِّكْرَىٰ ﴿٢٣﴾}\\
\textamh{24.\  } & \mytextarabic{يَقُولُ يَـٰلَيْتَنِى قَدَّمْتُ لِحَيَاتِى ﴿٢٤﴾}\\
\textamh{25.\  } & \mytextarabic{فَيَوْمَئِذٍۢ لَّا يُعَذِّبُ عَذَابَهُۥٓ أَحَدٌۭ ﴿٢٥﴾}\\
\textamh{26.\  } & \mytextarabic{وَلَا يُوثِقُ وَثَاقَهُۥٓ أَحَدٌۭ ﴿٢٦﴾}\\
\textamh{27.\  } & \mytextarabic{يَـٰٓأَيَّتُهَا ٱلنَّفْسُ ٱلْمُطْمَئِنَّةُ ﴿٢٧﴾}\\
\textamh{28.\  } & \mytextarabic{ٱرْجِعِىٓ إِلَىٰ رَبِّكِ رَاضِيَةًۭ مَّرْضِيَّةًۭ ﴿٢٨﴾}\\
\textamh{29.\  } & \mytextarabic{فَٱدْخُلِى فِى عِبَٰدِى ﴿٢٩﴾}\\
\textamh{30.\  } & \mytextarabic{وَٱدْخُلِى جَنَّتِى ﴿٣٠﴾}\\
\end{longtable}
