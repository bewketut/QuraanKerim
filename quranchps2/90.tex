\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ لَآ أُقْسِمُ بِهَـٰذَا ٱلْبَلَدِ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَأَنتَ حِلٌّۢ بِهَـٰذَا ٱلْبَلَدِ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{وَوَالِدٍۢ وَمَا وَلَدَ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{لَقَدْ خَلَقْنَا ٱلْإِنسَـٰنَ فِى كَبَدٍ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{أَيَحْسَبُ أَن لَّن يَقْدِرَ عَلَيْهِ أَحَدٌۭ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{يَقُولُ أَهْلَكْتُ مَالًۭا لُّبَدًا ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{أَيَحْسَبُ أَن لَّمْ يَرَهُۥٓ أَحَدٌ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{أَلَمْ نَجْعَل لَّهُۥ عَيْنَيْنِ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{وَلِسَانًۭا وَشَفَتَيْنِ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{وَهَدَيْنَـٰهُ ٱلنَّجْدَيْنِ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{فَلَا ٱقْتَحَمَ ٱلْعَقَبَةَ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{وَمَآ أَدْرَىٰكَ مَا ٱلْعَقَبَةُ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{فَكُّ رَقَبَةٍ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{أَوْ إِطْعَـٰمٌۭ فِى يَوْمٍۢ ذِى مَسْغَبَةٍۢ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{يَتِيمًۭا ذَا مَقْرَبَةٍ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{أَوْ مِسْكِينًۭا ذَا مَتْرَبَةٍۢ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{ثُمَّ كَانَ مِنَ ٱلَّذِينَ ءَامَنُوا۟ وَتَوَاصَوْا۟ بِٱلصَّبْرِ وَتَوَاصَوْا۟ بِٱلْمَرْحَمَةِ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{أُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلْمَيْمَنَةِ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{وَٱلَّذِينَ كَفَرُوا۟ بِـَٔايَـٰتِنَا هُمْ أَصْحَـٰبُ ٱلْمَشْـَٔمَةِ ﴿١٩﴾}\\
\textamh{20.\  } & \mytextarabic{عَلَيْهِمْ نَارٌۭ مُّؤْصَدَةٌۢ ﴿٢٠﴾}\\
\end{longtable}
