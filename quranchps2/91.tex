\begin{center}\section{\LR{\textamhsec{ሱራቱ አልሸምስ -}  \textarabic{سوره  الشمس}}}\end{center}

  @{}
    p{.5\textwidth}
  @{~~~}
    p{.5\textwidth}
    @{}
}
\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ وَٱلشَّمْسِ وَضُحَىٰهَا ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَٱلْقَمَرِ إِذَا تَلَىٰهَا ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{وَٱلنَّهَارِ إِذَا جَلَّىٰهَا ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{وَٱلَّيْلِ إِذَا يَغْشَىٰهَا ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{وَٱلسَّمَآءِ وَمَا بَنَىٰهَا ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{وَٱلْأَرْضِ وَمَا طَحَىٰهَا ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{وَنَفْسٍۢ وَمَا سَوَّىٰهَا ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{فَأَلْهَمَهَا فُجُورَهَا وَتَقْوَىٰهَا ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{قَدْ أَفْلَحَ مَن زَكَّىٰهَا ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{وَقَدْ خَابَ مَن دَسَّىٰهَا ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{كَذَّبَتْ ثَمُودُ بِطَغْوَىٰهَآ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{إِذِ ٱنۢبَعَثَ أَشْقَىٰهَا ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{فَقَالَ لَهُمْ رَسُولُ ٱللَّهِ نَاقَةَ ٱللَّهِ وَسُقْيَـٰهَا ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{فَكَذَّبُوهُ فَعَقَرُوهَا فَدَمْدَمَ عَلَيْهِمْ رَبُّهُم بِذَنۢبِهِمْ فَسَوَّىٰهَا ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{وَلَا يَخَافُ عُقْبَٰهَا ﴿١٥﴾}\\
\end{longtable}
