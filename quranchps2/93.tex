  @{}
    p{.5\textwidth}
  @{~~~}

}
\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ وَٱلضُّحَىٰ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَٱلَّيْلِ إِذَا سَجَىٰ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{مَا وَدَّعَكَ رَبُّكَ وَمَا قَلَىٰ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{وَلَلْءَاخِرَةُ خَيْرٌۭ لَّكَ مِنَ ٱلْأُولَىٰ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{وَلَسَوْفَ يُعْطِيكَ رَبُّكَ فَتَرْضَىٰٓ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{أَلَمْ يَجِدْكَ يَتِيمًۭا فَـَٔاوَىٰ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{وَوَجَدَكَ ضَآلًّۭا فَهَدَىٰ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{وَوَجَدَكَ عَآئِلًۭا فَأَغْنَىٰ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{فَأَمَّا ٱلْيَتِيمَ فَلَا تَقْهَرْ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{وَأَمَّا ٱلسَّآئِلَ فَلَا تَنْهَرْ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{وَأَمَّا بِنِعْمَةِ رَبِّكَ فَحَدِّثْ ﴿١١﴾}\\
\end{longtable}
