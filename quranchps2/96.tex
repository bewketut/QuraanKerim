\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ ٱقْرَأْ بِٱسْمِ رَبِّكَ ٱلَّذِى خَلَقَ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{خَلَقَ ٱلْإِنسَـٰنَ مِنْ عَلَقٍ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{ٱقْرَأْ وَرَبُّكَ ٱلْأَكْرَمُ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{ٱلَّذِى عَلَّمَ بِٱلْقَلَمِ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{عَلَّمَ ٱلْإِنسَـٰنَ مَا لَمْ يَعْلَمْ ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{كَلَّآ إِنَّ ٱلْإِنسَـٰنَ لَيَطْغَىٰٓ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{أَن رَّءَاهُ ٱسْتَغْنَىٰٓ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{إِنَّ إِلَىٰ رَبِّكَ ٱلرُّجْعَىٰٓ ﴿٨﴾}\\
\textamh{9.\  } & \mytextarabic{أَرَءَيْتَ ٱلَّذِى يَنْهَىٰ ﴿٩﴾}\\
\textamh{10.\  } & \mytextarabic{عَبْدًا إِذَا صَلَّىٰٓ ﴿١٠﴾}\\
\textamh{11.\  } & \mytextarabic{أَرَءَيْتَ إِن كَانَ عَلَى ٱلْهُدَىٰٓ ﴿١١﴾}\\
\textamh{12.\  } & \mytextarabic{أَوْ أَمَرَ بِٱلتَّقْوَىٰٓ ﴿١٢﴾}\\
\textamh{13.\  } & \mytextarabic{أَرَءَيْتَ إِن كَذَّبَ وَتَوَلَّىٰٓ ﴿١٣﴾}\\
\textamh{14.\  } & \mytextarabic{أَلَمْ يَعْلَم بِأَنَّ ٱللَّهَ يَرَىٰ ﴿١٤﴾}\\
\textamh{15.\  } & \mytextarabic{كَلَّا لَئِن لَّمْ يَنتَهِ لَنَسْفَعًۢا بِٱلنَّاصِيَةِ ﴿١٥﴾}\\
\textamh{16.\  } & \mytextarabic{نَاصِيَةٍۢ كَـٰذِبَةٍ خَاطِئَةٍۢ ﴿١٦﴾}\\
\textamh{17.\  } & \mytextarabic{فَلْيَدْعُ نَادِيَهُۥ ﴿١٧﴾}\\
\textamh{18.\  } & \mytextarabic{سَنَدْعُ ٱلزَّبَانِيَةَ ﴿١٨﴾}\\
\textamh{19.\  } & \mytextarabic{كَلَّا لَا تُطِعْهُ وَٱسْجُدْ وَٱقْتَرِب ۩ ﴿١٩﴾}\\
\end{longtable}
