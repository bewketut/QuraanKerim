\textamh{\ \ \ \ \ \  ቢስሚላሂ አራህመኒ ራሂይም } &  بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ\\
\textamh{1.\  } &  ٱقْرَأْ بِٱسْمِ رَبِّكَ ٱلَّذِى خَلَقَ ﴿١﴾\\
\textamh{2.\  } & خَلَقَ ٱلْإِنسَـٰنَ مِنْ عَلَقٍ ﴿٢﴾\\
\textamh{3.\  } & ٱقْرَأْ وَرَبُّكَ ٱلْأَكْرَمُ ﴿٣﴾\\
\textamh{4.\  } & ٱلَّذِى عَلَّمَ بِٱلْقَلَمِ ﴿٤﴾\\
\textamh{5.\  } & عَلَّمَ ٱلْإِنسَـٰنَ مَا لَمْ يَعْلَمْ ﴿٥﴾\\
\textamh{6.\  } & كَلَّآ إِنَّ ٱلْإِنسَـٰنَ لَيَطْغَىٰٓ ﴿٦﴾\\
\textamh{7.\  } & أَن رَّءَاهُ ٱسْتَغْنَىٰٓ ﴿٧﴾\\
\textamh{8.\  } & إِنَّ إِلَىٰ رَبِّكَ ٱلرُّجْعَىٰٓ ﴿٨﴾\\
\textamh{9.\  } & أَرَءَيْتَ ٱلَّذِى يَنْهَىٰ ﴿٩﴾\\
\textamh{10.\  } & عَبْدًا إِذَا صَلَّىٰٓ ﴿١٠﴾\\
\textamh{11.\  } & أَرَءَيْتَ إِن كَانَ عَلَى ٱلْهُدَىٰٓ ﴿١١﴾\\
\textamh{12.\  } & أَوْ أَمَرَ بِٱلتَّقْوَىٰٓ ﴿١٢﴾\\
\textamh{13.\  } & أَرَءَيْتَ إِن كَذَّبَ وَتَوَلَّىٰٓ ﴿١٣﴾\\
\textamh{14.\  } & أَلَمْ يَعْلَم بِأَنَّ ٱللَّهَ يَرَىٰ ﴿١٤﴾\\
\textamh{15.\  } & كَلَّا لَئِن لَّمْ يَنتَهِ لَنَسْفَعًۢا بِٱلنَّاصِيَةِ ﴿١٥﴾\\
\textamh{16.\  } & نَاصِيَةٍۢ كَـٰذِبَةٍ خَاطِئَةٍۢ ﴿١٦﴾\\
\textamh{17.\  } & فَلْيَدْعُ نَادِيَهُۥ ﴿١٧﴾\\
\textamh{18.\  } & سَنَدْعُ ٱلزَّبَانِيَةَ ﴿١٨﴾\\
\textamh{19.\  } & كَلَّا لَا تُطِعْهُ وَٱسْجُدْ وَٱقْتَرِب ۩ ﴿١٩﴾\\
\end{longtable} \newpage
