\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{بِّسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ إِنَّآ أَنزَلْنَـٰهُ فِى لَيْلَةِ ٱلْقَدْرِ ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَمَآ أَدْرَىٰكَ مَا لَيْلَةُ ٱلْقَدْرِ ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{لَيْلَةُ ٱلْقَدْرِ خَيْرٌۭ مِّنْ أَلْفِ شَهْرٍۢ ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{تَنَزَّلُ ٱلْمَلَـٰٓئِكَةُ وَٱلرُّوحُ فِيهَا بِإِذْنِ رَبِّهِم مِّن كُلِّ أَمْرٍۢ ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{سَلَـٰمٌ هِىَ حَتَّىٰ مَطْلَعِ ٱلْفَجْرِ ﴿٥﴾}\\
\end{longtable}
