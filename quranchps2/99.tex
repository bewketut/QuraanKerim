\begin{center}\section{\LR{\textamhsec{ሱራቱ አልዘልዘላ -}  \textarabic{سوره  الزلزلة}}}\end{center}

  @{~~~}

}
\textamh{ቢስሚላሂ አራህመኒ ራሂይም } &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
\textamh{1.\  } & \mytextarabic{ إِذَا زُلْزِلَتِ ٱلْأَرْضُ زِلْزَالَهَا ﴿١﴾}\\
\textamh{2.\  } & \mytextarabic{وَأَخْرَجَتِ ٱلْأَرْضُ أَثْقَالَهَا ﴿٢﴾}\\
\textamh{3.\  } & \mytextarabic{وَقَالَ ٱلْإِنسَـٰنُ مَا لَهَا ﴿٣﴾}\\
\textamh{4.\  } & \mytextarabic{يَوْمَئِذٍۢ تُحَدِّثُ أَخْبَارَهَا ﴿٤﴾}\\
\textamh{5.\  } & \mytextarabic{بِأَنَّ رَبَّكَ أَوْحَىٰ لَهَا ﴿٥﴾}\\
\textamh{6.\  } & \mytextarabic{يَوْمَئِذٍۢ يَصْدُرُ ٱلنَّاسُ أَشْتَاتًۭا لِّيُرَوْا۟ أَعْمَـٰلَهُمْ ﴿٦﴾}\\
\textamh{7.\  } & \mytextarabic{فَمَن يَعْمَلْ مِثْقَالَ ذَرَّةٍ خَيْرًۭا يَرَهُۥ ﴿٧﴾}\\
\textamh{8.\  } & \mytextarabic{وَمَن يَعْمَلْ مِثْقَالَ ذَرَّةٍۢ شَرًّۭا يَرَهُۥ ﴿٨﴾}\\
\end{longtable}
